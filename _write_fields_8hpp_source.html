126 template< typename Solver, typename Species >
127 class WriteFields<FieldTmpOperation<Solver, Species> >
128 {
129 public:
130  /*
131  * This is only a wrapper function to allow disable nvcc warnings.
132  * Warning: calling a __host__ function from __host__ __device__
133  * function.
134  * Use of PMACC_NO_NVCC_HDWARNING is not possible if we call a virtual
135  * method inside of the method were we disable the warnings.
136  * Therefore we create this method and call a new method were we can
137  * call virtual functions.
138  */
139  PMACC_NO_NVCC_HDWARNING
140  HDINLINE void operator()(ThreadParams* tparam)
141  {
142  this->operator_impl(tparam);
143  }
144 
145 private:
146  typedef typename FieldTmp::ValueType ValueType;
147 
150  static std::string getName()
151  {
152  return FieldTmpOperation<Solver, Species>::getName();
153  }
154 
156  static std::vector<float_64> getUnit()
157  {
158  typedef typename FieldTmp::UnitValueType UnitType;
159  UnitType unit = FieldTmp::getUnit<Solver>();
160  const uint32_t components = GetNComponents<ValueType>::value;
161  return CreateUnit::createUnit(unit, components);
162  }
163 
164  HINLINE void operator_impl(ThreadParams* params)
165  {
166  DataConnector &dc = Environment<>::get().DataConnector();
167 
168  /*## update field ##*/
169 
170  /*load FieldTmp without copy data to host*/
171  PMACC_CASSERT_MSG(
172  _please_allocate_at_least_one_FieldTmp_in_memory_param,
173  fieldTmpNumSlots > 0
174  );
175  auto fieldTmp = dc.get< FieldTmp >( FieldTmp::getUniqueId( 0 ), true );
176  /*load particle without copy particle data to host*/
177  auto speciesTmp = dc.get< Species >( Species::FrameType::getName(), true );
178 
179  fieldTmp->getGridBuffer().getDeviceBuffer().setValue(ValueType::create(0.0));
180  /*run algorithm*/
181  fieldTmp->template computeValue< CORE + BORDER, Solver >(*speciesTmp, params->currentStep);
182 
183  EventTask fieldTmpEvent = fieldTmp->asyncCommunication(__getTransactionEvent());
184  __setTransactionEvent(fieldTmpEvent);
185  /* copy data to host that we can write same to disk*/
186  fieldTmp->getGridBuffer().deviceToHost();
187  dc.releaseData( Species::FrameType::getName() );
188  /*## finish update field ##*/
189 
190  /*wrap in a one-component vector for writeField API*/
191  const traits::FieldPosition<typename fields::Solver::NummericalCellType, FieldTmp>
192  fieldPos;
193 
194  std::vector<std::vector<float_X> > inCellPosition;
195  std::vector<float_X> inCellPositonComponent;
196  for( uint32_t d = 0; d < simDim; ++d )
197  inCellPositonComponent.push_back( fieldPos()[0][d] );
198  inCellPosition.push_back( inCellPositonComponent );
199 
202  const float_X timeOffset = 0.0;
203 
204  params->gridLayout = fieldTmp->getGridLayout();
205  /*write data to HDF5 file*/
206  Field::writeField(params,
207  getName(),
208  getUnit(),
209  FieldTmp::getUnitDimension<Solver>(),
210  inCellPosition,
211  timeOffset,
212  fieldTmp->getHostDataBox(),
213  ValueType());
214 
215  dc.releaseData( FieldTmp::getUniqueId( 0 ) );
216 
217  }
