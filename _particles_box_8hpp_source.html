93  DINLINE FramePtr getEmptyFrame( )
94  {
95  FrameType* tmp = nullptr;
96  const int maxTries = 13; //magic number is not performance critical
97  for ( int numTries = 0; numTries < maxTries; ++numTries )
98  {
99 #if( PMACC_CUDA_ENABLED == 1 )
100  tmp = (FrameType*) m_deviceHeapHandle.malloc( sizeof (FrameType) );
101 #else
102  tmp = new FrameType;
103 #endif
104  if ( tmp != nullptr )
105  {
106  /* disable all particles since we can not assume that newly allocated memory contains zeros */
107  for ( int i = 0; i < (int) math::CT::volume<typename FrameType::SuperCellSize>::type::value; ++i )
108  ( *tmp )[i][multiMask_] = 0;
109 #if( PMACC_CUDA_ENABLED == 1 )
110  /* takes care that changed values are visible to all threads inside this block*/
111  __threadfence_block( );
112 #endif
113  break;
114  }
115  else
116  {
117  printf( "%s: mallocMC out of memory (try %i of %i)\n",
118  (numTries + 1) == maxTries ? "ERROR" : "WARNING",
119  numTries + 1,
120  maxTries );
121  }
122  }
123 
124  return FramePtr( tmp );
125  }

307  DINLINE bool removeLastFrame( const DataSpace<DIM> &idx )
308  {
310  FrameType** lastFrameNativPtr = &(getSuperCell( idx ).lastFramePtr);
311 
312  FramePtr last( *lastFrameNativPtr );
313  if ( last.isValid( ) )
314  {
315  FramePtr prev( last->previousFrame );
316 
317  if ( prev.isValid( ) )
318  {
319  prev->nextFrame = FramePtr( ); //set to invalid frame
320  *lastFrameNativPtr = prev.ptr; //set new last frame
321  removeFrame( last );
322  return true;
323  }
324  //remove last frame of supercell
325  getSuperCell( idx ).firstFramePtr = nullptr;
326  getSuperCell( idx ).lastFramePtr = nullptr;
327 
328  removeFrame( last );
329  }
330  return false;
331  }
