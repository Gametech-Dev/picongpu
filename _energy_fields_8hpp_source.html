200  void getEnergyFields(uint32_t currentStep)
201  {
202  DataConnector &dc = Environment<>::get().DataConnector();
203 
204  auto fieldE = dc.get< FieldE >( FieldE::getName(), true );
205  auto fieldB = dc.get< FieldB >( FieldB::getName(), true );
206 
207  /* idx == 0 -> fieldB
208  * idx == 1 -> fieldE
209  */
210  EneVectorType globalFieldEnergy[2];
211  globalFieldEnergy[0]=EneVectorType::create(0.0);
212  globalFieldEnergy[1]=EneVectorType::create(0.0);
213 
214  EneVectorType localReducedFieldEnergy[2];
215  localReducedFieldEnergy[0] = reduceField(fieldB);
216  localReducedFieldEnergy[1] = reduceField(fieldE);
217 
218  mpiReduce(nvidia::functors::Add(),
219  globalFieldEnergy,
220  localReducedFieldEnergy,
221  2,
222  mpi::reduceMethods::Reduce());
223 
224  float_64 energyFieldBReduced=0.0;
225  float_64 energyFieldEReduced=0.0;
226 
227  for(int d=0; d<FieldB::numComponents; ++d)
228  {
229  /* B field convert */
230  globalFieldEnergy[0][d] *= float_64(0.5 / MUE0 * CELL_VOLUME);
231  /* E field convert */
232  globalFieldEnergy[1][d] *= float_64(EPS0 * CELL_VOLUME * 0.5);
233 
234  /* add all to one */
235  energyFieldBReduced+= globalFieldEnergy[0][d];
236  energyFieldEReduced+= globalFieldEnergy[1][d];
237  }
238 
239  float_64 globalEnergy = energyFieldEReduced + energyFieldBReduced;
240 
241 
242  if (writeToFile)
243  {
244  using dbl = std::numeric_limits<float_64>;
245 
246  outFile.precision(dbl::digits10);
247  outFile << currentStep << " " << std::scientific << globalEnergy * UNIT_ENERGY << " "
248  << (globalFieldEnergy[0] * UNIT_ENERGY).toString(" ","") << " "
249  << (globalFieldEnergy[1] * UNIT_ENERGY).toString(" ","") << std::endl;
250  }
251  }
252 
253 private:
254 
255  template<typename T_Field>
256  EneVectorType reduceField( std::shared_ptr< T_Field > field )
257  {
258  /*define stacked DataBox's for reduce algorithm*/
259  typedef DataBoxUnaryTransform<typename T_Field::DataBoxType, energyFields::squareComponentWise > TransformedBox;
260  typedef DataBoxUnaryTransform<TransformedBox, energyFields::cast64Bit > Box64bit;
261  using D1Box = DataBoxDim1Access<Box64bit>;
262 
263  /* reduce field E*/
264  DataSpace<simDim> fieldSize = field->getGridLayout().getDataSpaceWithoutGuarding();
265  DataSpace<simDim> fieldGuard = field->getGridLayout().getGuard();
266 
267  TransformedBox fieldTransform(field->getDeviceDataBox().shift(fieldGuard));
268  Box64bit field64bit(fieldTransform);
269  D1Box d1Access(field64bit, fieldSize);
270 
271  EneVectorType fieldEnergyReduced = (*localReduce)(nvidia::functors::Add(),
272  d1Access,
273  fieldSize.productOfComponents());
274 
275  return fieldEnergyReduced;
276  }
