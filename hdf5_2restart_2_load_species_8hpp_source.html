96  HINLINE void operator()(ThreadParams* params, const uint32_t restartChunkSize)
97  {
98  std::string const speciesName = FrameType::getName();
99  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) load species: %1%") % speciesName;
100  DataConnector &dc = Environment<>::get().DataConnector();
101  GridController<simDim> &gc = Environment<simDim>::get().GridController();
102 
103  const std::string speciesSubGroup(
104  std::string("particles/") + speciesName + std::string("/")
105  );
106  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
107  const pmacc::Selection<simDim>& globalDomain = Environment<simDim>::get().SubGrid().getGlobalDomain();
108 
109  // load particle without copying particle data to host
110  auto speciesTmp = dc.get< ThisSpecies >( FrameType::getName(), true );
111 
112  // count total number of particles on the device
113  uint64_cu totalNumParticles = 0;
114  uint64_t particleOffset = 0;
115 
116  // load particle patches offsets to find own patch
117  const std::string particlePatchesPath(
118  speciesSubGroup + std::string("particlePatches/")
119  );
120 
121  // avoid deadlock between not finished pmacc tasks and mpi calls in splash/HDF5
122  __getTransactionEvent().waitForFinished();
123 
124  // read particle patches
125  openPMD::PatchReader patchReader;
126 
127  picongpu::openPMD::ParticlePatches particlePatches(
128  patchReader(
129  params->dataCollector,
130  gc.getGlobalSize(),
131  simDim,
132  params->currentStep,
133  particlePatchesPath
134  )
135  );
136 
147  const DataSpace<simDim> patchOffset =
148  globalDomain.offset +
149  params->window.globalDimensions.offset +
150  params->window.localDimensions.offset;
151  const DataSpace<simDim> patchExtent =
152  params->window.localDimensions.size;
153 
154  for( size_t i = 0; i < gc.getGlobalSize(); ++i )
155  {
156  bool exactlyMyPatch = true;
157 
158  for( uint32_t d = 0; d < simDim; ++d )
159  {
160  if( particlePatches.getOffsetComp( d )[ i ] != (uint64_t)patchOffset[ d ] )
161  exactlyMyPatch = false;
162  if( particlePatches.getExtentComp( d )[ i ] != (uint64_t)patchExtent[ d ] )
163  exactlyMyPatch = false;
164  }
165 
166  if( exactlyMyPatch )
167  {
168  totalNumParticles = particlePatches.numParticles[ i ];
169  particleOffset = particlePatches.numParticlesOffset[ i ];
170  break;
171  }
172  }
173 
174  log<picLog::INPUT_OUTPUT > ("Loading %1% particles from offset %2%") %
175  (long long unsigned) totalNumParticles % (long long unsigned) particleOffset;
176 
177  Hdf5FrameType hostFrame;
178  log<picLog::INPUT_OUTPUT > ("HDF5: malloc mapped memory: %1%") % speciesName;
179  /*malloc mapped memory*/
180  ForEach<typename Hdf5FrameType::ValueTypeSeq, MallocMemory<bmpl::_1> > mallocMem;
181  mallocMem(forward(hostFrame), totalNumParticles);
182 
183  log<picLog::INPUT_OUTPUT > ("HDF5: get mapped memory device pointer: %1%") % speciesName;
184  /*load device pointer of mapped memory*/
185  Hdf5FrameType deviceFrame;
186  ForEach<typename Hdf5FrameType::ValueTypeSeq, GetDevicePtr<bmpl::_1> > getDevicePtr;
187  getDevicePtr(forward(deviceFrame), forward(hostFrame));
188 
189  ForEach<typename Hdf5FrameType::ValueTypeSeq, LoadParticleAttributesFromHDF5<bmpl::_1> > loadAttributes;
190  loadAttributes(forward(params), forward(hostFrame), speciesSubGroup, particleOffset, totalNumParticles);
191 
192  if (totalNumParticles != 0)
193  {
194  pmacc::particles::operations::splitIntoListOfFrames(
195  *speciesTmp,
196  deviceFrame,
197  totalNumParticles,
198  restartChunkSize,
199  globalDomain.offset + localDomain.offset,
200  totalCellIdx_,
201  *(params->cellDescription),
202  picLog::INPUT_OUTPUT()
203  );
204 
205  /*free host memory*/
206  ForEach<typename Hdf5FrameType::ValueTypeSeq, FreeMemory<bmpl::_1> > freeMem;
207  freeMem(forward(hostFrame));
208  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) load species: %1%") % speciesName;
209  }
210  }
211 };
