54 
55  log<picLog::INPUT_OUTPUT> ("ADIOS: prepare write %1%D scalars: %2%") % simDim % name;
56 
57  params.adiosGroupSize += sizeof(T_Scalar);
58  if(!attrName.empty())
59  params.adiosGroupSize += sizeof(T_Attribute);
60 
61  // Size over all processes

114 template<typename T_Scalar, typename T_Attribute = uint64_t>
115 struct ReadNDScalars
116 {
118  void operator()(ThreadParams& params,
119  const std::string& name, T_Scalar* value,
120  const std::string& attrName = "", T_Attribute* attribute = nullptr)
121  {
122  log<picLog::INPUT_OUTPUT> ("ADIOS: read %1%D scalars: %2%") % simDim % name;
123  std::string datasetName = params.adiosBasePath + name;
124 
125  ADIOS_VARINFO* varInfo;
126  ADIOS_CMD_EXPECT_NONNULL( varInfo = adios_inq_var(params.fp, datasetName.c_str()) );
127  if(varInfo->ndim != simDim)
128  throw std::runtime_error(std::string("Invalid dimensionality for ") + name);
129  if(varInfo->type != traits::PICToAdios<T_Scalar>().type)
130  throw std::runtime_error(std::string("Invalid type for ") + name);
131 
132  DataSpace<simDim> gridPos = Environment<simDim>::get().GridController().getPosition();
133  uint64_t start[varInfo->ndim];
134  uint64_t count[varInfo->ndim];
135  for(int d = 0; d < varInfo->ndim; ++d)
136  {
137  /* \see adios_define_var: z,y,x in C-order */
138  start[d] = gridPos.revert()[d];
139  count[d] = 1;
140  }
141 
142  ADIOS_SELECTION* fSel = adios_selection_boundingbox(varInfo->ndim, start, count);
143 
144  // avoid deadlock between not finished pmacc tasks and mpi calls in adios
145  __getTransactionEvent().waitForFinished();
146 
147  /* specify what we want to read, but start reading at below at `adios_perform_reads` */
148  /* magic parameters (0, 1): `from_step` (not used in streams), `nsteps` to read (must be 1 for stream) */
149  log<picLog::INPUT_OUTPUT > ("ADIOS: Schedule read skalar %1%)") % datasetName;
150  ADIOS_CMD( adios_schedule_read(params.fp, fSel, datasetName.c_str(), 0, 1, (void*)value) );
151 
152  /* start a blocking read of all scheduled variables */
153  ADIOS_CMD( adios_perform_reads(params.fp, 1) );
154 
155  adios_selection_delete(fSel);
156  adios_free_varinfo(varInfo);
157 
158  if(!attrName.empty())
159  {
160  log<picLog::INPUT_OUTPUT> ("ADIOS: read attribute %1% for scalars: %2%") % attrName % name;
161  *attribute = readAttribute<T_Attribute>(params.fp, datasetName, attrName);
162  }
163  }
164 };
