50 template<typename T_Shape, typename T_pos>
51 struct EvalAssignmentFunction
52 {
53  using ParticleAssign = typename T_Shape::ChargeAssignmentOnSupport;
54 
55  HDINLINE float_X
56  operator()(const float_X parPos)
57  {
58  ParticleAssign shape;
59  const float_X gridPos=T_pos::value;
60  return shape(gridPos-parPos);
61  }
