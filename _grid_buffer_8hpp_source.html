197  void addExchange(uint32_t dataPlace, const Mask &receive, DataSpace<DIM> guardingCells, uint32_t communicationTag, bool sizeOnDeviceSend, bool sizeOnDeviceReceive )
198  {
199 
200  if (hasOneExchange && (communicationTag != lastUsedCommunicationTag))
201  throw std::runtime_error("It is not allowed to give the same GridBuffer different communicationTags");
202 
203  lastUsedCommunicationTag = communicationTag;
204 
205  receiveMask = receiveMask + receive;
206  sendMask = this->receiveMask.getMirroredMask();
207  Mask send = receive.getMirroredMask();
208 
209 
210 
211  for (uint32_t ex = 1; ex< -12 * (int) DIM + 6 * (int) DIM * (int) DIM + 9; ++ex)
212  {
213  if (send.isSet(ex))
214  {
215  uint32_t uniqCommunicationTag = (communicationTag << 5) | ex;
216 
217  if (!hasOneExchange && !privateGridBuffer::UniquTag::getInstance().isTagUniqu(uniqCommunicationTag))
218  {
219  std::stringstream message;
220  message << "unique exchange communication tag ("
221  << uniqCommunicationTag << ") witch is created from communicationTag ("
222  << communicationTag << ") already used for other GridBuffer exchange";
223  throw std::runtime_error(message.str());
224  }
225  hasOneExchange = true;
226 
227  if (sendExchanges[ex] != nullptr)
228  {
229  throw std::runtime_error("Exchange already added!");
230  }
231 
232  maxExchange = std::max(maxExchange, ex + 1u);
233  sendExchanges[ex] = new ExchangeIntern<BORDERTYPE, DIM > (this->getDeviceBuffer(), gridLayout, guardingCells,
234  (ExchangeType) ex, uniqCommunicationTag,
235  dataPlace == GUARD ? BORDER : GUARD, sizeOnDeviceSend);
236  ExchangeType recvex = Mask::getMirroredExchangeType(ex);
237  maxExchange = std::max(maxExchange, recvex + 1u);
238  receiveExchanges[recvex] =
239  new ExchangeIntern<BORDERTYPE, DIM > (
240  this->getDeviceBuffer(),
241  gridLayout,
242  guardingCells,
243  recvex,
244  uniqCommunicationTag,
245  dataPlace == GUARD ? GUARD : BORDER,
246  sizeOnDeviceReceive);
247  }
248  }
249  }
