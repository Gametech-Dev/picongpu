95  void pluginLoad()
96  {
97 
98  if (0 != notifyPeriod.size())
99  {
100  if (0 != slicePoints.size() &&
101  0 != axis.size())
102  {
103  for (int i = 0; i < (int) slicePoints.size(); ++i)
104  {
105  std::string period = getValue(notifyPeriod, i);
106  if(!period.empty())
107  {
108 
109  if (getValue(axis, i).length() == 2u)
110  {
111  std::stringstream o_slicePoint;
112  o_slicePoint << getValue(slicePoints, i);
113  /*add default value for folder*/
114  if (folders.empty())
115  {
116  folders.push_back(std::string("."));
117  }
118  std::string filename(pluginPrefix + "_" + getValue(axis, i) + "_" + o_slicePoint.str());
119  typename VisType::CreatorType pngCreator(filename, getValue(folders, i));
126  DataSpace<DIM2 > transpose(
127  charToAxisNumber(getValue(axis, i)[0]),
128  charToAxisNumber(getValue(axis, i)[1])
129  );
130  /* if simulation run in 2D ignore all xz, yz slices (we had no z direction)*/
131  const bool isAllowed2DSlice= (simDim==DIM3) || (transpose.x()!=2 && transpose.y()!=2);
132  const bool isSlidingWindowActive=MovingWindow::getInstance().isSlidingWindowActive();
133  /* if sliding window is active we are not allowed to create pngs from xz slice
134  * This means one dimension in transpose must contain 1 (y direction)
135  */
136  const bool isAllowedMovingWindowSlice=!isSlidingWindowActive ||
137  (transpose.x()==1 || transpose.y()==1);
138  if( isAllowed2DSlice && isAllowedMovingWindowSlice )
139  {
140  VisType* tmp = new VisType(pluginName, pngCreator, period, transpose, getValue(slicePoints, i));
141  visIO.push_back(tmp);
142  tmp->setMappingDescription(cellDescription);
143  tmp->init();
144  }
145  else
146  {
147  if(!isAllowedMovingWindowSlice)
148  std::cerr << "[WARNING] You are running a simulation with moving window: png output along the axis "<<
149  getValue(axis, i) << " will be ignored" << std::endl;
150  if(!isAllowed2DSlice)
151  std::cerr << "[WARNING] You are running a 2D simulation: png output along the axis "<<
152  getValue(axis, i) << " will be ignored" << std::endl;
153  }
154  }
155  else
156  throw std::runtime_error((std::string("[Png Plugin] wrong charecter count in axis: ") + getValue(axis, i)).c_str());
157  }
158  }
159  }
160  else
161  {
162  throw std::runtime_error("[Png Plugin] One parameter is missing");
163  }
164  }
