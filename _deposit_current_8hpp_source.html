149  template<
150  typename CursorJ,
151  typename T_Line,
152  typename T_Acc
153  >
154  DINLINE void
155  cptCurrent1D(
156  T_Acc const & acc,
157  CursorJ cursorJ,
158  const T_Line& line,
159  const float_X currentSurfaceDensity
160  ) const
161  {
162  if( line.m_pos0[2] == line.m_pos1[2] )
163  return;
164  /* pick every cell in the xy-plane that is overlapped by particle's
165  * form factor and deposit the current for the cells above and beneath
166  * that cell and for the cell itself.
167  */
168  for( int i = T_begin ; i < T_end ; ++i )
169  {
170  const float_X s0i = this->S0( line, i, 0 );
171  const float_X dsi = this->S1( line, i, 0 ) - s0i;
172  for( int j = T_begin ; j < T_end ; ++j )
173  {
174  const float_X s0j = this->S0( line, j, 1 );
175  const float_X dsj = this->S1( line, j, 1 ) - s0j;
176 
177  float_X tmp =
178  -currentSurfaceDensity * (
179  s0i * s0j +
180  float_X( 0.5 ) * ( dsi * s0j + s0i * dsj ) +
181  ( float_X( 1.0 ) / float_X( 3.0 ) ) * dsj * dsi
182  );
183 
184  float_X accumulated_J = float_X( 0.0 );
185  for( int k = T_begin ; k < T_end - 1 ; ++k )
186  {
187  /* This is the implementation of the FORTRAN W(i,j,k,3)/ C style W(i,j,k,2) version from
188  * Esirkepov paper. All coordinates are rotated before thus we can
189  * always use C style W(i,j,k,2).
190  */
191  const float_X W = this->DS( line, k, 2 ) * tmp;
192  accumulated_J += W;
193  atomicAdd(
194  &( (*cursorJ( i, j, k ) ).z( ) ),
195  accumulated_J,
196  ::alpaka::hierarchy::Threads{}
197  );
198  }
199  }
200  }
201  }
202  };

324  {
325  const float_X s0j = this->S0( line, j, 1 );
326  const float_X dsj = this->S1( line, j, 1 ) - s0j;
327  for( int i = T_begin; i < T_end; ++i )
328  {
329  const float_X s0i = this->S0( line, i, 0 );
330  const float_X dsi = this->S1( line, i, 0 ) - s0i;
331  float_X W = s0i * this->S0( line, j, 1 ) +
332  float_X( 0.5 ) * ( dsi * s0j + s0i * dsj ) +
333  ( float_X( 1.0 ) / float_X( 3.0 ) ) * dsi * dsj;
334 
335  const float_X j_z = W * currentSurfaceDensityZ;
336  atomicAdd(
337  &( ( *cursorJ( i, j ) ).z( ) ),
338  j_z,
339  ::alpaka::hierarchy::Threads{}
