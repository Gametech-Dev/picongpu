228  /* Functor implementation
229  *
230  * Ionization model specific particle creation
231  *
232  * \tparam T_parentIon type of ion species that is being ionized
233  * \tparam T_childElectron type of electron species that is created
234  * \param parentIon ion instance that is ionized
235  * \param childElectron electron instance that is created
236  */
237  template<typename T_parentIon, typename T_childElectron, typename T_Acc>
238  DINLINE void operator()(T_Acc const & acc, T_parentIon& parentIon,T_childElectron& childElectron)
239  {
240 
241  /* for not mixing operations::assign up with the nvidia functor assign */
242  namespace partOp = pmacc::particles::operations;
243  /* each thread sets the multiMask hard on "particle" (=1) */
244  childElectron[multiMask_] = 1u;
245  const float_X weighting = parentIon[weighting_];
246 
247  /* each thread initializes a clone of the parent ion but leaving out
248  * some attributes:
249  * - multiMask: reading from global memory takes longer than just setting it again explicitly
250  * - momentum: because the electron would get a higher energy because of the ion mass
251  * - boundElectrons: because species other than ions or atoms do not have them
252  * (gets AUTOMATICALLY deselected because electrons do not have this attribute)
253  */
254  auto targetElectronClone = partOp::deselect<bmpl::vector2<multiMask, momentum> >(childElectron);
255 
256  partOp::assign(targetElectronClone, partOp::deselect<particleId>(parentIon));
257 
258  const float_X massIon = attribute::getMass(weighting,parentIon);
259  const float_X massElectron = attribute::getMass(weighting,childElectron);
260 
261  const float3_X electronMomentum (parentIon[momentum_]*(massElectron/massIon));
262 
263  childElectron[momentum_] = electronMomentum;
264 
265  /* conservation of momentum
266  * \todo add conservation of mass */
267  parentIon[momentum_] -= electronMomentum;
268 
275  parentIon[boundElectrons_] -= float_X(1.);
276  }
