290 struct PushAllSpecies
291 {
298  HINLINE void operator()(
299  const uint32_t currentStep,
300  const EventTask& eventInt,
301  EventTask& pushEvent,
302  EventTask& commEvent
303  ) const
304  {
305  using EventList = std::list<EventTask>;
306  EventList updateEventList;
307  EventList commEventList;
308 
309  /* push all species */
310  typedef typename pmacc::particles::traits::FilterByFlag
311  <
312  VectorAllSpecies,
313  particlePusher<>
314  >::type VectorSpeciesWithPusher;
315  ForEach< VectorSpeciesWithPusher, particles::PushSpecies< bmpl::_1 > > pushSpecies;
316  pushSpecies( currentStep, eventInt, forward(updateEventList) );
317 
318  /* join all push events */
319  for (typename EventList::iterator iter = updateEventList.begin();
320  iter != updateEventList.end();
321  ++iter)
322  {
323  pushEvent += *iter;
324  }
325 
326  /* call communication for all species */
327  ForEach< VectorSpeciesWithPusher, particles::CommunicateSpecies< bmpl::_1> > communicateSpecies;
328  communicateSpecies( forward(updateEventList), forward(commEventList) );
329 
330  /* join all communication events */
331  for (typename EventList::iterator iter = commEventList.begin();
332  iter != commEventList.end();
333  ++iter)
334  {
335  commEvent += *iter;
336  }
337  }

349  VectorAllSpecies,

352  using SelectIonizer = T_SelectIonizer;

355  /* define the type of the species to be created
356  * from inside the ionization model specialization
357  */
358  using DestSpecies = typename SelectIonizer::DestSpecies;
359  using DestFrameType = typename DestSpecies::FrameType;
360 
368  template<typename T_CellDescription>
369  HINLINE void operator()(
370  T_CellDescription* cellDesc,
371  const uint32_t currentStep
372  ) const
373  {
374  DataConnector &dc = Environment<>::get().DataConnector();
375 
376  // alias for pointer on source species
377  auto srcSpeciesPtr = dc.get< SpeciesType >( FrameType::getName(), true );
378  // alias for pointer on destination species
379  auto electronsPtr = dc.get< DestSpecies >( DestFrameType::getName(), true );
380 
381  SelectIonizer selectIonizer(currentStep);
382 
383  creation::createParticlesFromSpecies(*srcSpeciesPtr, *electronsPtr, selectIonizer, cellDesc);
384 
385  /* fill the gaps in the created species' particle frames to ensure that only
386  * the last frame is not completely filled but every other before is full
387  */
388  electronsPtr->fillAllGaps();
389 
390  dc.releaseData( FrameType::getName() );
391  dc.releaseData( DestFrameType::getName() );
392 
393  }
