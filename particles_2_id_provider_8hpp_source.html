121  template<unsigned T_dim>
122  bool IdProvider<T_dim>::isOverflown()
123  {
124  State curState = getState();
125  /* Overflow happens, when an id puts bits into the bits used for ensuring uniqueness.
126  * This are the n upper bits with n = highest bit set in the maximum id (which is maxNumProc_ - 1)
127  * when counting the bits from 1 = right most bit
128  * So first we calculate n, then remove the lowest bits of the next id so we have only the n upper bits
129  * If any of them is non-zero, it is an overflow and we can have duplicate ids.
130  * If not, then all ids are probably unique (still a chance, the id is overflown so much, that detection is impossible)
131  */
132  uint64_t tmp = curState.maxNumProc - 1;
133  int32_t bitsToCheck = 0;
134  while(tmp)
135  {
136  bitsToCheck++;
137  tmp >>= 1;
138  }
139 
140  // Number of bits in the ids
141  static constexpr int32_t numBitsOfType = sizeof(curState.maxNumProc) * CHAR_BIT;
142 
143  // Get current id
144  uint64_t nextId = curState.nextId;
145  // Cancel out start id via xor -> Upper n bits should be 0
146  nextId ^= curState.startId;
147  /* Prepare to compare only upper n bits for 0
148  * Example: maxNumProc_ has 3 set bits (<8 ranks), 64bit value used
149  * --> Shift by 61 bits
150  * => 3 upper bits are left untouched (besides moving), rest is zero
151  */
152  nextId >>= numBitsOfType - bitsToCheck;
153 
154  return nextId != 0;
155  }

157  template<unsigned T_dim>
158  uint64_t IdProvider<T_dim>::calcStartId()
159  {
160  uint64_t rank = Environment<T_dim>::get().GridController().getScalarPosition();
161 
162  /* We put the rank into the upper bits to have the lower bits for counting up and still
163  * getting unique numbers. Reversing the bits instead of shifting gives some more room
164  * as the upper bits of the rank are often also zero
165  * Note: Overflow detection will still return true for that case
166  */
167  return reverseBits(rank);
168  }
