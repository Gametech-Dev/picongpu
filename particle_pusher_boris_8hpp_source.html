33 struct Push
34 {
35  /* this is an optional extension for sub-sampling pushes that enables grid to particle interpolation
36  * for particle positions outside the super cell in one push
37  */
38  typedef typename pmacc::math::CT::make_Int<simDim,0>::type LowerMargin;
39  typedef typename pmacc::math::CT::make_Int<simDim,0>::type UpperMargin;

41  template< typename T_FunctorFieldE, typename T_FunctorFieldB, typename T_Particle, typename T_Pos >
42  HDINLINE void operator()(
43  const T_FunctorFieldB functorBField,
44  const T_FunctorFieldE functorEField,
45  T_Particle & particle,
46  T_Pos & pos,
47  const uint32_t
48  )
49  {
50  float_X const weighting = particle[ weighting_ ];
51  float_X const mass = attribute::getMass( weighting, particle );
52  float_X const charge = attribute::getCharge( weighting, particle );
53 
54  using MomType = momentum::type;
55  MomType const mom = particle[ momentum_ ];
56 
57  auto bField = functorBField(pos);
58  auto eField = functorEField(pos);
59 
60  const float_X QoM = charge / mass;
61 
62  const float_X deltaT = DELTA_T;
63 
64  const MomType mom_minus = mom + float_X(0.5) * charge * eField * deltaT;
65 
66  Gamma gamma;
67  const float_X gamma_reci = float_X(1.0) / gamma(mom_minus, mass);
68  const float3_X t = float_X(0.5) * QoM * bField * gamma_reci * deltaT;
69  auto s = float_X(2.0) * t * (float_X(1.0) / (float_X(1.0) + math::abs2(t)));
70 
71  const MomType mom_prime = mom_minus + math::cross(mom_minus, t);
72  const MomType mom_plus = mom_minus + math::cross(mom_prime, s);
73 
74  const MomType new_mom = mom_plus + float_X(0.5) * charge * eField * deltaT;
75 
76  particle[ momentum_ ] = new_mom;
77 
78  Velocity velocity;
79  const float3_X vel = velocity(new_mom, mass);
80 
81  for(uint32_t d=0;d<simDim;++d)
82  {
83  pos[d] += (vel[d] * deltaT) / cellSize[d];
84  }
85 
86  }
