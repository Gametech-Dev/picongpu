77  virtual void restart(uint32_t restartStep, const std::string restartDirectory)
78  {
79  // restart simulation by loading from persistent data
80  // the simulation will start after restartStep
81  log<picLog::SIMULATION_STATE > ("Restarting simulation from timestep %1% in directory '%2%'") %
82  restartStep % restartDirectory;
83 
84  Environment<>::get().PluginConnector().restartPlugins(restartStep, restartDirectory);
85  __getTransactionEvent().waitForFinished();
86 
87  CUDA_CHECK(cudaDeviceSynchronize());
88  CUDA_CHECK(cudaGetLastError());
89 
90  GridController<simDim> &gc = Environment<simDim>::get().GridController();
91 
92  // avoid deadlock between not finished pmacc tasks and MPI_Barrier
93  __getTransactionEvent().waitForFinished();
94  /* can be spared for better scalings, but guarantees the user
95  * that the restart was successful */
96  MPI_CHECK(MPI_Barrier(gc.getCommunicator().getMPIComm()));
97 
98  log<picLog::SIMULATION_STATE > ("Loading from persistent data finished");
99  }
