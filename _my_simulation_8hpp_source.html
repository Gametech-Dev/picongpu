154  virtual void pluginLoad()
155  {
156  //fill periodic with 0
157  while (periodic.size() < 3)
158  periodic.push_back(0);
159 
160  // check on correct number of devices. fill with default value 1 for missing dimensions
161  if (devices.size() > 3)
162  {
163  std::cerr << "Invalid number of devices.\nuse [-d dx=1 dy=1 dz=1]" << std::endl;
164  }
165  else
166  while (devices.size() < 3)
167  devices.push_back(1);
168 
169  // check on correct grid size. fill with default grid size value 1 for missing 3. dimension
170  if (gridSize.size() < 2 || gridSize.size() > 3)
171  {
172  std::cerr << "Invalid or missing grid size.\nuse -g width height [depth=1]" << std::endl;
173  }
174  else
175  if (gridSize.size() == 2)
176  gridSize.push_back(1);
177 
178  if (slidingWindow && devices[1] == 1)
179  {
180  std::cerr << "Invalid configuration. Can't use moving window with one device in Y direction" << std::endl;
181  }
182 
183  DataSpace<simDim> global_grid_size;
184  DataSpace<simDim> gpus;
185  DataSpace<simDim> isPeriodic;
186 
187  for (uint32_t i = 0; i < simDim; ++i)
188  {
189  global_grid_size[i] = gridSize[i];
190  gpus[i] = devices[i];
191  isPeriodic[i] = periodic[i];
192  }
193 
194  Environment<simDim>::get().initDevices(gpus, isPeriodic);
195  pmacc::GridController< simDim > & gc = pmacc::Environment<simDim>::get().GridController();
196 
197  DataSpace<simDim> myGPUpos(gc.getPosition());
198 
199  if( gc.getGlobalRank() == 0 )
200  {
201  if( showVersionOnce )
202  {
203  void( getSoftwareVersions( std::cout ) );
204  }
205  }
206 
207  // calculate the number of local grid cells and
208  // the local cell offset to the global box
209  for (uint32_t dim = 0; dim < gridDistribution.size() && dim < simDim; ++dim)
210  {
211  // parse string
212  ParserGridDistribution parserGD(gridDistribution.at(dim));
213 
214  // calculate local grid points & offset
215  gridSizeLocal[dim] = parserGD.getLocalSize(myGPUpos[dim]);
216  gridOffset[dim] = parserGD.getOffset(myGPUpos[dim], global_grid_size[dim]);
217  }
218  // by default: use an equal distributed box for all omitted params
219  for (uint32_t dim = gridDistribution.size(); dim < simDim; ++dim)
220  {
221  gridSizeLocal[dim] = global_grid_size[dim] / gpus[dim];
222  gridOffset[dim] = gridSizeLocal[dim] * myGPUpos[dim];
223  }
224 
225  Environment<simDim>::get().initGrids(global_grid_size, gridSizeLocal, gridOffset);
226 
227  MovingWindow::getInstance().setSlidingWindow(slidingWindow);
228 
229  log<picLog::DOMAINS > ("rank %1%; localsize %2%; localoffset %3%;") %
230  myGPUpos.toString() % gridSizeLocal.toString() % gridOffset.toString();
231 
232  SimulationHelper<simDim>::pluginLoad();
233 
234  GridLayout<simDim> layout(gridSizeLocal, GuardSize::toRT() * SuperCellSize::toRT());
235  cellDescription = new MappingDesc(layout.getDataSpace(), DataSpace<simDim>(GuardSize::toRT()));
236 
237  checkGridConfiguration(global_grid_size, cellDescription->getGridLayout());
238 
239  if (gc.getGlobalRank() == 0)
240  {
241  if (slidingWindow)
242  log<picLog::PHYSICS > ("Sliding Window is ON");
243  else
244  log<picLog::PHYSICS > ("Sliding Window is OFF");
245  }
246 
247  for (uint32_t i = 0; i < simDim; ++i)
248  {
249 
250  /*
251  * absorber must be smaller than local gridsize if direction is periodic.
252  * absorber can't go over more than one device.
253  */
254  if (isPeriodic[i] == 0)
255  {
256  /*negativ direction*/
257  PMACC_VERIFY((int) ABSORBER_CELLS[i][0] <= (int) cellDescription->getGridLayout().getDataSpaceWithoutGuarding()[i]);
258  /*positiv direction*/
259  PMACC_VERIFY((int) ABSORBER_CELLS[i][1] <= (int) cellDescription->getGridLayout().getDataSpaceWithoutGuarding()[i]);
260  }
261  }
262  }

291  virtual void init()
292  {
293  namespace nvmem = pmacc::nvidia::memory;
294 
295  DataConnector &dc = Environment<>::get().DataConnector();
296 
297  // create simulation data such as fields and particles
298  auto fieldB = new FieldB( *cellDescription );
299  dc.share( std::shared_ptr< ISimulationData >( fieldB ) );
300  auto fieldE = new FieldE( *cellDescription );
301  dc.share( std::shared_ptr< ISimulationData >( fieldE ) );
302  auto fieldJ = new FieldJ( *cellDescription );
303  dc.share( std::shared_ptr< ISimulationData >( fieldJ ) );
304 
305  std::vector< FieldTmp * > fieldTmp;
306  for( uint32_t slot = 0; slot < fieldTmpNumSlots; ++slot)
307  {
308  auto newFld = new FieldTmp( *cellDescription, slot );
309  fieldTmp.push_back( newFld );
310  dc.share( std::shared_ptr< ISimulationData >( newFld ) );
311  }
312  pushBGField = new cellwiseOperation::CellwiseOperation < CORE + BORDER + GUARD > (*cellDescription);
313  currentBGField = new cellwiseOperation::CellwiseOperation < CORE + BORDER > (*cellDescription);
314 
315  // Initialize random number generator and synchrotron functions, if there are synchrotron or bremsstrahlung Photons
316  typedef typename pmacc::particles::traits::FilterByFlag<VectorAllSpecies,
317  synchrotronPhotons<> >::type AllSynchrotronPhotonsSpecies;
318  typedef typename pmacc::particles::traits::FilterByFlag<VectorAllSpecies,
319  bremsstrahlungPhotons<> >::type AllBremsstrahlungPhotonsSpecies;
320 
321  // create factory for the random number generator
322  const uint32_t userSeed = random::seed::ISeed< random::SeedGenerator >{}();
323  const uint32_t seed = std::hash<std::string>{}(
324  std::to_string( userSeed )
325  );
326 
327  using RNGFactory = pmacc::random::RNGProvider< simDim, random::Generator >;
328  auto rngFactory = new RNGFactory( Environment<simDim>::get().SubGrid().getLocalDomain().size );
329  if (Environment<simDim>::get().GridController().getGlobalRank() == 0)
330  {
331  log<picLog::PHYSICS >("used Random Number Generator: %1% seed: %2%") %
332  rngFactory->getName() %
333  userSeed;
334  }
335 
336  // init and share random number generator
337  pmacc::GridController<simDim>& gridCon = pmacc::Environment<simDim>::get().GridController();
338  rngFactory->init( gridCon.getScalarPosition() ^ seed );
339  dc.share( std::shared_ptr< ISimulationData >( rngFactory ) );
340 
341  // Initialize synchrotron functions, if there are synchrotron photon species
342  if(!bmpl::empty<AllSynchrotronPhotonsSpecies>::value)
343  {
344  this->synchrotronFunctions.init();
345  }
346 #if( PMACC_CUDA_ENABLED == 1 )
347  // Initialize bremsstrahlung lookup tables, if there are species containing bremsstrahlung photons
348  if(!bmpl::empty<AllBremsstrahlungPhotonsSpecies>::value)
349  {
350  ForEach<
351  AllBremsstrahlungPhotonsSpecies,
352  particles::bremsstrahlung::FillScaledSpectrumMap< bmpl::_1 >
353  > fillScaledSpectrumMap;
354  fillScaledSpectrumMap(forward(this->scaledBremsstrahlungSpectrumMap));
355 
356  this->bremsstrahlungPhotonAngle.init();
357  }
358 
359  /* Create an empty allocator. This one is resized after all exchanges
360  * for particles are created */
361  deviceHeap.reset(new DeviceHeap(0));
362 #endif
363 
364  /* Allocate helper fields for FLYlite population kinetics for atomic physics
365  * (histograms, rate matrix, etc.)
366  */
367  using AllFlyLiteIons = typename pmacc::particles::traits::FilterByFlag<
368  VectorAllSpecies,
369  populationKinetics<>
370  >::type;
371 
372  ForEach<
373  AllFlyLiteIons,
374  particles::CallPopulationKineticsInit< bmpl::_1 >,
375  bmpl::_1
376  > initPopulationKinetics;
377  initPopulationKinetics(
378  gridSizeLocal
379  );
380 
381  // Allocate and initialize particle species with all left-over memory below
382  ForEach< VectorAllSpecies, particles::CreateSpecies<bmpl::_1> > createSpeciesMemory;
383  createSpeciesMemory( deviceHeap, cellDescription );
384 
385  size_t freeGpuMem(0);
386  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
387  if(freeGpuMem < reservedGpuMemorySize)
388  {
389  pmacc::log< picLog::MEMORY > ("%1% MiB free memory < %2% MiB required reserved memory")
390  % (freeGpuMem / 1024 / 1024) % (reservedGpuMemorySize / 1024 / 1024) ;
391  std::stringstream msg;
392  msg << "Cannot reserve "
393  << (reservedGpuMemorySize / 1024 / 1024) << " MiB as there is only "
394  << (freeGpuMem / 1024 / 1024) << " MiB free device memory left";
395  throw std::runtime_error(msg.str());
396  }
397 
398 #if( PMACC_CUDA_ENABLED == 1 )
399  size_t heapSize = freeGpuMem - reservedGpuMemorySize;
400 
401  if( Environment<>::get().MemoryInfo().isSharedMemoryPool() )
402  {
403  heapSize /= 2;
404  log<picLog::MEMORY > ("Shared RAM between GPU and host detected - using only half of the 'device' memory.");
405  }
406  else
407  log<picLog::MEMORY > ("RAM is NOT shared between GPU and host.");
408 
409  // initializing the heap for particles
410  deviceHeap->destructiveResize(heapSize);
411  MallocMCBuffer<DeviceHeap>* mallocMCBuffer = new MallocMCBuffer<DeviceHeap>(deviceHeap);
412  dc.share( std::shared_ptr< ISimulationData >( mallocMCBuffer ) );
413 #endif
414  ForEach< VectorAllSpecies, particles::LogMemoryStatisticsForSpecies<bmpl::_1> > logMemoryStatisticsForSpecies;
415  logMemoryStatisticsForSpecies( deviceHeap );
416 
417  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
418  log<picLog::MEMORY > ("free mem after all mem is allocated %1% MiB") % (freeGpuMem / 1024 / 1024);
419 
420  IdProvider<simDim>::init();
421 
422  // create field solver
423  this->myFieldSolver = new fields::Solver(*cellDescription);
424 
425  // create current interpolation
426  this->myCurrentInterpolation = new typename fields::Solver::CurrentInterpolation;
427 #if( PMACC_CUDA_ENABLED == 1 )
428  /* add CUDA streams to the StreamController for concurrent execution */
429  Environment<>::get().StreamController().addStreams(6);
430 #endif
431  }

433  virtual uint32_t fillSimulation()
434  {
435  /* assume start (restart in initialiserController might change that) */
436  uint32_t step = 0;
437 
438  /* set slideCounter properties for PIConGPU MovingWindow: assume start
439  * (restart in initialiserController might change this again)
440  */
441  MovingWindow::getInstance().setSlideCounter(0, 0);
442  /* Update MPI domain decomposition: will also update SubGrid domain
443  * information such as local offsets in y-direction
444  */
445  GridController<simDim> &gc = Environment<simDim>::get().GridController();
446  gc.setStateAfterSlides(0);
447 
448  DataConnector &dc = Environment<>::get().DataConnector();
449  auto fieldE = dc.get< FieldE >( FieldE::getName(), true );
450  auto fieldB = dc.get< FieldB >( FieldB::getName(), true );
451 
452  /* fill all objects registed in DataConnector */
453  if (initialiserController)
454  {
455  initialiserController->printInformation();
456  if (this->restartRequested)
457  {
458  /* we do not require '--checkpoint.restart.step' if a master checkpoint file is found */
459  if (this->restartStep < 0)
460  {
461  std::vector<uint32_t> checkpoints = readCheckpointMasterFile();
462 
463  if (checkpoints.empty())
464  {
465  throw std::runtime_error(
466  "Restart failed. You must provide the '--checkpoint.restart.step' argument. See picongpu --help."
467  );
468  } else
469  this->restartStep = checkpoints.back();
470  }
471 
472  initialiserController->restart((uint32_t)this->restartStep, this->restartDirectory);
473  step = this->restartStep;
474 
482  cellwiseOperation::CellwiseOperation< GUARD > guardBGField( *cellDescription );
483  namespace nvfct = pmacc::nvidia::functors;
484  guardBGField( fieldE, nvfct::Add(), FieldBackgroundE( fieldE->getUnit() ),
485  step, FieldBackgroundE::InfluenceParticlePusher );
486  guardBGField( fieldB, nvfct::Add(), FieldBackgroundB( fieldB->getUnit() ),
487  step, FieldBackgroundB::InfluenceParticlePusher );
488  }
489  else
490  {
491  initialiserController->init();
492  ForEach< particles::InitPipeline, particles::CallFunctor<bmpl::_1> > initSpecies;
493  initSpecies( step );
494  }
495  }
496 
497  size_t freeGpuMem(0u);
498  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
499  log<picLog::MEMORY > ("free mem after all particles are initialized %1% MiB") % (freeGpuMem / 1024 / 1024);
500 
501  // generate valid GUARDS (overwrite)
502  EventTask eRfieldE = fieldE->asyncCommunication(__getTransactionEvent());
503  __setTransactionEvent(eRfieldE);
504  EventTask eRfieldB = fieldB->asyncCommunication(__getTransactionEvent());
505  __setTransactionEvent(eRfieldB);
506 
507  dc.releaseData( FieldE::getName() );
508  dc.releaseData( FieldB::getName() );
509 
510  return step;
511  }

518  virtual void runOneStep(uint32_t currentStep)
519  {
520  namespace nvfct = pmacc::nvidia::functors;
521 
522  typedef typename pmacc::particles::traits::FilterByIdentifier
523  <
524  VectorAllSpecies,
525  momentumPrev1
526  >::type VectorSpeciesWithMementumPrev1;
527 
528  /* copy attribute momentum to momentumPrev1 */
529  ForEach<
530  VectorSpeciesWithMementumPrev1,
531  particles::Manipulate<
532  particles::manipulators::unary::CopyAttribute<
533  momentumPrev1,
534  momentum
535  >,
536  bmpl::_1
537  >
538  > copyMomentumPrev1;
539  copyMomentumPrev1( currentStep );
540 
541  DataConnector &dc = Environment<>::get().DataConnector();
542 
543  /* Initialize ionization routine for each species with the flag `ionizers<>` */
544  using VectorSpeciesWithIonizers = typename pmacc::particles::traits::FilterByFlag<
545  VectorAllSpecies,
546  ionizers<>
547  >::type;
548  ForEach< VectorSpeciesWithIonizers, particles::CallIonization< bmpl::_1 > > particleIonization;
549  particleIonization( cellDescription, currentStep );
550 
551  /* FLYlite population kinetics for atomic physics */
552  using AllFlyLiteIons = typename pmacc::particles::traits::FilterByFlag<
553  VectorAllSpecies,
554  populationKinetics<>
555  >::type;
556 
557  ForEach<
558  AllFlyLiteIons,
559  particles::CallPopulationKinetics< bmpl::_1 >,
560  bmpl::_1
561  > populationKinetics;
562  populationKinetics( currentStep );
563 
564  /* call the synchrotron radiation module for each radiating species (normally electrons) */
565  typedef typename pmacc::particles::traits::FilterByFlag<VectorAllSpecies,
566  synchrotronPhotons<> >::type AllSynchrotronPhotonsSpecies;
567 
568  ForEach<
569  AllSynchrotronPhotonsSpecies,
570  particles::CallSynchrotronPhotons< bmpl::_1 >
571  > synchrotronRadiation;
572  synchrotronRadiation( cellDescription, currentStep, this->synchrotronFunctions );
573 
574 #if( PMACC_CUDA_ENABLED == 1 )
575  /* Bremsstrahlung */
576  typedef typename pmacc::particles::traits::FilterByFlag
577  <
578  VectorAllSpecies,
579  bremsstrahlungIons<>
580  >::type VectorSpeciesWithBremsstrahlung;
581  ForEach<
582  VectorSpeciesWithBremsstrahlung,
583  particles::CallBremsstrahlung< bmpl::_1 >
584  > particleBremsstrahlung;
585  particleBremsstrahlung(
586  cellDescription,
587  currentStep,
588  this->scaledBremsstrahlungSpectrumMap,
589  this->bremsstrahlungPhotonAngle);
590 #endif
591  EventTask initEvent = __getTransactionEvent();
592  EventTask updateEvent;
593  EventTask commEvent;
594 
595  /* push all species */
596  particles::PushAllSpecies pushAllSpecies;
597  pushAllSpecies( currentStep, initEvent, updateEvent, commEvent );
598 
599  __setTransactionEvent(updateEvent);
601  auto fieldE = dc.get< FieldE >( FieldE::getName(), true );
602  auto fieldB = dc.get< FieldB >( FieldB::getName(), true );
603  (*pushBGField)(fieldE, nvfct::Sub(), FieldBackgroundE(fieldE->getUnit()),
604  currentStep, FieldBackgroundE::InfluenceParticlePusher);
605  (*pushBGField)(fieldB, nvfct::Sub(), FieldBackgroundB(fieldB->getUnit()),
606  currentStep, FieldBackgroundB::InfluenceParticlePusher);
607  dc.releaseData( FieldE::getName() );
608  dc.releaseData( FieldB::getName() );
609 
610  this->myFieldSolver->update_beforeCurrent(currentStep);
611 
612  auto fieldJ = dc.get< FieldJ >( FieldJ::getName(), true );
613  FieldJ::ValueType zeroJ( FieldJ::ValueType::create(0.) );
614  fieldJ->assign( zeroJ );
615 
616  __setTransactionEvent(commEvent);
617  (*currentBGField)(fieldJ, nvfct::Add(), FieldBackgroundJ(fieldJ->getUnit()),
618  currentStep, FieldBackgroundJ::activated);
619 
620  typedef typename pmacc::particles::traits::FilterByFlag
621  <
622  VectorAllSpecies,
623  current<>
624  >::type VectorSpeciesWithCurrentSolver;
625  ForEach<
626  VectorSpeciesWithCurrentSolver,
627  ComputeCurrent<
628  bmpl::_1,
629  bmpl::int_<CORE + BORDER>
630  >
631  > computeCurrent;
632  computeCurrent( currentStep );
633 
634  if(bmpl::size<VectorSpeciesWithCurrentSolver>::type::value > 0)
635  {
636  EventTask eRecvCurrent = fieldJ->asyncCommunication(__getTransactionEvent());
637 
638  const DataSpace<simDim> currentRecvLower( GetMargin<typename fields::Solver::CurrentInterpolation>::LowerMargin( ).toRT( ) );
639  const DataSpace<simDim> currentRecvUpper( GetMargin<typename fields::Solver::CurrentInterpolation>::UpperMargin( ).toRT( ) );
640 
641  /* without interpolation, we do not need to access the FieldJ GUARD
642  * and can therefor overlap communication of GUARD->(ADD)BORDER & computation of CORE */
643  if( currentRecvLower == DataSpace<simDim>::create(0) &&
644  currentRecvUpper == DataSpace<simDim>::create(0) )
