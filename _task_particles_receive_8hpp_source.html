37  typedef T_Particles Particles;

41 
42  enum
43  {
44  Dim = Particles::Dim,
45  Exchanges = traits::NumberOfExchanges<Dim>::value
46  };
47 
48  TaskParticlesReceive(Particles &parBase) :
49  parBase(parBase),
50  state(Constructor){ }
51 
52  virtual void init()
53  {
54  state = Init;
55  EventTask serialEvent = __getTransactionEvent();
56  HandleExchanged handleExchanged;
57  HandleNotExchanged handleNotExchanged;
58 
59  for (int i = 1; i < Exchanges; ++i)
60  {
61  /* Start new transaction */
62  __startTransaction(serialEvent);
63 
64  /* Handle particles */
65  if (parBase.getParticlesBuffer().hasReceiveExchange(i))
66  handleExchanged.handleIncoming(parBase, i);
67  else
68  handleNotExchanged.handleIncoming(parBase, i);
69 
70  /* End transaction */
71  tmpEvent += __endTransaction();
72  }
73 
74  state = WaitForReceived;
75  }
76 
77  bool executeIntern()
78  {
79  switch (state)
80  {
81  case Init:
82  break;
83  case WaitForReceived:
84  if (nullptr == Environment<>::get().Manager().getITaskIfNotFinished(tmpEvent.getTaskId()))
85  state = CallFillGaps;
86  break;
87  case CallFillGaps:
88  state = WaitForFillGaps;
89  __startTransaction();
90  parBase.fillBorderGaps();
91  tmpEvent = __endTransaction();
92  state = Finish;
93  break;
94  case WaitForFillGaps:
95  break;
96  case Finish:
97  return nullptr == Environment<>::get().Manager().getITaskIfNotFinished(tmpEvent.getTaskId());
98  default:
99  return false;
100  }
101 
102  return false;
103  }
