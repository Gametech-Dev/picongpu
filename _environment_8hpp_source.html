442  void EnvironmentContext::setDevice(int deviceNumber)
443  {
444  int num_gpus = 0; //number of gpus
445  cudaGetDeviceCount(&num_gpus);
446 #if (PMACC_CUDA_ENABLED == 1)
447  //##ERROR handling
448  if (num_gpus < 1) //check if cuda device is found
449  {
450  throw std::runtime_error("no CUDA capable devices detected");
451  }
452  else if (deviceNumber >= num_gpus) //check if device can be selected by deviceNumber
453  {
454  std::cerr << "no CUDA device " << deviceNumber << ", only " << num_gpus << " devices found" << std::endl;
455  throw std::runtime_error("CUDA capable devices can't be selected");
456  }
457 #endif
458 
459  int maxTries = num_gpus;
460 #if (PMACC_CUDA_ENABLED == 1)
461  cudaDeviceProp devProp;
462  CUDA_CHECK((cuplaError_t)cudaGetDeviceProperties(&devProp, deviceNumber));
463  /* if the gpu compute mode is set to default we use the given `deviceNumber` */
464  if (devProp.computeMode == cudaComputeModeDefault)
465  maxTries = 1;
466 #endif
467  cudaError rc;
468 
469  for (int deviceOffset = 0; deviceOffset < maxTries; ++deviceOffset)
470  {
471  const int tryDeviceId = (deviceOffset + deviceNumber) % num_gpus;
472  rc = cudaSetDevice(tryDeviceId);
473 
474  if(rc == cudaSuccess)
475  {
476  cudaStream_t stream;
477  /* \todo: Check if this workaround is needed
478  *
479  * - since NVIDIA change something in driver cudaSetDevice never
480  * return an error if another process already use the selected
481  * device if gpu compute mode is set "process exclusive"
482  * - create a dummy stream to check if the device is already used by
483  * an other process.
484  * - cudaStreamCreate fail if gpu is already in use
485  */
486  rc = cudaStreamCreate(&stream);
487  }
488 
489  if (rc == cudaSuccess)
490  {
491 #if (PMACC_CUDA_ENABLED == 1)
492  cudaDeviceProp dprop;
493  CUDA_CHECK((cuplaError_t)cudaGetDeviceProperties(&dprop, tryDeviceId));
494  log<ggLog::CUDA_RT > ("Set device to %1%: %2%") % tryDeviceId % dprop.name;
495  if(cudaErrorSetOnActiveProcess == cudaSetDeviceFlags(cudaDeviceScheduleSpin))
496  {
497  cudaGetLastError(); //reset all errors
498  /* - because of cudaStreamCreate was called cudaSetDeviceFlags crashed
499  * - to set the flags reset the device and set flags again
500  */
501  CUDA_CHECK(cudaDeviceReset());
502  CUDA_CHECK((cuplaError_t)cudaSetDeviceFlags(cudaDeviceScheduleSpin));
503  }
504 #endif
505  CUDA_CHECK(cudaGetLastError());
506  break;
507  }
508  else if (rc == cudaErrorDeviceAlreadyInUse
509 #if (PMACC_CUDA_ENABLED == 1)
510  || rc==(cudaError)cudaErrorDevicesUnavailable
511 #endif
512  )
513  {
514  cudaGetLastError(); //reset all errors
515  log<ggLog::CUDA_RT > ("Device %1% already in use, try next.") % tryDeviceId;
516  continue;
517  }
518  else
519  {
520  CUDA_CHECK(rc); /*error message*/
521  }
522  }
523 
524  m_isDeviceSelected = true;
525  }
