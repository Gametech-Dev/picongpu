47  template<typename T_Type, typename T_Vector, typename T_FieldType>
48  HDINLINE void
49  operator()(T_Type& cursor, T_Vector& pos, const T_FieldType& fieldPos)
50  {
51  const uint32_t dim = T_Vector::dim;
52  using ValueType = typename T_Vector::type;
53 
54  using Supports = T_Supports;
55  using Component = T_Component;
56 
57  const uint32_t component = Component::x::value;
58  const uint32_t support = Supports::template at<component>::type::value;
59  const bool isEven = (support % 2) == 0;
60 
61 
62  const ValueType v_pos = pos[component] - fieldPos[component];
63  DataSpace< dim > intShift;
64  intShift[component] = GetOffsetToStaticShapeSystem <isEven>()(v_pos);
65  cursor = cursor(intShift);
66  pos[component] = v_pos - ValueType(intShift[component]);
67  }
