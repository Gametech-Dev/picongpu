50  HDINLINE float_X operator()( const float_X N, const float_X omega, vector_X const & observer_unit_vec ) const
51  {
52  float_X sincValue = float_X( 1.0 );
53  for( uint32_t d = 0; d < DIM3; ++d )
54  sincValue *= math::sinc( observer_unit_vec[d] * cellSize[d] / ( SPEED_OF_LIGHT * float_X( 2.0 ) ) * omega );
55 
56  // here we combine sinc^2(..) with (...)^T_shapeOrder to ...^(2 * T_shapeOrder)
57  return math::sqrt( N + ( N * N - N ) * util::pow( sincValue , 2 * T_shapeOrder ) );
58  }
