71 template< uint32_t T_numWorkers >
72 struct KernelBinEnergyParticles
73 {
74  /* sum up the energy of all particles
75  *
76  * the kinetic energy of all active particles will be calculated
77  *
78  * @tparam T_ParBox pmacc::ParticlesBox, particle box type
79  * @tparam T_BinBox pmacc::DataBox, box type for the histogram in global memory
80  * @tparam T_Mapping type of the mapper to map a cuda block to a supercell index
81  * @tparam T_Acc alpaka accelerator type
82  *
83  * @param acc alpaka accelerator
84  * @param pb box with access to the particles of the current used species
85  * @param gBins box with memory for resulting histogram
86  * @param numBins number of bins in the histogram (must be fit into the shared memory)
87  * @param minEnergy particle energy for the first bin
88  * @param maxEnergy particle energy for the last bin
89  * @param mapper functor to map a cuda block to a supercells index
90  */
91  template<
92  typename T_ParBox,
93  typename T_BinBox,
94  typename T_Mapping,
95  typename T_Filter,
96  typename T_Acc
97  >
98  DINLINE void operator()(
99  T_Acc const & acc,
100  T_ParBox pb,
101  T_BinBox gBins,
102  int const numBins,
103  float_X const minEnergy,
104  float_X const maxEnergy,
105  T_Mapping const mapper,
106  T_Filter filter
107  ) const
108  {
109  using namespace pmacc::mappings::threads;
110  using SuperCellSize = typename MappingDesc::SuperCellSize;
111  using FramePtr = typename T_ParBox::FramePtr;
112  constexpr uint32_t maxParticlesPerFrame = pmacc::math::CT::volume< SuperCellSize >::type::value;
113  constexpr uint32_t numWorkers = T_numWorkers;
114 
115  PMACC_SMEM(
116  acc,
117  frame,
118  FramePtr
119  );
120 
121  PMACC_SMEM(
122  acc,
123  particlesInSuperCell,
124  lcellId_t
125  );
126 
127  /* shBins index can go from 0 to (numBins+2)-1
128  * 0 is for <minEnergy
129  * (numBins+2)-1 is for >maxEnergy
130  */
131  sharedMemExtern(shBin,float_X); /* size must be numBins+2 because we have <min and >max */
132 
133 
134  int const realNumBins = numBins + 2;
135 
136  uint32_t const workerIdx = threadIdx.x;
137 
138  using MasterOnly = IdxConfig<
139  1,
140  numWorkers
141  >;
142 
143  DataSpace< simDim > const superCellIdx(
144  mapper.getSuperCellIndex( DataSpace< simDim >( blockIdx ) )
145  );
146 
147  ForEachIdx< MasterOnly >{ workerIdx }(
148  [&](
149  uint32_t const,
150  uint32_t const
151  )
152  {
153  frame = pb.getLastFrame( superCellIdx );
154  particlesInSuperCell = pb.getSuperCell( superCellIdx ).getSizeLastFrame( );
155  }
156  );
157 
158  ForEachIdx<
159  IdxConfig<
160  numWorkers,
161  numWorkers
162  >
163  >{ workerIdx }(
164  [&](
165  uint32_t const linearIdx,
166  uint32_t const
167  )
168  {
169  /* set all bins to 0 */
170  for( int i = linearIdx; i < realNumBins; i += numWorkers )
171  shBin[ i ] = float_X( 0. );
172  }
173  );
174 
175  __syncthreads();
176 
177  if( !frame.isValid( ) )
178  return; /* end kernel if we have no frames */
179 
180  auto accFilter = filter(
181  acc,
182  superCellIdx - mapper.getGuardingSuperCells( ),
183  WorkerCfg< numWorkers >{ workerIdx }
184  );
185 
186  while( frame.isValid() )
187  {
188  // move over all particles in a frame
189  ForEachIdx<
190  IdxConfig<
191  maxParticlesPerFrame,
192  numWorkers
193  >
194  >{ workerIdx }(
195  [&](
196  uint32_t const linearIdx,
197  uint32_t const
198  )
199  {
200  if( linearIdx < particlesInSuperCell )
201  {
202  auto const particle = frame[ linearIdx ];
203  if(
204  accFilter(
205  acc,
206  particle
207  )
208  )
209  {
210  /* kinetic Energy for Particles: E^2 = p^2*c^2 + m^2*c^4
211  * = c^2 * [p^2 + m^2*c^2]
212  */
213  float3_X const mom = particle[ momentum_ ];
214  float_X const weighting = particle[ weighting_ ];
215  float_X const mass = attribute::getMass(
216  weighting,
217  particle
218  );
219 
220  // calculate kinetic energy of the macro particle
221  float_X localEnergy = KinEnergy< >( )(
222  mom,
223  mass
224  );
225 
226  localEnergy /= weighting;
227 
228  /* +1 move value from 1 to numBins+1 */
229  int binNumber = math::floor(
230  ( localEnergy - minEnergy ) /
231  ( maxEnergy - minEnergy ) * static_cast< float_X >( numBins )
232  ) + 1;
233 
234  int const maxBin = numBins + 1;
235 
236  /* all entries larger than maxEnergy go into bin maxBin */
237  binNumber = binNumber < maxBin ? binNumber : maxBin;
238 
239  /* all entries smaller than minEnergy go into bin zero */
240  binNumber = binNumber > 0 ? binNumber : 0;
241 
250  float_X const normedWeighting = weighting /
251  float_X( particles::TYPICAL_NUM_PARTICLES_PER_MACROPARTICLE );
252  atomicAdd(
253  &( shBin[ binNumber ] ),
254  normedWeighting,
255  ::alpaka::hierarchy::Threads{}
256  );
257  }
258  }
259  }
260  );
261 
262  __syncthreads();
263 
264  ForEachIdx< MasterOnly >{ workerIdx }(
265  [&](
266  uint32_t const,
267  uint32_t const
268  )
269  {
270  frame = pb.getPreviousFrame( frame );
271  particlesInSuperCell = maxParticlesPerFrame;
272  }
273  );
274  __syncthreads();
275  }
276 
277  ForEachIdx<
278  IdxConfig<
279  numWorkers,
280  numWorkers
281  >
282  >{ workerIdx }(
283  [&](
284  uint32_t const linearIdx,
285  uint32_t const
286  )
287  {
288  for( int i = linearIdx; i < realNumBins; i += numWorkers )
289  atomicAdd(
290  &( gBins[ i ] ),
291  float_64( shBin[ i ] ),
292  ::alpaka::hierarchy::Blocks{}
293  );
294  }
295  );
296  }
297 };

624  template< uint32_t AREA >
625  void calBinEnergyParticles(uint32_t currentStep)
626  {
627  gBins->getDeviceBuffer().setValue(0);
628 
629  DataConnector &dc = Environment<>::get().DataConnector();
630  auto particles = dc.get< ParticlesType >( ParticlesType::FrameType::getName(), true );
631 
632  /* convert energy values from keV to PIConGPU units */
633  float_X const minEnergy = minEnergy_keV * UNITCONV_keV_to_Joule / UNIT_ENERGY;
634  float_X const maxEnergy = maxEnergy_keV * UNITCONV_keV_to_Joule / UNIT_ENERGY;
635 
636  constexpr uint32_t numWorkers = pmacc::traits::GetNumWorkers<
637  pmacc::math::CT::volume< SuperCellSize >::type::value
638  >::value;
639 
640  AreaMapping<
641  AREA,
642  MappingDesc
643  > mapper( *m_cellDescription );
644 
645  auto kernel = PMACC_KERNEL( KernelBinEnergyParticles< numWorkers >{ } )(
646  mapper.getGridDim( ),
647  numWorkers,
648  realNumBins * sizeof( float_X )
649  );
650 
651  auto bindKernel = std::bind(
652  kernel,
653  particles->getDeviceParticlesBox( ),
654  gBins->getDeviceBuffer( ).getDataBox( ),
655  numBins,
656  minEnergy,
657  maxEnergy,
658  mapper,
659  std::placeholders::_1
660  );
661 
662  ForEach<
663  typename Help::EligibleFilters,
664  plugins::misc::ExecuteIfNameIsEqual< bmpl::_1 >
665  >{ }(
666  m_help->filter.get( m_id ),
667  currentStep,
668  bindKernel
669  );
670 
671  dc.releaseData( ParticlesType::FrameType::getName() );
672  gBins->deviceToHost();
673 
674  reduce(nvidia::functors::Add(),
675  binReduced,
676  gBins->getHostBuffer().getBasePointer(),
677  realNumBins, mpi::reduceMethods::Reduce());
678 
679 
680  if (writeToFile)
681  {
682  using dbl = std::numeric_limits<float_64>;
683 
684  outFile.precision(dbl::digits10);
685 
686  /* write data to file */
687  float_64 count_particles = 0.0;
688  outFile << currentStep << " "
689  << std::scientific; /* for floating points, ignored for ints */
690 
691  for (int i = 0; i < realNumBins; ++i)
692  {
693  count_particles += float_64( binReduced[i]);
694  outFile << std::scientific << (binReduced[i]) * float_64(particles::TYPICAL_NUM_PARTICLES_PER_MACROPARTICLE) << " ";
695  }
696  /* endl: Flush any step to the file.
697  * Thus, we will have data if the program should crash.
698  */
699  outFile << std::scientific << count_particles * float_64(particles::TYPICAL_NUM_PARTICLES_PER_MACROPARTICLE)
700  << std::endl;
701  }
702  }
