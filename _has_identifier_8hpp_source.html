40 template<typename T_Object, typename T_Key>
41 struct HasIdentifier
42 {
43  /* The compiler is allowed to evaluate an expression that does not depend on a template parameter
44  * even if the class is never instantiated. In that case static assert is always
45  * evaluated (e.g. with clang), this results in an error if the condition is false.
46  * http://www.boost.org/doc/libs/1_60_0/doc/html/boost_staticassert.html
47  *
48  * A workaround is to add a template dependency to the expression.
49  * `sizeof(ANY_TYPE) != 0` is always true and defers the evaluation.
50  */
51  PMACC_CASSERT_MSG_TYPE(
52  ___HasIdentifier_is_not_specialized_for_T_Object,
53  T_Object,
54  false && ( sizeof(T_Object) != 0 )
55  );
56 };
57 
58 template<typename T_Object, typename T_Key>
59 bool hasIdentifier(const T_Object& obj,const T_Key& key)
60 {
61  return HasIdentifier<T_Object,T_Key>::type::value;
62 }
