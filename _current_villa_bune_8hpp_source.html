38 template<typename T_ParticleShape>
39 struct VillaBune
40 {
41  template<class BoxJ, typename PosType, typename VelType, typename ChargeType, typename T_Acc >
42  DINLINE void operator()(const T_Acc& acc,
43  BoxJ& boxJ_par, /*box which is shifted to particles cell*/
44  const PosType pos,
45  const VelType velocity,
46  const ChargeType charge, const float_X deltaTime)
47  {
48  /* VillaBune: field to particle interpolation _requires_ the CIC shape */
49  PMACC_CASSERT_MSG_TYPE(currentSolverVillaBune_requires_shapeCIC_in_particleConfig,
50  T_ParticleShape,
51  T_ParticleShape::support == 2);
52 
53  // normalize deltaPos to innerCell units [0.; 1.)
54  // that means: dx_real = v.x() * dt
55  // dx_inCell = v.x() * dt / cellSize.x()
56  const float3_X deltaPos(
57  velocity.x() * deltaTime / cellSize.x(),
58  velocity.y() * deltaTime / cellSize.y(),
59  velocity.z() * deltaTime / cellSize.z());
60 
61  const PosType oldPos = (PosType) (precisionCast<float_X > (pos) - deltaPos);
62 
63  addCurrentSplitX(acc, oldPos, pos, charge, boxJ_par, deltaTime);
64  }

81  T_Acc const & acc,
82  const float3_X& oldPos,
83  const float3_X& newPos,
84  const float_X charge,

86  const float_X deltaTime

89 
90  if (math::float2int_rd(oldPos.x()) != math::float2int_rd(newPos.x()))
91  {
92  const float3_X interPos = intersectXPlane(oldPos, newPos,
93  math::max(math::float2int_rd(oldPos.x()), math::float2int_rd(newPos.x())));
94  addCurrentSplitY(acc, oldPos, interPos, charge, mem, deltaTime);
95  addCurrentSplitY(acc, interPos, newPos, charge, mem, deltaTime);
96  return;
97  }

101  template<
102  typename Buffer,
103  typename T_Acc
104  >
105  DINLINE void addCurrentToSingleCell(
106  T_Acc const & acc,
107  float3_X meanPos,
108  const float3_X& deltaPos,
109  const float_X charge,
110  Buffer & memIn,
111  const float_X deltaTime
112  )
113  {
114  //shift to the cell meanPos belongs to
115  //because meanPos may exceed the range [0,1)
116  DataSpace<DIM3> off(math::float2int_rd(meanPos.x()),
117  math::float2int_rd(meanPos.y()),
118  math::float2int_rd(meanPos.z()));
119 
120  auto mem = memIn.shift(off);
121 
122  //fit meanPos into the range [0,1)
123  meanPos.x() -= math::floor(meanPos.x());
124  meanPos.y() -= math::floor(meanPos.y());
125  meanPos.z() -= math::floor(meanPos.z());
126 
127  //for the formulas used in here see Villasenor/Buneman paper page 314
128  const float_X tmp = deltaPos.x() * deltaPos.y() * deltaPos.z() * (float_X(1.0) / float_X(12.0));
129 
130  // j = rho * v
131  // = rho * dr / dt
132  //const float_X rho = charge * (1.0 / (CELL_WIDTH * CELL_HEIGHT * CELL_DEPTH));
133  //const float_X rho_dt = rho * (1.0 / deltaTime);
134 
135  // now carefully:
136  // deltaPos is in "inCell" coordinates, that means:
137  // deltaPos.x() = deltaPos_real.x() / cellSize.x()
138  // to calculate the current density in realUnits it is
139  // j.x() = rho * deltaPos_real.x() / dt
140  // = rho * deltaPos.x() * cellSize.x() / dt
141  // So put adding the constant directly to rho results in:
142  // const float_X rho_dtX = rho * CELL_WIDTH;
143  // const float_X rho_dtY = rho * CELL_HEIGHT;
144  // const float_X rho_dtZ = rho * CELL_DEPTH;
145 
146  // This is exactly the same like:
147  // j = Q / A / t
148  // j.x() = Q.x() * (1.0 / (CELL_HEIGHT * CELL_DEPTH * deltaTime));
149  // j.y() = Q.y() * (1.0 / (CELL_WIDTH * CELL_DEPTH * deltaTime));
150  // j.z() = Q.z() * (1.0 / (CELL_WIDTH * CELL_HEIGHT * deltaTime));
151  // with the difference, that (imagine a moving quader)
152  // Q.x() = charge * deltaPos_real.x() / cellsize.x()
153  // = charge * deltaPos.x() / 1.0
154  //
155  const float_X rho_dtX = charge * (float_X(1.0) / (CELL_HEIGHT * CELL_DEPTH * deltaTime));
156  const float_X rho_dtY = charge * (float_X(1.0) / (CELL_WIDTH * CELL_DEPTH * deltaTime));
157  const float_X rho_dtZ = charge * (float_X(1.0) / (CELL_WIDTH * CELL_HEIGHT * deltaTime));
158 
159  atomicAdd(&(mem[1][1][0].x()), rho_dtX * (deltaPos.x() * meanPos.y() * meanPos.z() + tmp), ::alpaka::hierarchy::Threads{});
160  atomicAdd(&(mem[1][0][0].x()), rho_dtX * (deltaPos.x() * (float_X(1.0) - meanPos.y()) * meanPos.z() - tmp), ::alpaka::hierarchy::Threads{});
161  atomicAdd(&(mem[0][1][0].x()), rho_dtX * (deltaPos.x() * meanPos.y() * (float_X(1.0) - meanPos.z()) - tmp), ::alpaka::hierarchy::Threads{});
162  atomicAdd(&(mem[0][0][0].x()), rho_dtX * (deltaPos.x() * (float_X(1.0) - meanPos.y()) * (float_X(1.0) - meanPos.z()) + tmp), ::alpaka::hierarchy::Threads{});
163 
164  atomicAdd(&(mem[1][0][1].y()), rho_dtY * (deltaPos.y() * meanPos.z() * meanPos.x() + tmp), ::alpaka::hierarchy::Threads{});
165  atomicAdd(&(mem[0][0][1].y()), rho_dtY * (deltaPos.y() * (float_X(1.0) - meanPos.z()) * meanPos.x() - tmp), ::alpaka::hierarchy::Threads{});
166  atomicAdd(&(mem[1][0][0].y()), rho_dtY * (deltaPos.y() * meanPos.z() * (float_X(1.0) - meanPos.x()) - tmp), ::alpaka::hierarchy::Threads{});
167  atomicAdd(&(mem[0][0][0].y()), rho_dtY * (deltaPos.y() * (float_X(1.0) - meanPos.z()) * (float_X(1.0) - meanPos.x()) + tmp), ::alpaka::hierarchy::Threads{});
168 
169  atomicAdd(&(mem[0][1][1].z()), rho_dtZ * (deltaPos.z() * meanPos.x() * meanPos.y() + tmp), ::alpaka::hierarchy::Threads{});
170  atomicAdd(&(mem[0][1][0].z()), rho_dtZ * (deltaPos.z() * (float_X(1.0) - meanPos.x()) * meanPos.y() - tmp), ::alpaka::hierarchy::Threads{});
171  atomicAdd(&(mem[0][0][1].z()), rho_dtZ * (deltaPos.z() * meanPos.x() * (float_X(1.0) - meanPos.y()) - tmp), ::alpaka::hierarchy::Threads{});
172  atomicAdd(&(mem[0][0][0].z()), rho_dtZ * (deltaPos.z() * (float_X(1.0) - meanPos.x()) * (float_X(1.0) - meanPos.y()) + tmp), ::alpaka::hierarchy::Threads{});
173 
174  }

176  //calculates the intersection point of the [pos1,pos2] beam with an y,z-plane at position x0
177 
178  DINLINE float3_X intersectXPlane(const float3_X& pos1, const float3_X& pos2, const float_X x0)
179  {
180  const float_X t = (x0 - pos1.x()) / (pos2.x() - pos1.x());
181 
182  return float3_X(x0, pos1.y() + t * (pos2.y() - pos1.y()), pos1.z() + t * (pos2.z() - pos1.z()));
183  }
184 
185  DINLINE float3_X intersectYPlane(const float3_X& pos1, const float3_X& pos2, const float_X y0)
186  {
187  const float_X t = (y0 - pos1.y()) / (pos2.y() - pos1.y());
188 
189  return float3_X(pos1.x() + t * (pos2.x() - pos1.x()), y0, pos1.z() + t * (pos2.z() - pos1.z()));
190  }
191 
192  DINLINE float3_X intersectZPlane(const float3_X& pos1, const float3_X& pos2, const float_X z0)
193  {
194  const float_X t = (z0 - pos1.z()) / (pos2.z() - pos1.z());
195 
196  return float3_X(pos1.x() + t * (pos2.x() - pos1.x()), pos1.y() + t * (pos2.y() - pos1.y()), z0);
197  }

202  template<
203  typename Buffer,
204  typename T_Acc
205  >
206  DINLINE void addCurrentSplitZ(
207  T_Acc const & acc,
208  const float3_X &oldPos,
209  const float3_X &newPos,
210  const float_X charge,
211  Buffer & mem,
212  const float_X deltaTime
213  )
214  {
215 
216  if (math::float2int_rd(oldPos.z()) != math::float2int_rd(newPos.z()))
217  {
218  const float3_X interPos = intersectZPlane(oldPos, newPos,
219  math::max(math::float2int_rd(oldPos.z()), math::float2int_rd(newPos.z())));
220  float3_X deltaPos = interPos - oldPos;
221  float3_X meanPos = oldPos + float_X(0.5) * deltaPos;
222  addCurrentToSingleCell(acc, meanPos, deltaPos, charge, mem, deltaTime);
223 
224  deltaPos = newPos - interPos;
225  meanPos = interPos + float_X(0.5) * deltaPos;
226  addCurrentToSingleCell(acc, meanPos, deltaPos, charge, mem, deltaTime);
227  return;
228  }
229  const float3_X deltaPos = newPos - oldPos;
230  const float3_X meanPos = oldPos + float_X(0.5) * deltaPos;
231  addCurrentToSingleCell(acc, meanPos, deltaPos, charge, mem, deltaTime);
232  }
