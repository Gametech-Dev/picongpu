74  void update(bool enabled, void* pointer)
75  {
76  const SubGrid<simDim>& subGrid = Environment< simDim >::get().SubGrid();
77  DataConnector &dc = Environment< simDim >::get().DataConnector();
78  auto pField = dc.get< FieldType >( FieldType::getName(), true );
79  DataSpace< simDim > guarding = SuperCellSize::toRT() * cellDescription->getGuardingSuperCells();
80  if (movingWindow)
81  {
82  GridController<simDim> &gc = Environment<simDim>::get().GridController();
83  if (gc.getPosition()[1] == 0) //first gpu
84  {
85  uint32_t* currentStep = (uint32_t*)pointer;
86  Window window( MovingWindow::getInstance().getWindow( *currentStep ) );
87  guarding += subGrid.getLocalDomain().size - window.localDimensions.size;
88  }
89  }
90  typename FieldType::DataBoxType dataBox = pField->getDeviceDataBox();
91  shifted = dataBox.shift( guarding );
92  dc.releaseData( FieldType::getName() );
93  /* avoid deadlock between not finished pmacc tasks and potential blocking operations
94  * within ISAAC
95  */
96  __getTransactionEvent().waitForFinished();
97 
98  }

139  void update(bool enabled, void* pointer)
140  {
141  if (enabled)
142  {
143  uint32_t* currentStep = (uint32_t*)pointer;
144  const SubGrid<simDim>& subGrid = Environment< simDim >::get().SubGrid();
145  DataConnector &dc = Environment< simDim >::get().DataConnector();
146 
147  PMACC_CASSERT_MSG(
148  _please_allocate_at_least_one_FieldTmp_in_memory_param,
149  fieldTmpNumSlots > 0
150  );
151  auto fieldTmp = dc.get< FieldTmp >( FieldTmp::getUniqueId( 0 ), true );
152  auto particles = dc.get< ParticleType >( ParticleType::FrameType::getName(), true );
153 
154  fieldTmp->getGridBuffer().getDeviceBuffer().setValue( FieldTmp::ValueType(0.0) );
155  fieldTmp->template computeValue < CORE + BORDER, FrameSolver > (*particles, *currentStep);
156  EventTask fieldTmpEvent = fieldTmp->asyncCommunication(__getTransactionEvent());
157 
158  __setTransactionEvent(fieldTmpEvent);
159  __getTransactionEvent().waitForFinished();
160 
161  dc.releaseData( ParticleType::FrameType::getName() );
162 
163  DataSpace< simDim > guarding = SuperCellSize::toRT() * cellDescription->getGuardingSuperCells();
164  if (movingWindow)
165  {
166  GridController<simDim> &gc = Environment<simDim>::get().GridController();
167  if (gc.getPosition()[1] == 0) //first gpu
168  {
169  Window window(MovingWindow::getInstance().getWindow( *currentStep ));
170  guarding += subGrid.getLocalDomain().size - window.localDimensions.size;
171  }
172  }
173  typename FieldTmp::DataBoxType dataBox = fieldTmp->getDeviceDataBox();
174  shifted = dataBox.shift( guarding );
175  dc.releaseData( FieldTmp::getUniqueId( 0 ) );
176  }
177  }

259  void notify(uint32_t currentStep)
260  {
261  uint64_t simulation_time = visualization->getTicksUs() - last_notify;
262  step++;
263  if (step >= render_interval)
264  {
265  step = 0;
266  bool pause = false;
267  do
268  {
269  //update of the position for moving window simulations
270  if ( movingWindow )
271  {
272  Window window(MovingWindow::getInstance().getWindow( currentStep ));
273  visualization->updatePosition( window.localDimensions.offset );
274  visualization->updateLocalSize( window.localDimensions.size );
275  visualization->updateBounding();
276  }
277  if (rank == 0 && visualization->kernel_time)
278  {
279  json_object_set_new( visualization->getJsonMetaRoot(), "time step", json_integer( currentStep ) );
280  json_object_set_new( visualization->getJsonMetaRoot(), "drawing_time" , json_integer( drawing_time ) );
281  json_object_set_new( visualization->getJsonMetaRoot(), "simulation_time", json_integer( simulation_time ) );
282  simulation_time = 0;
283  json_object_set_new( visualization->getJsonMetaRoot(), "cell count", json_integer( cell_count ) );
284  json_object_set_new( visualization->getJsonMetaRoot(), "particle count", json_integer( particle_count ) );
285  }
286  uint64_t start = visualization->getTicksUs();
287  json_t* meta = visualization->doVisualization(META_MASTER, &currentStep, !pause);
288  drawing_time = visualization->getTicksUs() - start;
289  json_t* json_pause = nullptr;
290  if ( meta && (json_pause = json_object_get(meta, "pause")) && json_boolean_value( json_pause ) )
291  pause = !pause;
292  if ( meta && json_integer_value( json_object_get(meta, "exit") ) )
293  exit(1);
294  json_t* js;
295  if ( meta && ( js = json_object_get(meta, "interval") ) )
296  {
297  render_interval = math::max( int(1), int( json_integer_value ( js ) ) );
298  //Feedback for other clients than the changing one
299  if (rank == 0)
300  json_object_set_new( visualization->getJsonMetaRoot(), "interval", json_integer( render_interval ) );
301  }
302  json_decref( meta );
303  if (direct_pause)
304  {
305  pause = true;
306  direct_pause = false;
307  }
308  }
309  while (pause);
310  }
311  last_notify = visualization->getTicksUs();
312  }

371  void pluginLoad()
372  {
373  if(!notifyPeriod.empty())
374  {
375  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
376  MPI_Comm_size(MPI_COMM_WORLD, &numProc);
377  if ( MovingWindow::getInstance().isSlidingWindowActive() )
378  movingWindow = true;
379  float_X minCellSize = math::min( cellSize[0], math::min( cellSize[1], cellSize[2] ) );
380  float3_X cellSizeFactor = cellSize / minCellSize;
381 
382  const SubGrid<simDim>& subGrid = Environment< simDim >::get().SubGrid();
383 
384  isaac_size2 framebuffer_size =
385  {
386  cupla::IdxType(width),
387  cupla::IdxType(height)
388  };
389 
390  isaac_for_each_params( sources, SourceInitIterator(), cellDescription, movingWindow );
391 
392  visualization = new VisualizationType (
393  cupla::manager::Device< cupla::AccHost >::get().current( ),
394  cupla::manager::Device< cupla::AccDev >::get().current( ),
395  cupla::manager::Stream< cupla::AccDev, cupla::AccStream >::get().stream( ),
396  name,
397  0,
398  url,
399  port,
400  framebuffer_size,
401  subGrid.getGlobalDomain().size,
402  subGrid.getLocalDomain().size,
403  subGrid.getLocalDomain().offset,
404  sources,
405  cellSizeFactor
406  );
407  visualization->setJpegQuality(jpeg_quality);
408  //Defining the later periodicly sent meta data
409  if (rank == 0)
410  {
411  json_object_set_new( visualization->getJsonMetaRoot(), "time step", json_string( "Time step" ) );
412  json_object_set_new( visualization->getJsonMetaRoot(), "drawing time", json_string( "Drawing time in us" ) );
413  json_object_set_new( visualization->getJsonMetaRoot(), "simulation time", json_string( "Simulation time in us" ) );
414  json_object_set_new( visualization->getJsonMetaRoot(), "cell count", json_string( "Total numbers of cells" ) );
415  json_object_set_new( visualization->getJsonMetaRoot(), "particle count", json_string( "Total numbers of particles" ) );
416  }
417  CommunicatorSetting communicatorBehaviour = reconnect ? RetryEverySend : ReturnAtError;
418  if (visualization->init( communicatorBehaviour ) != 0)
419  {
420  if (rank == 0)
421  log<picLog::INPUT_OUTPUT > ("ISAAC Init failed, disable plugin");
422  notifyPeriod = "";
423  }
424  else
425  {
426  const int localNrOfCells = cellDescription->getGridLayout().getDataSpaceWithoutGuarding().productOfComponents();
427  cell_count = localNrOfCells * numProc;
428  particle_count = localNrOfCells * particles::TYPICAL_PARTICLES_PER_CELL * (bmpl::size<VectorAllSpecies>::type::value) * numProc;
429  last_notify = visualization->getTicksUs();
430  if (rank == 0)
431  log<picLog::INPUT_OUTPUT > ("ISAAC Init succeded");
432  }
433  }
434  Environment<>::get().PluginConnector().setNotificationPeriod(this, notifyPeriod);
435  }
