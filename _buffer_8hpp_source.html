107  virtual DataSpace<DIM> getCurrentDataSpace(size_t currentSize)
108  {
109  DataSpace<DIM> tmp;
110  int64_t current_size = static_cast<int64_t>(currentSize);
111 
113  //call first get current size (but const not allow this)
114 
115  if (DIM == DIM1)
116  {
117  tmp[0] = current_size;
118  }
119  if (DIM == DIM2)
120  {
121  if (current_size <= data_space[0])
122  {
123  tmp[0] = current_size;
124  tmp[1] = 1;
125  } else
126  {
127  tmp[0] = data_space[0];
128  tmp[1] = (current_size+data_space[0]-1) / data_space[0];
129  }
130  }
131  if (DIM == DIM3)
132  {
133  if (current_size <= data_space[0])
134  {
135  tmp[0] = current_size;
136  tmp[1] = 1;
137  tmp[2] = 1;
138  } else if (current_size <= (data_space[0] * data_space[1]))
139  {
140  tmp[0] = data_space[0];
141  tmp[1] = (current_size+data_space[0]-1) / data_space[0];
142  tmp[2] = 1;
143  } else
144  {
145  tmp[0] = data_space[0];
146  tmp[1] = data_space[1];
147  tmp[2] = (current_size+(data_space[0] * data_space[1])-1) / (data_space[0] * data_space[1]);
148  }
149  }
150 
151  return tmp;
152  }
