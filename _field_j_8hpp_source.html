149 template<typename T_SpeciesType, typename T_Area>
150 struct ComputeCurrent
151 {
152  using SpeciesType = T_SpeciesType;
153  using FrameType = typename SpeciesType::FrameType;
154 
155  HINLINE void operator()( const uint32_t currentStep ) const
156  {
157  DataConnector &dc = Environment<>::get().DataConnector();
158  auto species = dc.get< SpeciesType >( FrameType::getName(), true );
159  auto fieldJ = dc.get< FieldJ >( FieldJ::getName(), true );
160 
161  fieldJ->computeCurrent< T_Area::value, SpeciesType >( *species, currentStep );
162 
163  dc.releaseData( FrameType::getName() );
