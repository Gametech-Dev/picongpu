136 template<typename T_Type>
137 struct FreeMemory
138 {
139  template<typename ValueType >
140  HINLINE void operator()(ValueType& value) const
141  {
142  typedef typename pmacc::traits::Resolve<T_Type>::type::type type;
143 
144  type* ptr = value.getIdentifier(T_Type()).getPointer();
145  if (ptr != nullptr)
146  {
147 #if( PMACC_CUDA_ENABLED == 1 )
148 /* cupla 0.1.0 does not support the function cudaHostAlloc to create mapped memory.
149  * Therefore we need to call the native CUDA function cudaFreeHost to free memory.
150  * Due to the renaming of cuda functions with cupla via macros we need to remove
151  * the renaming to get access to the native cuda function.
152  * @todo this is a workaround please fix me. We need to investigate if
153  * it is possible to have mapped/unified memory in alpaka.
154  *
155  * corresponding alpaka issues:
156  * https://github.com/ComputationalRadiationPhysics/alpaka/issues/296
157  * https://github.com/ComputationalRadiationPhysics/alpaka/issues/612
158  */
159 # undef cudaFreeHost
160  CUDA_CHECK((cuplaError_t)cudaFreeHost(ptr));
161 // re-introduce the cupla macro
162 # define cudaFreeHost(...) cuplaFreeHost(__VA_ARGS__)
163 #else
164  __deleteArray(ptr);
165 #endif
166  }
167  }
