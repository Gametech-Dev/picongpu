43 template<class T_Shape, class InterpolationMethod>
44 struct FieldToParticleInterpolation
45 {
46  using AssignmentFunction = typename T_Shape::ChargeAssignmentOnSupport;
47  static constexpr int supp = AssignmentFunction::support;
48 
49  static constexpr int lowerMargin = supp / 2 ;
50  static constexpr int upperMargin = (supp + 1) / 2;
51  typedef typename pmacc::math::CT::make_Int<simDim,lowerMargin>::type LowerMargin;
52  typedef typename pmacc::math::CT::make_Int<simDim,upperMargin>::type UpperMargin;
53 
54  PMACC_CASSERT_MSG(
55  __FieldToParticleInterpolation_supercell_is_too_small_for_stencil,
56  pmacc::math::CT::min<
57  typename pmacc::math::CT::mul<
58  SuperCellSize,
59  GuardSize
60  >::type
61  >::type::value >= lowerMargin &&
62  pmacc::math::CT::min<
63  typename pmacc::math::CT::mul<
64  SuperCellSize,
65  GuardSize
66  >::type
67  >::type::value >= upperMargin
68  );
69 
70  /*(supp + 1) % 2 is 1 for even supports else 0*/
71  static constexpr int begin = -supp / 2 + (supp + 1) % 2;
72  static constexpr int end = begin+supp-1;
73 
74 
75  template<class Cursor, class VecVector>
76  HDINLINE typename Cursor::ValueType operator()(Cursor field,
77  const floatD_X& particlePos,
78  const VecVector& fieldPos)
79  {
84  typedef typename pmacc::math::CT::make_Int<simDim,supp>::type Supports;
85 
86  typename Cursor::ValueType result;
87  for(uint32_t i = 0; i < Cursor::ValueType::dim; i++)
88  {
89  auto fieldComponent = pmacc::cursor::make_FunctorCursor(
90  field,
91  pmacc::algorithm::functor::GetComponent<float_X>(i)
92  );
93  floatD_X particlePosShifted = particlePos;
94  ShiftCoordinateSystem<Supports>()(fieldComponent, particlePosShifted, fieldPos[i]);
95  result[i] = InterpolationMethod::template interpolate<AssignmentFunction, begin, end > (fieldComponent, particlePosShifted);
96  }
97 
98  return result;
99  }
100 
101  static pmacc::traits::StringProperty getStringProperties()
102  {
103  GetStringProperties<InterpolationMethod> propList;
104  return propList;
105  }
106 
107 };
