55  HDINLINE void operator()(
56  T_Particle & particle,

59  {
60  uint32_t maxNumMacroParticles = pmacc::math::CT::volume<
61  typename T_ParamClass::numParticlesPerDimension
62  >::type::value;
63 
64  /* reset the particle position if the operator is called more times
65  * than allowed (m_currentMacroParticles underflow protection for)
66  */
67  if( maxNumMacroParticles <= m_currentMacroParticles )
68  m_currentMacroParticles = maxNumMacroParticles - 1u;
69 
70  // spacing between particles in each direction in the cell
71  DataSpace< simDim > const numParDirection( T_ParamClass::numParticlesPerDimension::toRT() );
72  floatD_X spacing;
73  for( uint32_t i = 0; i < simDim; ++i )
74  spacing[i] = float_X( 1.0 ) / float_X( numParDirection[ i ] );
75 
76  /* coordinate in the local in-cell lattice
77  * x = [0, numParsPerCell_X-1]
78  * y = [0, numParsPerCell_Y-1]
79  * z = [0, numParsPerCell_Z-1]
80  */
81  DataSpace< simDim > inCellCoordinate = DataSpaceOperations< simDim >::map(
82  numParDirection,
83  m_currentMacroParticles
84  );
85 
86  particle[ position_ ] = precisionCast< float_X >( inCellCoordinate ) * spacing +
87  spacing * float_X( 0.5 );
88  particle[ weighting_ ] = m_weighting;
89 
90  --m_currentMacroParticles;
91 
92  }

94  template< typename T_Particle >
95  HDINLINE uint32_t
96  numberOfMacroParticles( float_X const realParticlesPerCell )
97  {
98  auto numParInCell = T_ParamClass::numParticlesPerDimension::toRT();
99 
100  m_weighting = float_X( 0.0 );
101  uint32_t numMacroParticles = pmacc::math::CT::volume<
102  typename T_ParamClass::numParticlesPerDimension
103  >::type::value;
104 
105  if( numMacroParticles > 0u )
106  m_weighting = realParticlesPerCell / float_X( numMacroParticles );
107 
108  while(
109  m_weighting < MIN_WEIGHTING &&
110  numMacroParticles > 0u
111  )
112  {
113  /* decrement component with greatest value*/
114  uint32_t max_component = 0u;
115  for( uint32_t i = 1; i < simDim; ++i )
116  {
117  if( numParInCell[ i ] > numParInCell[ max_component ] )
118  max_component = i;
119  }
120  numParInCell[ max_component ] -= 1u;
121 
122  numMacroParticles = numParInCell.productOfComponents( );
123 
124  if( numMacroParticles > 0u )
125  m_weighting = realParticlesPerCell / float_X( numMacroParticles );
126  else
127  m_weighting = float_X( 0.0 );
128  }
129  m_currentMacroParticles = numMacroParticles - 1u;
130  return numMacroParticles;
131  }
