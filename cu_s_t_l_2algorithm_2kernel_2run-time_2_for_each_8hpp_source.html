100 template<int dim>
101 math::Size_t<DIM3> getBestCudaBlockDim(const math::Size_t<dim> gridDimension)
102 {
103  math::Size_t<DIM3> result = math::Size_t<DIM3>::create(1);
104 
105  /* The greatest common divisor of each component of the volume size
106  * and a certain power of two value yield the best suitable block size */
107  const math::Size_t<DIM3> maxThreads =
108  MaxCudaBlockDim<dim>::type::toRT(); /* max threads per axis */
109  for(int i = 0; i < dim; i++)
110  {
111  result[i] = boost::integer::gcd(gridDimension[i], maxThreads[i]);
112  }
113 
114  return result;
115 }

124 #define FOREACH_OPERATOR(Z, N, _) \
125  /* typename C0, ..., typename CN */ \
126  template<typename Zone, BOOST_PP_ENUM_PARAMS(N, typename C), typename Functor> \
127  /* C0 c0, ..., CN cN */ \
128  void operator()(const Zone& p_zone, BOOST_PP_ENUM_BINARY_PARAMS(N, C, c), const Functor& functor) \
129  { \
130  /* C0 c0_shifted = c0(p_zone.offset); ...; CN cN_shifted = cN(p_zone.offset); */ \
131  BOOST_PP_REPEAT(N, SHIFT_CURSOR_ZONE, _) \
132  \
133  if(this->_blockDim == math::Size_t<DIM3>::create(0)) \
134  this->_blockDim = getBestCudaBlockDim(p_zone.size); \
135  \
136  PMACC_VERIFY(this->_blockDim.productOfComponents() <= cudaSpecs::maxNumThreadsPerBlock); \
137  PMACC_VERIFY(this->_blockDim.x() <= cudaSpecs::MaxNumThreadsPerBlockDim::x::value); \
138  PMACC_VERIFY(this->_blockDim.y() <= cudaSpecs::MaxNumThreadsPerBlockDim::y::value); \
139  PMACC_VERIFY(this->_blockDim.z() <= cudaSpecs::MaxNumThreadsPerBlockDim::z::value); \
140  \
141  typename math::Size_t<3>::BaseType blockSize( \
142  this->_blockDim.x(), \
143  this->_blockDim.y(), \
144  this->_blockDim.z() \
145  ); \
146  uint32_t numWorkers = traits::GetNumWorkers< cudaSpecs::MaxNumThreadsPerBlockDim::x::value >::value; \
147  if( numWorkers > blockSize.productOfComponents() ) \
148  numWorkers = blockSize.productOfComponents(); \
149  kernel::detail::SphericMapper<Zone::dim> mapper; \
150  using namespace pmacc; \
151  PMACC_KERNEL(kernel::detail::RT::KernelForeachLockstep{})(mapper.cudaGridDim(p_zone.size, this->_blockDim), numWorkers) \
152  /* c0_shifted, ..., cN_shifted */ \
153  (mapper, blockSize, functor, BOOST_PP_ENUM(N, SHIFTED_CURSOR, _)); \
154  }

177  /* operator()(zone, cursor0, cursor1, ..., cursorN-1, functor or lambdaFun)
178  *
179  * \param zone Accepts currently only a zone::SphericZone object (e.g. containerObj.zone())
180  * \param cursorN cursor for the N-th data source (e.g. containerObj.origin())
181  * \param functor or lambdaFun either a functor with N arguments or a N-ary lambda function (e.g. _1 = _2)
182  *
183  * The functor or lambdaFun is called for each cell within the zone.

185  *
186  */
187  BOOST_PP_REPEAT_FROM_TO(1, BOOST_PP_INC(FOREACH_KERNEL_MAX_PARAMS), FOREACH_OPERATOR, _)
