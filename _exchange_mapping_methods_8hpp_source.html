60  template<unsigned DIM>
61  class ExchangeMappingMethods<GUARD, DIM>
62  {
63  public:
64 
65  template<class Base>
66  HINLINE static DataSpace< DIM > getGridDim(const Base &base, uint32_t exchangeType)
67  {
68  const DataSpace< DIM > guardingSupercells = base.getGuardingSuperCells();
69  DataSpace< DIM > result(base.getGridSuperCells() - 2 * guardingSupercells);
70 
71  const DataSpace< DIM > directions = Mask::getRelativeDirections< DIM > (exchangeType);
72 
73  for( uint32_t d = 0; d < DIM; ++d )
74  {
75  if (directions[ d ] != 0)
76  result[ d ] = guardingSupercells[ d ];
77  }
78 
79  return result;
80  }
81 
82  template<class Base>
83  HDINLINE static DataSpace< DIM > getBlockIndex(const Base &base,
84  const DataSpace< DIM >& _blockIdx, uint32_t exchangeType)
85  {
86  DataSpace< DIM > result(_blockIdx);
87 
88  const DataSpace< DIM > directions = Mask::getRelativeDirections< DIM > (exchangeType);
89  const DataSpace< DIM > guardingSupercells = base.getGuardingSuperCells();
90 
91  for( uint32_t d = 0; d < DIM; ++d )
92  {
93  if (directions[ d ] == 0)
94  result[ d ] += guardingSupercells[ d ];
95  else
96  if (directions[ d ] == 1)
97  result[ d ] += base.getGridSuperCells()[ d ] - guardingSupercells[ d ];
98  }
99 
100  return result;
101  }
102  };

112  template<class Base>
113  HINLINE static DataSpace< DIM > getGridDim(const Base &base, uint32_t exchangeType)
114  {
115  // skip 2 x (border + guard) == 4 x guard
116  DataSpace< DIM > result(base.getGridSuperCells() - 4 * base.getGuardingSuperCells());
117 
118  DataSpace< DIM > directions = Mask::getRelativeDirections< DIM > (exchangeType);
119 
120  for( uint32_t d = 0; d < DIM; ++d )
121  {
122  if (directions[ d ] != 0)
123  result[ d ] = base.getGuardingSuperCells()[ d ];
124  }
125 
126  return result;
127  }

129  template<class Base>
130  HDINLINE static DataSpace< DIM > getBlockIndex(const Base &base,
131  const DataSpace< DIM >& _blockIdx, uint32_t exchangeType)
132  {
133  DataSpace< DIM > result(_blockIdx);
134 
135  DataSpace< DIM > directions = Mask::getRelativeDirections< DIM > (exchangeType);
136 
137  DataSpace< DIM > guardingBlocks = base.getGuardingSuperCells();
138 
139  for( uint32_t d = 0; d < DIM; ++d )
140  {
141  switch (directions[ d ])
142  {
143  case 0:
144  result[ d ] += 2 * guardingBlocks[ d ];
145  break;
146  case -1:
147  result[ d ] += guardingBlocks[ d ];
148  break;
149  case 1:
150  result[ d ] += base.getGridSuperCells()[ d ] - 2 * guardingBlocks[ d ];
151  break;
152  }
153  }
154 
155  return result;
156  }
