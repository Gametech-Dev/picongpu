62 template<class T_ParticleShape, class T_DerivedAttribute>
63 template<class FrameType, class TVecSuperCell, class BoxTmp, typename T_Acc>
64 DINLINE void
65 ComputeGridValuePerFrame<T_ParticleShape, T_DerivedAttribute>::operator()
66 (
67  T_Acc const & acc,
68  FrameType& frame,
69  const int localIdx,
70  const TVecSuperCell superCell,
71  BoxTmp& tmpBox
72 )
73 {
74  /* \todo in the future and if useful, the functor can be a parameter */
75  T_DerivedAttribute particleAttribute;
76 
77  auto particle = frame[localIdx];
78 
79  /* particle attributes: in-cell position and generic, derived attribute */
80  const floatD_X pos = particle[position_];
81  const auto particleAttr = particleAttribute( particle );
82 
86  const int particleCellIdx = particle[localCellIdx_];
87  const DataSpace<TVecSuperCell::dim> particleCell(
88  DataSpaceOperations<TVecSuperCell::dim>::map( superCell, particleCellIdx )
89  );
90  auto fieldTmpShiftToParticle = tmpBox.shift(particleCell);
91 
92  /* loop around the particle's cell (according to shape) */
93  const DataSpace<simDim> lowMargin(LowerMargin().toRT());
94  const DataSpace<simDim> upMargin(UpperMargin().toRT());
95 
96  const DataSpace<simDim> marginSpace(upMargin + lowMargin + 1);
97 
98  const int numWriteCells = marginSpace.productOfComponents();
99 
100  for (int i = 0; i < numWriteCells; ++i)
101  {
105  const DataSpace<simDim> currentCell = DataSpaceOperations<simDim>::map(marginSpace, i);
106 
110  const DataSpace<simDim> offsetParticleCellToCurrentCell = currentCell - lowMargin;
111 
116  float_X assign( 1.0 );
117  for (uint32_t d = 0; d < simDim; ++d)
118  assign *= AssignmentFunction()(float_X(offsetParticleCellToCurrentCell[d]) - pos[d]);
119 
125  atomicAdd(
126  &(fieldTmpShiftToParticle(offsetParticleCellToCurrentCell).x()),
127  assign * particleAttr,
128  ::alpaka::hierarchy::Threads{}
129  );
130  }
131 }
