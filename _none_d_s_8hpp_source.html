38  template<uint32_t T_simDim, uint32_t T_plane>
39  struct LinearInterpolateComponentPlaneUpper
40  {
41  static constexpr uint32_t dim = T_simDim;
42 
43  /* UpperMargin is actually 0 in direction of T_plane */
44  using LowerMargin = typename pmacc::math::CT::make_Int<
45  dim,
46  0
47  >::type;
48  using UpperMargin = typename pmacc::math::CT::make_Int<
49  dim,

51  >::type;

53  template<typename DataBox>
54  HDINLINE float_X operator()( DataBox const & field ) const
55  {
56  DataSpace< dim > const self;
57  DataSpace< dim > up;
58  up[(T_plane + 1) % dim] = 1;
59 
60  using Avg = LinearInterpolateWithUpper< dim >;
61 
62  typename Avg::template GetInterpolatedValue< ( T_plane + 2 ) % dim > const avg;
63 
64  return float_X( 0.5 ) * ( avg( field )[ T_plane ] + avg( field.shift( up ) )[ T_plane ] );
65  }

163  struct NoneDS
164  {
165  static constexpr uint32_t dim = simDim;
166 
167  typedef typename pmacc::math::CT::make_Int<dim, 0>::type LowerMargin;
168  typedef typename pmacc::math::CT::make_Int<dim, 1>::type UpperMargin;
169 
170  template<
171  typename T_DataBoxE,
172  typename T_DataBoxB,
173  typename T_DataBoxJ
174  >
175  HDINLINE void operator()(
176  T_DataBoxE fieldE,
177  T_DataBoxB fieldB,
178  T_DataBoxJ const fieldJ
179  )
180  {
181  using TypeJ = typename T_DataBoxJ::ValueType;
182  using ComponentJ = typename GetComponentsType< TypeJ >::type;
183 
184  DataSpace< dim > const self;
185 
186  constexpr ComponentJ deltaT = DELTA_T;
187  ComponentJ const constE = ( float_X( 1.0 ) / EPS0 ) * deltaT;
188  ComponentJ const constB = ( float_X( 0.25 ) / EPS0 ) * deltaT * deltaT;
189 
190  detail::LinearInterpolateComponentPlaneUpper<
191  dim,
192  0
193  > const avgX;
194  ComponentJ const jXavg = avgX( fieldJ );
195  detail::LinearInterpolateComponentPlaneUpper<
196  dim,
197  1
198  > const avgY;
199  ComponentJ const jYavg = avgY( fieldJ );
200  detail::LinearInterpolateComponentPlaneUpper<
201  dim,
202  2
203  > const avgZ;
204  ComponentJ const jZavg = avgZ( fieldJ );
205 
206  TypeJ const jAvgE = TypeJ(
207  jXavg,
208  jYavg,
209  jZavg
210  );
211  fieldE( self ) -= jAvgE * constE;
212 
213  using CurlRight = fields::maxwellSolver::yee::Curl< DifferenceToUpper< dim > >;
214  using ShiftCurlRight = detail::ShiftCurl< DifferenceToUpper< dim > >;
215  CurlRight curl;
216  ShiftCurlRight shiftCurl;
217 
218  TypeJ const jAvgB = curl( fieldJ ) + shiftCurl( fieldJ );
219  fieldB(self) += jAvgB * constB;
220  }
