47  uint32_t
48  getOffset( const int gpuPos, const uint32_t maxCells ) const
49  {
50  value_type::const_iterator iter = parsedInput.begin();
51  // go to last gpu of this block b{n}
52  int i = iter->second - 1;
53  int sum = 0;
54 
55  while( i < gpuPos )
56  {
57  // add last block
58  sum += iter->first * iter->second;
59 
60  ++iter;
61  // go to last gpu of this block b{n}
62  i += iter->second;
63  }
64 
65  // add part of this block that is before me
66  sum += iter->first * ( gpuPos + iter->second - i - 1 );
67 
68  // check total number of cells
69  uint32_t sumTotal = 0;
70  for( iter = parsedInput.begin(); iter != parsedInput.end(); ++iter )
71  sumTotal += iter->first * iter->second;
72 
73  PMACC_VERIFY( sumTotal == maxCells );
74 
75  return sum;
76  }

88  int i = iter->second - 1;
89

112  void
113  parseString( const std::string s )
114  {
115  boost::regex regFind( "[0-9]+(\\{[0-9]+\\})*",
116  boost::regex_constants::perl );
117 
118  boost::sregex_token_iterator iter( s.begin( ), s.end( ),
119  regFind, 0 );
120  boost::sregex_token_iterator end;
121 
122  parsedInput.clear();
123  parsedInput.reserve( std::distance( iter, end ) );
124 
125  for(; iter != end; ++iter )
126  {
127  std::string pM = *iter;
128 
129  // find multiplier n and base b of b{n}
130  boost::regex regMultipl( "(.*\\{)|(\\})",
131  boost::regex_constants::perl );
132  std::string multipl = boost::regex_replace( pM, regMultipl, "" );
133  boost::regex regBase( "\\{.*\\}",
134  boost::regex_constants::perl );
135  std::string base = boost::regex_replace( pM, regBase, "" );
136 
137  // no Multiplier {n} given
138  if( multipl == *iter )
139  multipl = "1";
140 
141  const std::pair<uint32_t, uint32_t> g(
142  boost::lexical_cast<uint32_t > ( base ),
143  boost::lexical_cast<uint32_t > ( multipl ) );
144  parsedInput.push_back( g );
145  }
146  }
