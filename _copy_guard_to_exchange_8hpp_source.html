77  {
78  using namespace mappings::threads;
79 
80  using SuperCellSize = typename T_Mapping::SuperCellSize;
81 
82  // number of cells in a superCell
83  constexpr uint32_t numCells = pmacc::math::CT::volume< SuperCellSize >::type::value;
84  constexpr uint32_t numWorkers = T_numWorkers;
85  PMACC_CONSTEXPR_CAPTURE int dim = T_Mapping::Dim;
86 
87  uint32_t const workerIdx = threadIdx.x;
88 
89  DataSpace< dim > const blockCell(
90  mapper.getSuperCellIndex( DataSpace< dim >( blockIdx ) ) *
91  SuperCellSize::toRT()
92  );
93 
94  // origin in area from local GPU
95  DataSpace< dim > nullSourceCell(
96  mapper.getSuperCellIndex( DataSpace< dim > () ) *
97  SuperCellSize::toRT()
98  );
99 
100  auto const numGuardSuperCells = mapper.getGuardingSuperCells();
101 
102  ForEachIdx<
103  IdxConfig<
104  numCells,
105  numWorkers
106  >
107  >{ workerIdx }(
108  [&](
109  uint32_t const linearIdx,
110  uint32_t const
111  )
112  {
113  // cell index within the superCell
114  DataSpace< dim > const cellIdx = DataSpaceOperations< dim >::template map< SuperCellSize >( linearIdx );
115 
116  DataSpace< T_Mapping::Dim > const sourceCell( blockCell + cellIdx );
117  DataSpace< dim > targetCell( sourceCell - nullSourceCell );
118 
119  // supercell offset relative to the guard origin (in cells)
120  DataSpace< dim > superCellOffsetInGuard( ( targetCell / SuperCellSize::toRT() ) * SuperCellSize::toRT() );
121 
122  /* defines if the virtual worker needs to copy the value of
123  * the cell to to the exchange box
124  */
125  bool copyValue = true;
126 
127  for( uint32_t d = 0; d < dim; ++d )
128  {
129  if( direction[ d ] == -1 )
130  {
131  if(
132  superCellOffsetInGuard[ d ] + cellIdx[ d ] <
133  numGuardSuperCells[ d ] * SuperCellSize::toRT()[ d ] - exchangeSize[ d ]
134  )
135  copyValue = false;
136  targetCell[ d ] -= numGuardSuperCells[ d ] * SuperCellSize::toRT()[ d ] - exchangeSize[ d ];
137  }
138  else if(
139  direction[d] == 1 && superCellOffsetInGuard[ d ] + cellIdx[ d ] >=
140  exchangeSize[d]
141  )
142  copyValue = false;
143  }
144 
145  if( copyValue )
146  exchangeBox( targetCell ) = srcBox( sourceCell );
147  }
148  );
149  }
