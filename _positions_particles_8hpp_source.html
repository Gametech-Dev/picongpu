46 template<class FloatPos>
47 struct SglParticle
48 {
49  FloatPos position;
50  float3_X momentum;
51  float_X mass;
52  float_X weighting;
53  float_X charge;
54  float_X gamma;
55 
56  SglParticle() : position(FloatPos::create(0.0)), momentum(float3_X::create(0.0)), mass(0.0),
57  weighting(0.0), charge(0.0), gamma(0.0)
58  {
59  }
60 
61  DataSpace<simDim> globalCellOffset;
62 
64 
65  floatD_64 getGlobalCell() const
66  {
67  floatD_64 doubleGlobalCellOffset;
68  for(uint32_t i=0;i<simDim;++i)
69  doubleGlobalCellOffset[i]=float_64(globalCellOffset[i]);
70 
71  return floatD_64( doubleGlobalCellOffset + precisionCast<float_64>(position));
72  }
73 
74  template<typename T>
75  friend std::ostream& operator<<(std::ostream& out, const SglParticle<T>& v)
76  {
77  floatD_64 pos;
78  for(uint32_t i=0;i<simDim;++i)
79  pos[i]=( v.getGlobalCell()[i] * cellSize[i]*UNIT_LENGTH);
80 
81  const float3_64 mom( precisionCast<float_64>(v.momentum.x()) * UNIT_MASS * UNIT_SPEED,
82  precisionCast<float_64>(v.momentum.y()) * UNIT_MASS * UNIT_SPEED,
83  precisionCast<float_64>(v.momentum.z()) * UNIT_MASS * UNIT_SPEED );
84 
85  const float_64 mass = precisionCast<float_64>(v.mass) * UNIT_MASS;
86  const float_64 charge = precisionCast<float_64>(v.charge) * UNIT_CHARGE;
87 
88  using dbl = std::numeric_limits<float_64>;
89  out.precision(dbl::digits10);
90 
91  out << std::scientific << pos << " " << mom << " " << mass << " "
92  << precisionCast<float_64>(v.weighting)
93  << " " << charge << " " << precisionCast<float_64>(v.gamma);
94  return out;
95  }

102 struct KernelPositionsParticles
103 {
104  template<
105  typename ParBox,
106  typename FloatPos,
107  typename Mapping,
108  typename T_Acc
109  >
110  DINLINE void operator()(
111  T_Acc const & acc,
112  ParBox pb,
113  SglParticle<FloatPos>* gParticle,
114  Mapping mapper
115  ) const
116  {
117 
118  using FramePtr = typename ParBox::FramePtr;
119  PMACC_SMEM( acc, frame, FramePtr );
120 
121 
122  using SuperCellSize = typename Mapping::SuperCellSize;
123 
124  const DataSpace<simDim > threadIndex(threadIdx);
125  const int linearThreadIdx = DataSpaceOperations<simDim>::template map<SuperCellSize > (threadIndex);
126  const DataSpace<simDim> superCellIdx(mapper.getSuperCellIndex(DataSpace<simDim > (blockIdx)));
127 
128  if (linearThreadIdx == 0)
129  {
130  frame = pb.getLastFrame(superCellIdx);
131  }
132 
133  __syncthreads();
134  if (!frame.isValid())
135  return; //end kernel if we have no frames
136 
137  /* BUGFIX to issue #538
138  * volatile prohibits that the compiler creates wrong code*/
139  volatile bool isParticle = frame[linearThreadIdx][multiMask_];
140 
141  while (frame.isValid())
142  {
143  if (isParticle)
144  {
145  auto particle = frame[linearThreadIdx];
146  gParticle->position = particle[position_];
147  gParticle->momentum = particle[momentum_];
148  gParticle->weighting = particle[weighting_];
149  gParticle->mass = attribute::getMass(gParticle->weighting,particle);
150  gParticle->charge = attribute::getCharge(gParticle->weighting,particle);
151  gParticle->gamma = Gamma<>()(gParticle->momentum, gParticle->mass);
152 
153  // storage number in the actual frame
154  const lcellId_t frameCellNr = particle[localCellIdx_];
155 
156  // offset in the actual superCell = cell offset in the supercell
157  const DataSpace<simDim> frameCellOffset(DataSpaceOperations<simDim>::template map<MappingDesc::SuperCellSize > (frameCellNr));
158 
159 
160  gParticle->globalCellOffset = (superCellIdx - mapper.getGuardingSuperCells())
161  * MappingDesc::SuperCellSize::toRT()
162  + frameCellOffset;
163  }
164  __syncthreads();
165  if (linearThreadIdx == 0)
166  {
167  frame = pb.getPreviousFrame(frame);
168  }
169  isParticle = true;
170  __syncthreads();
171  }
172 
173  }

207  void notify(uint32_t currentStep)
208  {
209  const int rank = Environment<simDim>::get().GridController().getGlobalRank();

254  template< uint32_t AREA>
255  SglParticle<FloatPos> getPositionsParticles(uint32_t currentStep)
256  {
257  typedef typename MappingDesc::SuperCellSize SuperCellSize;
258  SglParticle<FloatPos> positionParticleTmp;
259 
260  DataConnector &dc = Environment<>::get().DataConnector();
261  auto particles = dc.get< ParticlesType >( ParticlesType::FrameType::getName(), true );
262 
263  gParticle->getDeviceBuffer().setValue(positionParticleTmp);
264  auto block = SuperCellSize::toRT();
265 
266  AreaMapping<AREA, MappingDesc> mapper(*cellDescription);
267  PMACC_KERNEL(KernelPositionsParticles{})
268  (mapper.getGridDim(), block)
269  (particles->getDeviceParticlesBox(),
270  gParticle->getDeviceBuffer().getBasePointer(),
271  mapper);
272 
273  dc.releaseData( ParticlesType::FrameType::getName() );
274  gParticle->deviceToHost();
275 
276  DataSpace<simDim> localSize(cellDescription->getGridLayout().getDataSpaceWithoutGuarding());
277  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(currentStep);
278 
279  DataSpace<simDim> gpuPhyCellOffset(Environment<simDim>::get().SubGrid().getLocalDomain().offset);
280  gpuPhyCellOffset.y() += (localSize.y() * numSlides);
281 
282  gParticle->getHostBuffer().getDataBox()[0].globalCellOffset += gpuPhyCellOffset;
283 
284 
285  return gParticle->getHostBuffer().getDataBox()[0];
286  }
