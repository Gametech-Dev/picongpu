101  void restart(uint32_t restartStep, const std::string & restartDirectory)
102  {
103  HBufCalorimeter hBufLeftParsCalorimeter(this->dBufLeftParsCalorimeter->size());
104 
105  pmacc::GridController<simDim>& gridCon = pmacc::Environment<simDim>::get().GridController();
106  pmacc::CommunicatorMPI<simDim>& comm = gridCon.getCommunicator();
107  uint32_t rank = comm.getRank();
108 
109  if(rank == 0)
110  {
111  splash::SerialDataCollector hdf5DataFile(1);
112  splash::DataCollector::FileCreationAttr fAttr;
113 
114  splash::DataCollector::initFileCreationAttr(fAttr);
115  fAttr.fileAccType = splash::DataCollector::FAT_READ;
116 
117  std::stringstream filename;
118  filename << restartDirectory << "/" << ( this->foldername + "/" + filenamePrefix ) << "_" << restartStep;
119 
120  hdf5DataFile.open(filename.str().c_str(), fAttr);
121 
122  splash::Dimensions dimensions;
123 
124  hdf5DataFile.read(restartStep,
125  this->leftParticlesDatasetName.c_str(),
126  dimensions,
127  &(*hBufLeftParsCalorimeter.origin()));
128 
129  hdf5DataFile.close();
130 
131  /* rank 0 divides and distributes the calorimeter to all ranks in equal parts */
132  uint32_t numRanks = gridCon.getGlobalSize();
133  // get a host accelerator
134  auto hostDev = cupla::manager::Device< cupla::AccHost >::get().device( );
135  pmacc::algorithm::host::Foreach()(hostDev,
136  hBufLeftParsCalorimeter.zone(),
137  hBufLeftParsCalorimeter.origin(),
138  DivideInPlace<float_X>(float_X(numRanks)));
139  }
140 
141  // avoid deadlock between not finished pmacc tasks and mpi blocking collectives
142  __getTransactionEvent().waitForFinished();
143  MPI_Bcast(&(*hBufLeftParsCalorimeter.origin()),
144  hBufLeftParsCalorimeter.size().productOfComponents() * sizeof(float_X),
145  MPI_CHAR,
146  0, /* rank 0 */
147  comm.getMPIComm());
148 
149  *this->dBufLeftParsCalorimeter = hBufLeftParsCalorimeter;
150  }

154  {
155  /* create folder for hdf5 checkpoint files*/
156  Environment<simDim>::get().Filesystem().createDirectoryWithPermissions( checkpointDirectory + "/" + this->foldername);
157  HBufCalorimeter hBufLeftParsCalorimeter(this->dBufLeftParsCalorimeter->size());

162  /* mpi reduce */
163  (*this->allGPU_reduce)(hBufTotal, hBufLeftParsCalorimeter, pmacc::algorithm::functor::Add{});
164  if(!this->allGPU_reduce->root())
165  return;

167  splash::SerialDataCollector hdf5DataFile(1);
168  splash::DataCollector::FileCreationAttr fAttr;

172  std::stringstream filename;
173  filename << checkpointDirectory << "/" << ( this->foldername + "/" + filenamePrefix ) << "_" << currentStep;
174 
175  hdf5DataFile.open(filename.str().c_str(), fAttr);
176 
177  typename PICToSplash<float_X>::type SplashTypeX;
178 
179  splash::Dimensions bufferSize(hBufTotal.size().x(),
180  hBufTotal.size().y(),
181  hBufTotal.size().z());
182 
183  /* if there is only one energy bin, omit the energy axis */
184  uint32_t dimension = this->numBinsEnergy == 1 ? DIM2 : DIM3;
185  hdf5DataFile.write(currentStep,

191 
192  hdf5DataFile.close();

195 private:
196  void initPlugin()
197  {
198  namespace pm = pmacc::math;
199 
200  if(!(this->openingYaw_deg > float_X(0.0) && this->openingYaw_deg <= float_X(360.0)))
201  {
202  std::stringstream msg;
203  msg << "[Plugin] [" << m_help->getOptionPrefix()
204  << "] openingYaw has to be within (0, 360]."
205  << std::endl;
206  throw std::runtime_error(msg.str());
207  }
208  if(!(this->openingPitch_deg > float_X(0.0) && this->openingPitch_deg <= float_X(180.0)))
209  {
210  std::stringstream msg;
211  msg << "[Plugin] [" << m_help->getOptionPrefix()
212  << "] openingPitch has to be within (0, 180]."
213  << std::endl;
214  throw std::runtime_error(msg.str());
215  }
216  if(this->minEnergy < float_X(0.0))
217  {
218  std::stringstream msg;
219  msg << "[Plugin] [" << m_help->getOptionPrefix()
220  << "] minEnergy can not be negative."
221  << std::endl;
222  throw std::runtime_error(msg.str());
223  }
224  if(this->logScale && this->minEnergy == float_X(0.0))
225  {
226  std::stringstream msg;
227  msg << "[Plugin] [" << m_help->getOptionPrefix()
228  << "] minEnergy can not be zero in logarithmic scaling."
229  << std::endl;
230  throw std::runtime_error(msg.str());
231  }
232  if(this->numBinsEnergy > 1 && this->maxEnergy <= this->minEnergy)
233  {
234  std::stringstream msg;
235  msg << "[Plugin] [" << m_help->getOptionPrefix()
236  << "] minEnergy has to be less than maxEnergy."
237  << std::endl;
238  throw std::runtime_error(msg.str());
239  }
240 
241  this->maxYaw_deg = float_X(0.5) * this->openingYaw_deg;
242  this->maxPitch_deg = float_X(0.5) * this->openingPitch_deg;
243  /* convert units */
244  const float_64 minEnergy_SI = this->minEnergy * UNITCONV_keV_to_Joule;
245  const float_64 maxEnergy_SI = this->maxEnergy * UNITCONV_keV_to_Joule;
246  this->minEnergy = minEnergy_SI / UNIT_ENERGY;
247  this->maxEnergy = maxEnergy_SI / UNIT_ENERGY;
248 
249  /* allocate memory buffers */
250  this->dBufCalorimeter = new DBufCalorimeter(this->numBinsYaw, this->numBinsPitch, this->numBinsEnergy);
251  this->dBufLeftParsCalorimeter = new DBufCalorimeter(this->dBufCalorimeter->size());
252  this->hBufCalorimeter = new HBufCalorimeter(this->dBufCalorimeter->size());
253  this->hBufTotalCalorimeter = new HBufCalorimeter(this->dBufCalorimeter->size());
254 
255  /* fill calorimeter for left particles with zero */
256  this->dBufLeftParsCalorimeter->assign(float_X(0.0));
257 
258  /* create mpi reduce algorithm */
259  pmacc::GridController<simDim>& con = pmacc::Environment<simDim>::get().GridController();
260  pm::Size_t<simDim> gpuDim = (pm::Size_t<simDim>)con.getGpuNodes();
261  zone::SphericZone<simDim> zone_allGPUs(gpuDim);
262  this->allGPU_reduce = AllGPU_reduce(new pmacc::algorithm::mpi::Reduce<simDim>(zone_allGPUs));
263 
264  /* calculate rotated calorimeter frame from posYaw_deg and posPitch_deg */
265  constexpr float_64 radsInDegree = pmacc::algorithms::math::Pi<float_64>::value / float_64(180.0);
266  const float_64 posYaw_rad = this->posYaw_deg * radsInDegree;
267  const float_64 posPitch_rad = this->posPitch_deg * radsInDegree;
268  this->calorimeterFrameVecY = float3_X(math::sin(posYaw_rad) * math::cos(posPitch_rad),
269  math::cos(posYaw_rad) * math::cos(posPitch_rad),
270  math::sin(posPitch_rad));
271  /* If the y-axis is pointing exactly up- or downwards we need to define the x-axis manually */
272  if(math::abs(this->calorimeterFrameVecY.z()) == float_X(1.0))
273  {
274  this->calorimeterFrameVecX = float3_X(1.0, 0.0, 0.0);
275  }
276  else
277  {
278  /* choose `calorimeterFrameVecX` so that the roll is zero. */
279  const float3_X vecUp(0.0, 0.0, -1.0);
280  this->calorimeterFrameVecX = math::cross(vecUp, this->calorimeterFrameVecY);
281  /* normalize vector */
282  this->calorimeterFrameVecX /= math::abs(this->calorimeterFrameVecX);
283  }
284  this->calorimeterFrameVecZ = math::cross(this->calorimeterFrameVecX, this->calorimeterFrameVecY);
285 
286  /* create calorimeter functor instance */
287  this->calorimeterFunctor = MyCalorimeterFunctorPtr(new MyCalorimeterFunctor(
288  this->maxYaw_deg * radsInDegree,
289  this->maxPitch_deg * radsInDegree,
290  this->numBinsYaw,
291  this->numBinsPitch,
292  this->numBinsEnergy,
293  this->logScale ? math::log10(this->minEnergy) : this->minEnergy,
294  this->logScale ? math::log10(this->maxEnergy) : this->maxEnergy,
295  this->logScale,
296  this->calorimeterFrameVecX,
297  this->calorimeterFrameVecY,
298  this->calorimeterFrameVecZ));
299 
300  /* create folder for hdf5 files*/
301  Environment<simDim>::get().Filesystem().createDirectoryWithPermissions(this->foldername);
302 
303  // set how often the plugin should be executed while PIConGPU is running
304  Environment<>::get( ).PluginConnector( ).setNotificationPeriod(
305  this,
306  m_help->notifyPeriod.get( m_id )
307  );
308  }

310  void writeToHDF5File(uint32_t currentStep)
311  {
312  splash::SerialDataCollector hdf5DataFile(1);
313  splash::DataCollector::FileCreationAttr fAttr;
314 
315  splash::DataCollector::initFileCreationAttr(fAttr);
316 
317  std::stringstream filename;
318  filename << this->foldername << "/" << filenamePrefix << "_" << currentStep;
319 
320  hdf5DataFile.open(filename.str().c_str(), fAttr);
321 
322  typename PICToSplash<float_X>::type SplashTypeX;
323  typename PICToSplash<float_64>::type SplashType64;
324  typename PICToSplash<bool>::type SplashTypeBool;
325 
326  splash::Dimensions bufferSize(this->hBufTotalCalorimeter->size().x(),
327  this->hBufTotalCalorimeter->size().y(),
328  this->hBufTotalCalorimeter->size().z());
329 
330  hdf5DataFile.write(currentStep,
331  SplashTypeX,
332  this->numBinsEnergy == 1 ? DIM2 : DIM3,
333  splash::Selection(bufferSize),
334  "calorimeter",
335  &(*this->hBufTotalCalorimeter->origin()));

338 
339  hdf5DataFile.writeAttribute(currentStep,
340  SplashType64,
341  "calorimeter",
342  "unitSI",
343  &unitSI);
344 
345  hdf5DataFile.writeAttribute(currentStep,
346  SplashType64,
347  "calorimeter",
348  "posYaw[deg]",
349  &posYaw_deg);
350 
351  hdf5DataFile.writeAttribute(currentStep,
352  SplashType64,
353  "calorimeter",
354  "posPitch[deg]",
355  &posPitch_deg);
356 
357  hdf5DataFile.writeAttribute(currentStep,
358  SplashTypeX,
359  "calorimeter",
360  "maxYaw[deg]",
361  &this->maxYaw_deg);
362 
363  hdf5DataFile.writeAttribute(currentStep,
364  SplashTypeX,
365  "calorimeter",
366  "maxPitch[deg]",
367  &this->maxPitch_deg);
368 
369  if(this->numBinsEnergy > 1)
370  {
371  const float_64 minEnergy_SI = this->minEnergy * UNIT_ENERGY;
372  const float_64 maxEnergy_SI = this->maxEnergy * UNIT_ENERGY;
373  const float_64 minEnergy_keV = minEnergy_SI * UNITCONV_Joule_to_keV;
374  const float_64 maxEnergy_keV = maxEnergy_SI * UNITCONV_Joule_to_keV;
375 
376  hdf5DataFile.writeAttribute(currentStep,
377  SplashType64,
378  "calorimeter",
379  "minEnergy[keV]",
380  &minEnergy_keV);
381 
382  hdf5DataFile.writeAttribute(currentStep,
383  SplashType64,
384  "calorimeter",
385  "maxEnergy[keV]",
386  &maxEnergy_keV);
387 
388  hdf5DataFile.writeAttribute(currentStep,
389  SplashTypeBool,
390  "calorimeter",
391  "logScale",
392  &this->logScale);
393  }
394 
395  hdf5DataFile.close();
396  }

679  {
680  /* initialize calorimeter with already detected particles */
681  *this->dBufCalorimeter = *this->dBufLeftParsCalorimeter;
