45  DINLINE Amplitude(vector_64 vec, picongpu::float_X phase)
46  {
47  picongpu::float_X cosValue;
48  picongpu::float_X sinValue;
49  picongpu::math::sincos(phase, sinValue, cosValue);
50  amp_x=picongpu::math::euler(vec.x(), picongpu::precisionCast<picongpu::float_64>(sinValue), picongpu::precisionCast<picongpu::float_64>(cosValue) );
51  amp_y=picongpu::math::euler(vec.y(), picongpu::precisionCast<picongpu::float_64>(sinValue), picongpu::precisionCast<picongpu::float_64>(cosValue) );
52  amp_z=picongpu::math::euler(vec.z(), picongpu::precisionCast<picongpu::float_64>(sinValue), picongpu::precisionCast<picongpu::float_64>(cosValue) );
53  }

113  HDINLINE picongpu::float_64 calc_radiation(void)
114  {
115  // const SI factor radiation
116  const picongpu::float_64 factor = 1.0 /
117  (16. * util::cube(pmacc::algorithms::math::Pi< picongpu::float_64 >::value) * picongpu::EPS0 * picongpu::SPEED_OF_LIGHT);
118 
119  return factor * (picongpu::math::abs2(amp_x) + picongpu::math::abs2(amp_y) + picongpu::math::abs2(amp_z));
