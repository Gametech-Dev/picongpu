48 /* count particles in an area
49  * is not optimized, it checks any particle position if it is really a particle
50  */
51 struct KernelIntensity
52 {
53  template<
54  typename FieldBox,
55  typename BoxMax,
56  typename BoxIntegral,
57  typename T_Acc
58  >
59  DINLINE void operator()(
60  T_Acc const & acc,
61  FieldBox field,
62  DataSpace<simDim> cellsCount,
63  BoxMax boxMax,
64  BoxIntegral integralBox
65  ) const
66  {
67 
68  typedef MappingDesc::SuperCellSize SuperCellSize;
69  PMACC_SMEM( acc, s_integrated, memory::Array< float_X,SuperCellSize::y::value > );
70  PMACC_SMEM( acc, s_max, memory::Array< float_X, SuperCellSize::y::value > );
71 
72 
73  /*descripe size of a worker block for cached memory*/
74  typedef SuperCellDescription<
75  pmacc::math::CT::Int<SuperCellSize::x::value,SuperCellSize::y::value>
76  > SuperCell2D;
77 
78  auto s_field = CachedBox::create<
79  0,
80  float_32
81  >(
82  acc,
83  SuperCell2D()
84  );
85 
86  int y = blockIdx.y * SuperCellSize::y::value + threadIdx.y;
87  int yGlobal = y + GuardSize::y::value * SuperCellSize::y::value;
88  const DataSpace<DIM2> threadId(threadIdx);
89 
90  if (threadId.x() == 0)
91  {
92  // clear destination arrays
93  s_integrated[threadId.y()] = float_X(0.0);
94  s_max[threadId.y()] = float_X(0.0);
95  }
96  __syncthreads();
97 
98  // move cell-wise over z direction (without guarding cells)
99  for (int z = GuardSize::z::value * SuperCellSize::z::value; z < cellsCount.z() - GuardSize::z::value * SuperCellSize::z::value; ++z)
100  {
101  // move supercell-wise over x direction without guarding
102  for (int x = GuardSize::x::value * SuperCellSize::x::value + threadId.x(); x < cellsCount.x() - GuardSize::x::value * SuperCellSize::x::value; x += SuperCellSize::x::value)
103  {
104  const float3_X field_at_point(field(DataSpace<DIM3 > (x, yGlobal, z)));
105  s_field(threadId) = math::abs2(field_at_point);
106  __syncthreads();
107  if (threadId.x() == 0)
108  {
109  // master thread moves cell-wise over 2D supercell
110  for (int x_local = 0; x_local < SuperCellSize::x::value; ++x_local)
111  {
112  DataSpace<DIM2> localId(x_local, threadId.y());
113  s_integrated[threadId.y()] += s_field(localId);
114  s_max[threadId.y()] = fmaxf(s_max[threadId.y()], s_field(localId));
115 
116  }
117  }
118  }
119  }
120  __syncthreads();
121 
122  if (threadId.x() == 0)
123  {
124  /*copy result to global array*/
125  integralBox[y] = s_integrated[threadId.y()];
126  boxMax[y] = s_max[threadId.y()];
127  }
128 
129 
130  }
131 };

234  /* reduce data from all gpus to one array
235  * @param currentStep simulation step
236  */
237  void combineData(uint32_t currentStep)
238  {
239 
240  const DataSpace<simDim> localSize(cellDescription->getGridLayout().getDataSpaceWithoutGuarding());
241  Window window(MovingWindow::getInstance().getWindow( currentStep));
242 
243  const SubGrid<simDim>& subGrid = Environment<simDim>::get().SubGrid();
244 
245  const int yGlobalSize = subGrid.getGlobalDomain().size.y();
246  const int yLocalSize = localSize.y();
247 
248  const int gpus = Environment<simDim>::get().GridController().getGpuNodes().productOfComponents();
249 
250 
252  DataSpace<simDim> globalRootCell(subGrid.getLocalDomain().offset);
253  int yOffset = globalRootCell.y();
254  int* yOffsetsAll = new int[gpus];
255  float_32* maxAll = new float_32[yGlobalSize];
256  float_32* maxAllTmp = new float_32[yLocalSize * gpus];
257  memset(maxAll, 0, sizeof (float_32) *yGlobalSize);
258  float_32* integretedAll = new float_32[yGlobalSize];
259  float_32* integretedAllTmp = new float_32[yLocalSize * gpus];
260  memset(integretedAll, 0, sizeof (float_32) *yGlobalSize);
261 
262  // avoid deadlock between not finished pmacc tasks and mpi blocking collectives
263  __getTransactionEvent().waitForFinished();
264  MPI_CHECK(MPI_Gather(&yOffset, 1, MPI_INT, yOffsetsAll, 1,
265  MPI_INT, 0, MPI_COMM_WORLD));
266 
267  MPI_CHECK(MPI_Gather(localMaxIntensity->getHostBuffer().getBasePointer(), yLocalSize, MPI_FLOAT,
268  maxAllTmp, yLocalSize, MPI_FLOAT,
269  0, MPI_COMM_WORLD));
270  MPI_CHECK(MPI_Gather(localIntegratedIntensity->getHostBuffer().getBasePointer(), yLocalSize, MPI_FLOAT,
271  integretedAllTmp, yLocalSize, MPI_FLOAT,
272  0, MPI_COMM_WORLD));
273 
274  if (writeToFile)
275  {
276  for (int i = 0; i < gpus; ++i)
277  {
278  int gOffset = yOffsetsAll[i];
279  int tmpOff = yLocalSize*i;
280  for (int y = 0; y < yLocalSize; ++y)
281  {
282  maxAll[gOffset + y] = std::max(maxAllTmp[tmpOff + y], maxAll[gOffset + y]);
283  integretedAll[gOffset + y] += integretedAllTmp[tmpOff + y];
284  }
285  }
286 
287  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(currentStep);
288  size_t physicelYCellOffset = numSlides * yLocalSize + window.globalDimensions.offset.y();
289  writeFile(currentStep,
290  maxAll + window.globalDimensions.offset.y(),
291  window.globalDimensions.size.y(),
292  physicelYCellOffset,
293  outFileMax,
294  UNIT_EFIELD
295  );
296 
297  float_64 unit=UNIT_EFIELD*CELL_VOLUME*SI::EPS0_SI;
298  for(uint32_t i=0;i<simDim;++i)
299  unit*=UNIT_LENGTH;
300 
301  writeFile(currentStep,
302  integretedAll + window.globalDimensions.offset.y(),
303  window.globalDimensions.size.y(),
304  physicelYCellOffset,
305  outFileIntegrated,
306  unit
307  );
308  }
309 
310  __deleteArray(yOffsetsAll);
311  __deleteArray(maxAll);
312  __deleteArray(integretedAll);
313  __deleteArray(maxAllTmp);
314  __deleteArray(integretedAllTmp);
315  }

317  /* write data from array to a file

325  * @param unit unit to scale values from pic units to si units
326  */
327  void writeFile(size_t currentStep, float* array, size_t count, size_t physicalYOffset, std::ofstream& stream, float_64 unit)
328  {
329  stream << currentStep << " ";
330  for (size_t i = 0; i < count; ++i)
331  {
332  stream << (physicalYOffset + i) * SI::CELL_HEIGHT_SI << " ";
333  }
334  stream << std::endl << currentStep << " ";
335  for (size_t i = 0; i < count; ++i)
336  {
337  stream << sqrt((float_64) (array[i])) * unit << " ";
338  }
339  stream << std::endl;
340  }

342  /* run calculation of intensity
343  * sync all result data to host side
344  *
345  * @param currenstep simulation step
346  */
347  void calcIntensity(uint32_t)
348  {
349  DataConnector &dc = Environment<>::get().DataConnector();
350 
351  auto fieldE = dc.get< FieldE >( FieldE::getName(), true );
352 
353  /*start only worker for any supercell in laser propagation direction*/
354  DataSpace<DIM2> grid(1,cellDescription->getGridSuperCells().y() - cellDescription->getGuardingSuperCells().y());
355  /*use only 2D slice XY for supercell handling*/
356  typedef typename MappingDesc::SuperCellSize SuperCellSize;
357  auto block = pmacc::math::CT::Vector<SuperCellSize::x,SuperCellSize::y>::toRT();
358 
359  PMACC_KERNEL(KernelIntensity{})
360  (grid, block)
361  (
362  fieldE->getDeviceDataBox(),
363  fieldE->getGridLayout().getDataSpace(),
364  localMaxIntensity->getDeviceBuffer().getDataBox(),
365  localIntegratedIntensity->getDeviceBuffer().getDataBox()
366  );
367 
368  dc.releaseData( FieldE::getName() );
369 
370  localMaxIntensity->deviceToHost();
371  localIntegratedIntensity->deviceToHost();
372 
373  }
