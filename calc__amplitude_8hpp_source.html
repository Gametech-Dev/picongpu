42 struct One_minus_beta_times_n
43 {
48 
49  // Taylor just includes a method, When includes just enum
50 
51  HDINLINE picongpu::float_32 operator()(const vector_64& n, const Particle & particle) const
52  {
53  // 1/gamma^2:
54 
55  const picongpu::float_64 gamma_inv_square(particle.get_gamma_inv_square<When::now > ());
56 
57  //picongpu::float_64 value; // storage for 1-\beta \times \vec n
58 
59  // if energy is high enough to cause numerical errors ( equals if 1/gamma^2 is close enough to zero)
60  // chose a Taylor approximation to to better calculate 1-\beta \times \vec n (which is close to 1-1)
61  // is energy is low, then the approximation will cause a larger error, therefor calculate
62  // 1-\beta \times \vec n directly
63  // with 0.18 the relative error will be below 0.001% for a Taylor series of 1-sqrt(1-x) of 5th order
64  if (gamma_inv_square < picongpu::GAMMA_INV_SQUARE_RAD_THRESH)
65  {
66  const picongpu::float_64 cos_theta(particle.get_cos_theta<When::now > (n)); // cosine between looking vector and momentum of particle
67  const picongpu::float_64 taylor_approx(cos_theta * Taylor()(gamma_inv_square) + (1.0 - cos_theta));
68  return (taylor_approx);
69  }
70  else
71  {
72  const vector_64 beta(particle.get_beta<When::now > ()); // calculate v/c=beta
73  return (1.0 - beta * n);
74  }
75 
76  }
