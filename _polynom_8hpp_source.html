97  template< typename T_Acc >
98  HDINLINE
99  void operator( )(
100  T_Acc const &,
101  DataSpace< simDim > const & cellIndexInSuperCell
102  )
103  {
104  // coordinate system to global simulation as origin
105  DataSpace< simDim > const localCell(
106  cellIndexInSuperCell +
107  m_superCellToLocalOriginCellOffset
108  );
109 
110  // transform coordinate system to center of x-z plane of initialization
111  constexpr uint8_t planeNormalDir = 1u;
112  DataSpace< simDim > offsetToCenterOfPlane( m_offsetToTotalDomain );
113  offsetToCenterOfPlane[ planeNormalDir ] = 0; // do not shift origin of plane normal
114  floatD_X const pos = precisionCast< float_X >( localCell + offsetToCenterOfPlane ) * cellSize.shrink< simDim >();
115  // @todo add half-cells via traits::FieldPosition< Solver::NumicalCellType, FieldE >()
116 
117  // transversal position only
118  float3_X const w0_3D( Unitless::W0_X, 0., Unitless::W0_Z );
119  auto const w0( w0_3D.shrink< simDim >().remove< planeNormalDir >() );
120  auto const pos_trans( pos.remove< planeNormalDir >() );
121  auto const exp_compos( pos_trans * pos_trans / ( w0 * w0 ) );
122  float_X const exp_arg( exp_compos.sumOfComponents() );
123 
124  m_elong *= math::exp( -1.0_X * exp_arg );
125 
126  if( Unitless::initPlaneY != 0 ) // compile time if
127  {
128  /* If the laser is not initialized in the first cell we emit a
129  * negatively and positively propagating wave. Therefore we need to multiply the
130  * amplitude with a correction factor depending of the cell size in
131  * propagation direction.
132  * The negatively propagating wave is damped by the absorber.
133  *
134  * The `correctionFactor` assume that the wave is moving in y direction.
135  */
136  auto const correctionFactor = ( SPEED_OF_LIGHT * DELTA_T ) / CELL_HEIGHT * 2._X;
137 
138  // jump over the guard of the electric field
139  m_dataBoxE( localCell + SuperCellSize::toRT() * GuardSize::toRT() ) += correctionFactor * m_elong;
140  }
141  else
142  {
143  // jump over the guard of the electric field
144  m_dataBoxE( localCell + SuperCellSize::toRT() * GuardSize::toRT() ) = m_elong;
145  }
146  }
147  };

150  template< typename T_Params >
151  struct Polynom : public polynom::Unitless< T_Params >
152  {
153  using Unitless = polynom::Unitless< T_Params >;
154 
155  float3_X elong;
156  float_X phase;
157  typename FieldE::DataBoxType dataBoxE;
158  DataSpace< simDim > offsetToTotalDomain;
159 
160  HDINLINE float_X
161  Tpolynomial( float_X const tau )
162  {
163  float_X result( 0.0_X );
164  if( tau >= 0.0_X && tau <= 1.0_X )
165  result = tau * tau * tau * ( 10.0_X - 15.0_X * tau + 6.0_X * tau * tau );
166  else if( tau > 1.0_X && tau <= 2.0_X )
167  result = ( 2.0_X - tau ) * ( 2.0_X - tau ) * ( 2.0_X - tau ) * ( 4.0_X - 9.0_X * tau + 6.0_X * tau * tau );
168 
169  return result;
170  }
171 
176  HINLINE Polynom( uint32_t currentStep ) :
177  phase( 0.0_X )
178  {
179  // get data
180  DataConnector & dc = Environment< >::get( ).DataConnector( );
181  dataBoxE = dc.get< FieldE >(
182  FieldE::getName(),
183  true
184  )->getDeviceDataBox();
185 
186  // get meta data for offsets
187  SubGrid< simDim > const & subGrid = Environment< simDim >::get().SubGrid();
188  // const DataSpace< simDim > totalCellOffset( subGrid.getGlobalDomain().offset );
189  DataSpace< simDim > const globalCellOffset( subGrid.getLocalDomain().offset );
190  DataSpace< simDim > const halfSimSize( subGrid.getGlobalDomain().size / 2 );
191 
192  // transform coordinate system to center of global simulation as origin [cells]
193  offsetToTotalDomain = /* totalCellOffset + */ globalCellOffset - halfSimSize;
194 
195  // @todo reset origin of direction of moving window
196  // offsetToTotalDomain.y() = 0
197 
198  float_64 const runTime = DELTA_T * currentStep - Unitless::laserTimeShift;
199 
200  elong = float3_X::create( 0.0_X );
201 
202  /* a symmetric pulse will be initialized at position z=0
203  * the laser amplitude rises for t_rise
204  * and falls for t_rise
205  * making the laser pulse 2*t_rise long
206  */
207 
208  const float_X t_rise = 0.5_X * Unitless::PULSE_LENGTH;
209  const float_X tau = runTime / t_rise;
210 
211  const float_X omegaLaser = 2.0_X * PI * Unitless::f;
212 
213  if( Unitless::Polarisation == Unitless::LINEAR_X )
214  {
215  elong.x() = Unitless::AMPLITUDE * Tpolynomial( tau ) *
216  math::sin( omegaLaser * ( runTime - t_rise ) + Unitless::LASER_PHASE );
217  }
218  else if( Unitless::Polarisation == Unitless::LINEAR_Z )
219  {
220  elong.z() = Unitless::AMPLITUDE * Tpolynomial( tau ) *
221  math::sin( omegaLaser * ( runTime - t_rise ) + Unitless::LASER_PHASE );
222  }
223  else if( Unitless::Polarisation == Unitless::CIRCULAR )
224  {
225  elong.x() = Unitless::AMPLITUDE * Tpolynomial( tau ) / math::sqrt( 2.0_X ) *
226  math::sin( omegaLaser * ( runTime - t_rise ) + Unitless::LASER_PHASE );
227  elong.z() = Unitless::AMPLITUDE * Tpolynomial( tau ) / math::sqrt( 2.0_X ) *
228  math::cos( omegaLaser * ( runTime - t_rise ) + Unitless::LASER_PHASE );
229  }
230  }
