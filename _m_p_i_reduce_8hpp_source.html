56  /* defines if the result of the MPI operation is valid
57  *
58  * @tparam MPIMethod type of the reduction method
59  * @param method used reduction method e.g.,
60  * reduceMethods::AllReduce, reduceMethods::Reduce
61  * @return if resut of operator() is valid*/
62  template<class MPIMethod>
63  bool hasResult(const MPIMethod & method)
64  {
65  if (!isMPICommInitialized)
66  participate(true);
67  return method.hasResult(mpiRank);
68  }
69 
76  bool hasResult()
77  {
78  if (!isMPICommInitialized)
79  participate(true);
80  return this->hasResult(::pmacc::mpi::reduceMethods::AllReduce());
81  }

83  /* Activate participation for reduce algorithm.
84  * Must called from any mpi process. This function use global blocking mpi calls.
85  * @param isActive true if mpi rank should be part of reduce operation, else false
86  */
87  void participate(bool isActive)
88  {
89  /*free old communicator of init is called again*/
90  if (isMPICommInitialized)
91  {
92  MPI_CHECK(MPI_Comm_free(&comm));
93  mpiRank = -1;
94  numRanks = 0;
95  isMPICommInitialized = false;
96  }
97 
98  int countRanks;
99  MPI_CHECK(MPI_Comm_size(MPI_COMM_WORLD, &countRanks));
100  std::vector<int> reduceRank(countRanks);
101  std::vector<int> groupRanks(countRanks);
102  MPI_CHECK(MPI_Comm_rank(MPI_COMM_WORLD, &mpiRank));
103 
104  if (!isActive)
105  mpiRank = -1;
106 
107  // avoid deadlock between not finished pmacc tasks and mpi blocking collectives
108  __getTransactionEvent().waitForFinished();
109  MPI_CHECK(MPI_Allgather(&mpiRank, 1, MPI_INT, &reduceRank[0], 1, MPI_INT, MPI_COMM_WORLD));
110 
111  for (int i = 0; i < countRanks; ++i)
112  {
113  if (reduceRank[i] != -1)
114  {
115  groupRanks[numRanks] = reduceRank[i];
116  numRanks++ ;
117  }
118  }
119 
120  MPI_Group group = MPI_GROUP_NULL;
121  MPI_Group newgroup = MPI_GROUP_NULL;
122  MPI_CHECK(MPI_Comm_group(MPI_COMM_WORLD, &group));
123  MPI_CHECK(MPI_Group_incl(group, numRanks, &groupRanks[0], &newgroup));
124 
125  MPI_CHECK(MPI_Comm_create(MPI_COMM_WORLD, newgroup, &comm));
126 
127  if (mpiRank != -1)
128  {
129  MPI_CHECK(MPI_Comm_rank(comm, &mpiRank));
130  isMPICommInitialized = true;
131  }
132  MPI_CHECK(MPI_Group_free(&group));
133  MPI_CHECK(MPI_Group_free(&newgroup));
134  }

136  /* Reduce elements on cpu memory
137  * call hasResult to see if returned value is valid
138  *
139  * @param func binary functor for reduce which takes two arguments, first argument is the source and get the new reduced value.
140  * Functor must specialize the function getMPI_Op.
141  * @param dest buffer for result data
142  * @param src a class or a pointer where the reduce algorithm can access the value by operator [] (one dimension access)
143  * @param n number of elements to reduce
144  * @param method mpi method for reduce
145  *
146  */
147  template<class Functor, typename Type, class ReduceMethod >
148  HINLINE void operator()(Functor func,
149  Type* dest,
150  Type* src,
151  const size_t n,
152  const ReduceMethod method)
153  {
154  if (!isMPICommInitialized)
155  participate(true);
156  typedef Type ValueType;
157 
158  method(func,
159  dest,
160  src,
161  n * ::pmacc::mpi::getMPI_StructAsArray<ValueType > ().sizeMultiplier,
162  ::pmacc::mpi::getMPI_StructAsArray<ValueType > ().dataType,
163  ::pmacc::mpi::getMPI_Op<Functor > (),
164  comm);
165  }
166 
167  /* Reduce elements on cpu memory
168  * the default reduce method is allReduce which means that any host get the reduced value back
169  *
170  * @param func binary functor for reduce which takes two arguments, first argument is the source and get the new reduced value.
171  * Functor must specialize the function getMPI_Op.
172  * @param dest buffer for result data
173  * @param src a class or a pointer where the reduce algorithm can access the value by operator [] (one dimension access)
174  * @param n number of elements to reduce
175  *
176  * @return reduced value
177  */
178  template<class Functor, typename Type >
179  HINLINE void operator()(Functor func,
180  Type* dest,
181  Type* src,
182  const size_t n)
183  {
184  if (!isMPICommInitialized)
185  participate(true);
186  this->operator ()(func, dest, src, n, ::pmacc::mpi::reduceMethods::AllReduce());
187  }

196 };
