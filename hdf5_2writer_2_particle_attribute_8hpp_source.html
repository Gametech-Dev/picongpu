48 template< typename T_Identifier>
49 struct ParticleAttribute
50 {
60  template<typename FrameType>
61  HINLINE void operator()(
62  ThreadParams* params,
63  FrameType& frame,
64  const std::string speciesPath,
65  const uint64_t elements,
66  const uint64_t elementsOffset,
67  const uint64_t numParticlesGlobal
68  )
69  {
70 
71  typedef T_Identifier Identifier;
72  typedef typename pmacc::traits::Resolve<Identifier>::type::type ValueType;
73  const uint32_t components = GetNComponents<ValueType>::value;
74  typedef typename GetComponentsType<ValueType>::type ComponentType;
75  typedef typename PICToSplash<ComponentType>::type SplashType;
76  typedef typename PICToSplash<float_X>::type SplashFloatXType;
77 
78  const ThreadParams *threadParams = params;
79 
80  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) write species attribute: %1%") % Identifier::getName();
81 
82  SplashType splashType;
83  ColTypeDouble ctDouble;
84  ColTypeUInt32 ctUInt32;
85  SplashFloatXType splashFloatXType;
86 
87  OpenPMDName<T_Identifier> openPMDName;
88  const std::string recordPath( speciesPath + std::string("/") + openPMDName() );
89 
90  const std::string name_lookup[] = {"x", "y", "z"};
91 
92  // get the SI scaling, dimensionality and weighting of the attribute
93  OpenPMDUnit<T_Identifier> openPMDUnit;
94  std::vector<float_64> unit = openPMDUnit();
95  OpenPMDUnitDimension<T_Identifier> openPMDUnitDimension;
96  std::vector<float_64> unitDimension = openPMDUnitDimension();
97  const bool macroWeightedBool = MacroWeighted<T_Identifier>::get();
98  const uint32_t macroWeighted = (macroWeightedBool ? 1 : 0);
99  const float_64 weightingPower = WeightingPower<T_Identifier>::get();
100 
101  PMACC_ASSERT(unit.size() == components); // unitSI for each component
102  PMACC_ASSERT(unitDimension.size() == 7); // seven openPMD base units
103 
104  /* globalSlideOffset due to gpu slides between origin at time step 0
105  * and origin at current time step
106  * ATTENTION: splash offset are globalSlideOffset + picongpu offsets
107  */
108  DataSpace<simDim> globalSlideOffset;
109  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
110  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(threadParams->currentStep);
111  globalSlideOffset.y() += numSlides * localDomain.size.y();
112 
113  Dimensions splashDomainOffset(0, 0, 0);
114  Dimensions splashGlobalDomainOffset(0, 0, 0);
115 
116  Dimensions splashDomainSize(1, 1, 1);
117  Dimensions splashGlobalDomainSize(1, 1, 1);
118 
119  for (uint32_t d = 0; d < simDim; ++d)
120  {
121  splashDomainOffset[d] = threadParams->window.localDimensions.offset[d] + globalSlideOffset[d];
122  splashGlobalDomainOffset[d] = threadParams->window.globalDimensions.offset[d] + globalSlideOffset[d];
123  splashGlobalDomainSize[d] = threadParams->window.globalDimensions.size[d];
124  splashDomainSize[d] = threadParams->window.localDimensions.size[d];
125  }
126 
127  typedef typename GetComponentsType<ValueType>::type ComponentValueType;
128 
129  ComponentValueType* tmpArray = new ComponentValueType[elements];
130 
131  for (uint32_t d = 0; d < components; d++)
132  {
133  std::stringstream datasetName;
134  datasetName << recordPath;
135  if (components > 1)
136  datasetName << "/" << name_lookup[d];
137 
138  ValueType* dataPtr = frame.getIdentifier(Identifier()).getPointer();
139  #pragma omp parallel for
140  for( uint64_t i = 0; i < elements; ++i )
141  {
142  tmpArray[i] = ((ComponentValueType*)dataPtr)[i * components + d];
143  }
144 
145  // avoid deadlock between not finished pmacc tasks and mpi calls in splash/HDF5
146  __getTransactionEvent().waitForFinished();
147  threadParams->dataCollector->writeDomain(
148  threadParams->currentStep,
149  /* Dimensions for global collective buffer */
150  Dimensions(numParticlesGlobal, 1, 1),
151  /* 3D-offset in the globalSize-buffer this process writes to */
152  Dimensions(elementsOffset, 1, 1),
153  /* Type information for data */
154  splashType,
155  /* Number of dimensions (1-3) of the buffer */
156  1u,
157  /* Selection: size in src buffer */
158  splash::Selection(
159  Dimensions(elements, 1, 1)
160  ),
161  /* Name of the dataset */
162  datasetName.str().c_str(),
163  /* Global domain information */
164  splash::Domain(
165  splashGlobalDomainOffset,
166  splashGlobalDomainSize
167  ),
168  /* Domain type annotation */
169  DomainCollector::PolyType,
170  /* Buffer with data */
171  tmpArray
172  );
173 
174  threadParams->dataCollector->writeAttribute(
175  threadParams->currentStep,
176  ctDouble, datasetName.str().c_str(),
177  "unitSI", &(unit.at(d)));
178 
179  }
180  __deleteArray(tmpArray);
181 
182 
183  threadParams->dataCollector->writeAttribute(
184  params->currentStep,
185  ctDouble, recordPath.c_str(),
186  "unitDimension",
187  1u, Dimensions(7,0,0),
188  &(*unitDimension.begin()));
189 
190  threadParams->dataCollector->writeAttribute(
191  params->currentStep,
192  ctUInt32, recordPath.c_str(),
193  "macroWeighted",
194  &macroWeighted);
195 
196  threadParams->dataCollector->writeAttribute(
197  params->currentStep,
198  ctDouble, recordPath.c_str(),
199  "weightingPower",
200  &weightingPower);
201 
204  const float_X timeOffset = 0.0;
205  threadParams->dataCollector->writeAttribute(params->currentStep,
206  splashFloatXType, recordPath.c_str(),
207  "timeOffset", &timeOffset);
208 
209  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) write species attribute: %1%") %
210  Identifier::getName();
211  }
212 
213 };
