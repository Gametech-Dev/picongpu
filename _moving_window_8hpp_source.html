45  void getCurrentSlideInfo(uint32_t currentStep, bool *doSlide, float_64 *offsetFirstGPU)
46  {
47  if (doSlide)
48  *doSlide = false;
49 
50  if (offsetFirstGPU)
51  *offsetFirstGPU = 0.0;
52 
53  if (slidingWindowActive)
54  {
55  const SubGrid<simDim>& subGrid = Environment<simDim>::get().SubGrid();
56 
57  /* speed of the moving window */
58  const float_64 windowMovingSpeed = float_64(SPEED_OF_LIGHT);
59 
60  /* defines in which direction the window moves
61  *
62  * 0 == x, 1 == y , 2 == z direction
63  *
64  * note: currently only y direction is supported
65  */
66  const uint32_t moveDirection = 1;
67 
68  /* the moving window is smaller than the global domain by exactly one
69  * GPU (local domain size)
70  * \todo calculation of the globalWindowSizeInMoveDirection is constant should be
71  * only done once in it's own central object/api
72  */
73  const uint32_t globalWindowSizeInMoveDirection =
74  subGrid.getGlobalDomain().size[moveDirection] - subGrid.getLocalDomain().size[moveDirection];
75 
76  const uint32_t gpuNumberOfCellsInMoveDirection = subGrid.getLocalDomain().size[moveDirection];
77 
78  /* unit PIConGPU length */
79  const float_64 cellSizeInMoveDirection = float_64(cellSize[moveDirection]);
80 
81  const float_64 deltaWayPerStep = (windowMovingSpeed * float_64(DELTA_T));
82 
83  /* How many cells the virtual particle with speed of light is pushed forward
84  * at the begin of the simulation.
85  * The number of cells is round up thus we avoid window moves and slides
86  * depends on half cells.
87  */
88  const uint32_t virtualParticleInitialStartCell = math::ceil(
89  float_64(globalWindowSizeInMoveDirection) * (float_64(1.0) - movePoint)
90  );
91 
92  /* Is the time step when the virtual particle **passed** the GPU next to the last
93  * in the current to the next step
94  */
95  const uint32_t firstSlideStep = math::ceil(
96  float_64(subGrid.getGlobalDomain().size[moveDirection] - virtualParticleInitialStartCell) *
97  cellSizeInMoveDirection / deltaWayPerStep
98  ) - 1;
99 
100  /* way which the virtual particle must move before the window begins
101  * to move the first time [in pic length] */
102  const float_64 wayToFirstMove =
103  float_64(globalWindowSizeInMoveDirection - virtualParticleInitialStartCell) *
104  cellSizeInMoveDirection;
105  /* Is the time step when the virtual particle **passed** the moving window
106  * in the current to the next step
107  * Signed type of firstMoveStep to allow for edge case movePoint = 0.0
108  * for a moving window right from the start of the simulation.
109  */
110  const int32_t firstMoveStep = math::ceil(
111  wayToFirstMove / deltaWayPerStep
112  ) - 1;
113 
114  if (firstMoveStep <= int32_t(currentStep) )
115  {
116  /* calculate the current position of the virtual particle */
117  const float_64 virtualParticleWayPassed =
118  deltaWayPerStep * float_64(currentStep);
119  const uint32_t virtualParticleWayPassedInCells = uint32_t(
120  math::floor(virtualParticleWayPassed / cellSizeInMoveDirection)
121  );
122  const uint32_t virtualParticlePositionInCells =
123  virtualParticleWayPassedInCells + virtualParticleInitialStartCell;
124 
125  /* calculate the position of the virtual particle after the current step is calculated */
126  const float_64 nextVirtualParticleWayPassed =
127  deltaWayPerStep * float_64(currentStep + 1);
128  const uint32_t nextVirtualParticleWayPassedInCells =
129  uint32_t(math::floor(nextVirtualParticleWayPassed / cellSizeInMoveDirection));
130  /* This position is used to detect the point in time where the virtual particle
131  * moves over a GPU border.
132  */
133  const uint32_t nextVirtualParticlePositionInCells =
134  nextVirtualParticleWayPassedInCells + virtualParticleInitialStartCell;
135 
136  /* within the to be simulated time step (currentStep -> currentStep+1)
137  * the virtual particle will have reached at least the position
138  * of the cell behind the end of the initial global domain
139  * (also true for all later time steps)
140  */
141  const bool endOfInitialGlobalDomain = firstSlideStep <= currentStep;
142 
143  /* virtual particle will pass a GPU border during the current
144  * (to be simulated) time step
145  */
146  const bool virtualParticlePassesGPUBorder =
147  (nextVirtualParticlePositionInCells % gpuNumberOfCellsInMoveDirection) <
148  (virtualParticlePositionInCells % gpuNumberOfCellsInMoveDirection);
149 
150  if (endOfInitialGlobalDomain && virtualParticlePassesGPUBorder)
151  {
152  incrementSlideCounter(currentStep);
153  if (doSlide)
154  *doSlide = true;
155  }
156 
157  /* valid range for the offset is [0;GPU number of cells in move direction) */
158  if (offsetFirstGPU)
159  {
160  /* since the moving window in PIConGPU always starts on the
161  * first plane (3D) / row (2D) of GPUs in move direction, this
162  * calculation is equal to the globalWindow.offset in move direction
163  *
164  * note: also works with windowMovingSpeed > c
165  */
166  *offsetFirstGPU = nextVirtualParticlePositionInCells % gpuNumberOfCellsInMoveDirection;
167  }
168  }
169  }
170 
171  }

296  Window getWindow(uint32_t currentStep)
297  {
298  const SubGrid<simDim>& subGrid = Environment<simDim>::get().SubGrid();
299 
300  /* Without moving window, the selected window spans the whole global domain.
301  * \see https://github.com/ComputationalRadiationPhysics/picongpu/wiki/PIConGPU-domain-definitions
302  *
303  * The window's global offset is therefore zero inside the global domain.
304  * The window's global and local size are equal to the SubGrid quantities.
305  * The local window offset is the offset within the global window which
306  * is equal to the local domain offset of the GPU.
307  */
308  Window window;
309  window.localDimensions = subGrid.getLocalDomain();
310  window.globalDimensions = Selection<simDim>(subGrid.getGlobalDomain().size);
311 
312  /* moving window can only slide in y direction */
313  if (slidingWindowActive)
314  {
315  /* the moving window is smaller than the global domain by exactly one
316  * GPU (local domain size) in moving (y) direction
317  */
318  window.globalDimensions.size.y() -= subGrid.getLocalDomain().size.y();
319 
320  float_64 offsetFirstGPU = 0.0;
321  getCurrentSlideInfo(currentStep, nullptr, &offsetFirstGPU);
322 
323  /* while moving, the windows global offset within the global domain is between 0
324  * and smaller than the local domain's size in y.
325  */
326  window.globalDimensions.offset.y() = offsetFirstGPU;
327 
328  /* set top/bottom if there are no communication partners
329  * for this GPU in the respective direction */
330  const Mask comm_mask = Environment<simDim>::get().GridController().getCommunicationMask();
331  const bool isTopGpu = !comm_mask.isSet(TOP);
332  const bool isBottomGpu = !comm_mask.isSet(BOTTOM);
333 
334  if (isTopGpu)
335  {
336  /* the windows local offset within the global window is reduced
337  * by the global window offset within the global domain
338  */
339  window.localDimensions.size.y() -= offsetFirstGPU;
340  }
341  else
342  {
343  window.localDimensions.offset.y() = subGrid.getLocalDomain().offset.y() - offsetFirstGPU;
344  if (isBottomGpu)
345  {
346  window.localDimensions.size.y() = offsetFirstGPU;
347  }
348  }
349  }
350 
351  return window;
352  }
