132  {
133  /* KernelForeachLockstep is always called as kernel with three dimensions
134  * therefore we have to reduce the dimension if the mapper is only 2D or 1D.
135  */
136  auto const blockSizeShrinked = blockSize.template shrink< T_Mapper::dim >( );
137  uint32_t const domainElementCount = blockSizeShrinked.productOfComponents();
138  DataSpace< T_Mapper::dim > const domainSize( blockSizeShrinked );

140  // map to the origin of the block

142  T_Mapper::dim

144  mapper(

154  );

162  i
