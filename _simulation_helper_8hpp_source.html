138  virtual void dumpOneStep(uint32_t currentStep)
139  {
140  /* trigger notification */
141  Environment<DIM>::get().PluginConnector().notifyPlugins(currentStep);
142 
143  /* trigger checkpoint notification */
144  if(
145  !checkpointPeriod.empty() &&
146  pluginSystem::containsStep(
147  seqCheckpointPeriod,
148  currentStep
149  )
150  )
151  {
152  /* first synchronize: if something failed, we can spare the time
153  * for the checkpoint writing */
154  CUDA_CHECK(cudaDeviceSynchronize());
155  CUDA_CHECK(cudaGetLastError());
156 
157  // avoid deadlock between not finished PMacc tasks and MPI_Barrier
158  __getTransactionEvent().waitForFinished();
159 
160  GridController<DIM> &gc = Environment<DIM>::get().GridController();
161  /* can be spared for better scalings, but allows to spare the
162  * time for checkpointing if some ranks died */
163  MPI_CHECK(MPI_Barrier(gc.getCommunicator().getMPIComm()));
164 
165  /* create directory containing checkpoints */
166  if (numCheckpoints == 0)
167  {
168  Environment<DIM>::get().Filesystem().createDirectoryWithPermissions(checkpointDirectory);
169  }
170 
171  Environment<DIM>::get().PluginConnector().checkpointPlugins(currentStep,
172  checkpointDirectory);
173 
174  /* important synchronize: only if no errors occured until this
175  * point guarantees that a checkpoint is usable */
176  CUDA_CHECK(cudaDeviceSynchronize());
177  CUDA_CHECK(cudaGetLastError());
178 
179  /* avoid deadlock between not finished PMacc tasks and MPI_Barrier */
180  __getTransactionEvent().waitForFinished();
181 
182  /* \todo in an ideal world with MPI-3, this would be an
183  * MPI_Ibarrier call and this function would return a MPI_Request
184  * that could be checked */
185  MPI_CHECK(MPI_Barrier(gc.getCommunicator().getMPIComm()));
186 
187  if (gc.getGlobalRank() == 0)
188  {
189  writeCheckpointStep(currentStep);
190  }
191  numCheckpoints++;
192  }
193  }

200  void dumpTimes(TimeIntervall &tSimCalculation, TimeIntervall&, double& roundAvg, uint32_t currentStep)
201  {
202  /*dump 100% after simulation*/
203  if (output && progress && (currentStep % showProgressAnyStep) == 0)
204  {
205  tSimCalculation.toggleEnd();
206  std::cout << std::setw(3) <<
207  uint16_t(
208  double( currentStep ) /
209  double( Environment<>::get().SimulationDescription().getRunSteps() ) *
210  100.
211  ) <<
212  " % = " << std::setw(8) << currentStep <<
213  " | time elapsed:" <<
214  std::setw(25) << tSimCalculation.printInterval() << " | avg time per step: " <<
215  TimeIntervall::printeTime(roundAvg / (double) showProgressAnyStep) << std::endl;
216  std::cout.flush();
217 
218  roundAvg = 0.0; //clear round avg timer
219  }

226  void startSimulation()
227  {
228  init();
229 
230  // translate checkpointPeriod string into checkpoint intervals
231  seqCheckpointPeriod = pluginSystem::toTimeSlice( checkpointPeriod );
232 
233  for (uint32_t nthSoftRestart = 0; nthSoftRestart <= softRestarts; ++nthSoftRestart)
234  {
235  resetAll(0);
236  uint32_t currentStep = fillSimulation();
237  Environment<>::get().SimulationDescription().setCurrentStep( currentStep );
238 
239  tInit.toggleEnd();
240  if (output)
241  {
242  std::cout << "initialization time: " << tInit.printInterval() <<
243  " = " <<
244  (int) (tInit.getInterval() / 1000.) << " sec" << std::endl;
245  }
246 
247  TimeIntervall tSimCalculation;
248  TimeIntervall tRound;
249  double roundAvg = 0.0;
250 
251  /* Since in the main loop movingWindow is called always before the dump, we also call it here for consistency.
252  * This becomes only important, if movingWindowCheck does more than merely checking for a slide.
253  * TO DO in a new feature: Turn this into a general hook for pre-checks (window slides are just one possible action).
254  */
255  movingWindowCheck(currentStep);
256 
257  /* dump initial step if simulation starts without restart */
258  if (!restartRequested)
259  {
260  dumpOneStep(currentStep);
261  }
262 
263  /* dump 0% output */
264  dumpTimes(tSimCalculation, tRound, roundAvg, currentStep);
265 
266 
272  while (currentStep < Environment<>::get().SimulationDescription().getRunSteps())
273  {
274  tRound.toggleStart();
275  runOneStep(currentStep);
276  tRound.toggleEnd();
277  roundAvg += tRound.getInterval();
278 
279  /* NEXT TIMESTEP STARTS HERE */
280  currentStep++;
281  Environment<>::get().SimulationDescription().setCurrentStep( currentStep );
282  /* output times after a round */
283  dumpTimes(tSimCalculation, tRound, roundAvg, currentStep);
284 
285  movingWindowCheck(currentStep);
286  /* dump at the beginning of the simulated step */
287  dumpOneStep(currentStep);
288  }
289 
290  // simulatation end
291  Environment<>::get().Manager().waitForAllTasks();
292 
293  tSimCalculation.toggleEnd();
294 
295  if (output)
296  {
297  std::cout << "calculation simulation time: " <<
298  tSimCalculation.printInterval() << " = " <<
299  (int) (tSimCalculation.getInterval() / 1000.) << " sec" << std::endl;
300  }
301 
302  } // softRestarts loop
303  }

412  void writeCheckpointStep(const uint32_t checkpointStep)
413  {
414  std::ofstream file;
415  const std::string checkpointMasterFile =
416  checkpointDirectory + std::string("/") + CHECKPOINT_MASTER_FILE;
417 
418  file.open(checkpointMasterFile.c_str(), std::ofstream::app);
419 
420  if (!file)

423  file << checkpointStep << std::endl;

427 protected:
433  std::vector<uint32_t> readCheckpointMasterFile()
434  {
435  std::vector<uint32_t> checkpoints;
436 
437  const std::string checkpointMasterFile =
438  this->restartDirectory + std::string("/") + this->CHECKPOINT_MASTER_FILE;
439 
440  if (!boost::filesystem::exists(checkpointMasterFile))
441  return checkpoints;
442 
443  std::ifstream file(checkpointMasterFile.c_str());
444 
445  /* read each line */
446  std::string line;
447  while (std::getline(file, line))
448  {
449  if (line.empty())
450  continue;
451  try
452  {
453  checkpoints.push_back(boost::lexical_cast<uint32_t>(line));
454  }
455  catch (boost::bad_lexical_cast const&)
456  {
457  std::cerr << "Warning: checkpoint master file contains invalid data ("
458  << line << ")" << std::endl;
459  }
460  }
461 
462  return checkpoints;
463  }
