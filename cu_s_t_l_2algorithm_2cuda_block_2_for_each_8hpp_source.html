43 #define FOREACH_OPERATOR(Z, N, _) \
44  /* < , typename C0, ..., typename C(N-1) , > */ \
45  template<typename Zone, BOOST_PP_ENUM_PARAMS(N, typename C), typename Functor, typename T_Acc> \
46  /* ( C0 c0, ..., C(N-1) c(N-1) , ) */ \
47  DINLINE void operator()(T_Acc const & acc, Zone, BOOST_PP_ENUM_BINARY_PARAMS(N, C, c), const Functor& functor) \
48  { \
49  const int dataVolume = math::CT::volume<typename Zone::Size>::type::value; \
50  const int blockVolume = math::CT::volume<BlockDim>::type::value; \
51  \
52  typedef typename math::Int<Zone::dim> PosType; \
53  using namespace pmacc::algorithms::precisionCast; \
54  \
55  for(int i = this->linearThreadIdx; i < dataVolume; i += blockVolume) \
56  { \
57  PosType pos = Zone::Offset::toRT() + \
58  precisionCast<typename PosType::type>( \
59  math::MapToPos<Zone::dim>()( typename Zone::Size(), i ) ); \
60  functor(acc, BOOST_PP_ENUM(N, SHIFTACCESS_CURSOR, _)); \
61  } \
62  }
