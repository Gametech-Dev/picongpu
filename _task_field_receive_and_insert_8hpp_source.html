54 
55  for (uint32_t i = 1; i < traits::NumberOfExchanges<Dim>::value; ++i)
56  {
57  if (m_buffer.getGridBuffer().hasReceiveExchange(i))
58  {
59  __startTransaction(serialEvent);
60  FieldFactory::getInstance().createTaskFieldReceiveAndInsertExchange(m_buffer, i);

67  bool executeIntern()
68  {
69  switch (m_state)
70  {
71  case Init:
72  break;
73  case WaitForReceived:
74  if (nullptr == Environment<>::get().Manager().getITaskIfNotFinished(m_tmpEvent.getTaskId()))
75  {
76  m_state = Insert;
77  }
78  break;
79  case Insert:
80  m_state = Wait;
81  __startTransaction();
82  for (uint32_t i = 1; i < traits::NumberOfExchanges<Dim>::value; ++i)
83  {
84  if (m_buffer.getGridBuffer().hasReceiveExchange(i))
85  {
86  m_buffer.insertField(i);
87  }
88  }
89  m_tmpEvent = __endTransaction();
90  m_state = WaitInsertFinished;
91  break;
92  case Wait:
93  break;
94  case WaitInsertFinished:
95  if (nullptr == Environment<>::get().Manager().getITaskIfNotFinished(m_tmpEvent.getTaskId()))
96  {
97  m_state = Finish;
98  return true;
99  }
100  break;
101  case Finish:
102  return true;
103  default:
104  return false;
105  }
