51 class RestartFieldLoader
52 {
53 public:
54  template<class Data>
55  static void loadField(Data& field, const uint32_t numComponents, std::string objectName, ThreadParams *params)
56  {
57  log<picLog::INPUT_OUTPUT > ("Begin loading field '%1%'") % objectName;
58 
59  const std::string name_lookup_tpl[] = {"x", "y", "z", "w"};
60  const DataSpace<simDim> field_guard = field.getGridLayout().getGuard();
61 
62  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
63 
64  field.getHostBuffer().setValue(float3_X::create(0.0));
65 
66  DataSpace<simDim> domain_offset = localDomain.offset;
67 
68  DataSpace<simDim> local_domain_size = params->window.localDimensions.size;
69 
70  auto destBox = field.getHostBuffer().getDataBox();
71  for (uint32_t n = 0; n < numComponents; ++n)
72  {
73  // Read the subdomain which belongs to our mpi position.
74  // The total grid size must match the grid size of the stored data.
75  log<picLog::INPUT_OUTPUT > ("ADIOS: Read from domain: offset=%1% size=%2%") %
76  domain_offset % local_domain_size;
77 
78  std::stringstream datasetName;
79  datasetName << params->adiosBasePath << ADIOS_PATH_FIELDS << objectName;
80  if (numComponents > 1)
81  datasetName << "/" << name_lookup_tpl[n];
82 
83  log<picLog::INPUT_OUTPUT > ("ADIOS: Read from field '%1%'") %
84  datasetName.str();
85 
86  ADIOS_VARINFO* varInfo = adios_inq_var( params->fp, datasetName.str().c_str() );
87  if( varInfo == nullptr )
88  {
89  std::string errMsg( adios_errmsg() );
90  if( errMsg.empty() ) errMsg = '\n';
91  std::stringstream s;
92  s << "ADIOS: error at adios_inq_var '"
93  << "' (" << adios_errno << ") in "
94  << __FILE__ << ":" << __LINE__ << " " << errMsg;
95  throw std::runtime_error(s.str());
96  }
97  uint64_t start[varInfo->ndim];
98  uint64_t count[varInfo->ndim];
99  for(int d = 0; d < varInfo->ndim; ++d)
100  {
101  /* \see adios_define_var: z,y,x in C-order */
102  start[d] = domain_offset.revert()[d];
103  count[d] = local_domain_size.revert()[d];
104  }
105 
106  ADIOS_SELECTION* fSel = adios_selection_boundingbox( varInfo->ndim, start, count );
107 
108  /* specify what we want to read, but start reading at below at
109  * `adios_perform_reads` */
110  log<picLog::INPUT_OUTPUT > ("ADIOS: Allocate %1% elements") %
111  local_domain_size.productOfComponents();
112 
114  float_X* field_container = new float_X[local_domain_size.productOfComponents()];
115  /* magic parameters (0, 1): `from_step` (not used in streams), `nsteps` to read (must be 1 for stream) */
116  log<picLog::INPUT_OUTPUT > ("ADIOS: Schedule read from field (%1%, %2%, %3%, %4%)") %
117  params->fp % fSel % datasetName.str() % (void*)field_container;
118 
119  // avoid deadlock between not finished pmacc tasks and mpi calls in adios
120  __getTransactionEvent().waitForFinished();
121  ADIOS_CMD(adios_schedule_read( params->fp, fSel, datasetName.str().c_str(), 0, 1, (void*)field_container ));
122 
123  /* start a blocking read of all scheduled variables */
124  ADIOS_CMD(adios_perform_reads( params->fp, 1 ));
125 
126  int elementCount = params->window.localDimensions.size.productOfComponents();
127 
128  #pragma omp parallel for
129  for (int linearId = 0; linearId < elementCount; ++linearId)
130  {
131  /* calculate index inside the moving window domain which is located on the local grid*/
132  DataSpace<simDim> destIdx = DataSpaceOperations<simDim>::map(params->window.localDimensions.size, linearId);
133  /* jump over guard and local sliding window offset*/
134  destIdx += field_guard + params->localWindowToDomainOffset;
135 
136  destBox(destIdx)[n] = field_container[linearId];
137  }
138 
139  __deleteArray(field_container);
140  adios_selection_delete(fSel);
141  adios_free_varinfo(varInfo);
142  }
143 
144  field.hostToDevice();
145 
146  __getTransactionEvent().waitForFinished();
147 
148  log<picLog::INPUT_OUTPUT > ("ADIOS: Read from domain: offset=%1% size=%2%") %
149  domain_offset % local_domain_size;
150  log<picLog::INPUT_OUTPUT > ("ADIOS: Finished loading field '%1%'") % objectName;
151  }
152 
153 };
