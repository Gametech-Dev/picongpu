46  bool restoreTxtFile( std::ofstream& outFile, std::string filename,
47  uint32_t restartStep, const std::string restartDirectory )
48  {
49  /* get restart time step as string */
50  std::stringstream sStep;
51  sStep << restartStep;
52 
53  /* set location of restart file and output file */
54  path src( restartDirectory + std::string("/") + filename +
55  std::string(".") + sStep.str() );
56  path dst( filename );
57 
58  /* check whether restart file exists */
59  if( !boost::filesystem::exists( src ) )
60  {
61  /* restart file does not exists */
62  log<picLog::INPUT_OUTPUT> ("Plugin restart file: %1% was not found. \
63  --> Starting plugin from current time step.") % src;
64  return true;
65  }
66  else
67  {
68  /* restart file found - fix output file created at restart */
69  if( outFile.is_open() )
70  outFile.close();
71 
72  copy_file( src,
73  dst,
74  copy_option::overwrite_if_exists );
75 
76  outFile.open( filename.c_str(), std::ofstream::out | std::ostream::app );
77  if( !outFile )
78  {
79  std::cerr << "[Plugin] Can't open file '" << filename
80  << "', output disabled" << std::endl;
81  return false;
82  }
83  return true;
84  }
85  }
