41 /* count particles
42  *
43  * it is allowed to call this kernel on frames with holes (without calling fillAllGAps before)
44  *
45  * @tparam T_numWorkers number of workers
46  */
47 template< uint32_t T_numWorkers >
48 struct KernelCountParticles

80  using namespace mappings::threads;
81 
82  using Frame = typename T_PBox::FrameType;
83  using FramePtr = typename T_PBox::FramePtr;
84  constexpr uint32_t dim = T_Mapping::Dim;
85  constexpr uint32_t frameSize = math::CT::volume< typename Frame::SuperCellSize >::type::value;
86  constexpr uint32_t numWorkers = T_numWorkers;

89  acc,
90  frame,
91  FramePtr

100  particlesInSuperCell,
101  lcellId_t
102  );
103 
104  using SuperCellSize = typename T_Mapping::SuperCellSize;
105 
106  DataSpace< dim > const threadIndex( threadIdx );
107  uint32_t const workerIdx = static_cast< uint32_t >(
108  DataSpaceOperations< dim >::template map< SuperCellSize >( threadIndex )
109  );
110 
111  DataSpace< dim > const superCellIdx( mapper.getSuperCellIndex( DataSpace< dim >( blockIdx ) ) );
112 
113  ForEachIdx<
114  IdxConfig<
115  1,
116  numWorkers
117  >
118  > onlyMaster{ workerIdx };
119 
120  onlyMaster(
121  [&](
122  uint32_t const,
123  uint32_t const
124  )
125  {
126  frame = pb.getLastFrame( superCellIdx );
127  particlesInSuperCell = pb.getSuperCell( superCellIdx ).getSizeLastFrame( );
128  counter = 0;
129  }
130  );
131 
132  __syncthreads( );
133 
134  if( !frame.isValid() )
135  return; //end kernel if we have no frames
136  filter.setSuperCellPosition(
137  ( superCellIdx - mapper.getGuardingSuperCells( ) ) *
138  mapper.getSuperCellSize( )
139  );
140 
141  auto accParFilter = parFilter(
142  acc,
143  superCellIdx - mapper.getGuardingSuperCells( ),
144  WorkerCfg< numWorkers >{ workerIdx }
145  );
146 
147  ForEachIdx<
148  IdxConfig<
149  frameSize,
150  numWorkers
151  >
152  > forEachParticle( workerIdx );
153 
154  while( frame.isValid( ) )
155  {
156  forEachParticle(
157  [&](
158  uint32_t const linearIdx,
159  uint32_t const idx
160  )
161  {
162  if( linearIdx < particlesInSuperCell )
163  {
164  bool const useParticle = filter(
165  *frame,
166  linearIdx
167  );
168  if( useParticle )
169  {
170  auto parSrc = ( frame[ linearIdx ] );
171  if(
172  accParFilter(
173  acc,
174  parSrc
175  )
176  )
177  nvidia::atomicAllInc( acc, &counter, ::alpaka::hierarchy::Threads{} );
178  }
179  }
180  }
181  );
182 
183  __syncthreads( );
184 
185  onlyMaster(
186  [&](
187  uint32_t const,
188  uint32_t const
189  )
190  {
191  frame = pb.getPreviousFrame( frame );
192  particlesInSuperCell = frameSize;
193  }
194  );
195 
196  __syncthreads( );
197  }
198 
199  onlyMaster(
200  [&](
201  uint32_t const,
202  uint32_t const
203  )
204  {
205 
206  atomicAdd(
207  gCounter,
208  static_cast< uint64_cu >( counter ),
209  ::alpaka::hierarchy::Blocks{}
210  );
211  }
212  );
213  }
