102 template<class T_Collection, typename T>

104 checkDuplicate(const T_Collection& col, const T& value, bool shouldFind)
105 {
106  if((std::find(col.begin(), col.end(), value) != col.end()) != shouldFind)
107  {
108  boost::test_tools::predicate_result res(false);
109  if(shouldFind)
110  res.message() << "Value not found found: ";
111  else

113  res.message() << value << ". Values=[";
114  for(typename T_Collection::const_iterator it = col.begin(); it != col.end(); ++it)
115  res.message() << *it << ",";
116  res.message() << "]";
117  return res;
118  }
119 
120  return true;

124 template<unsigned T_dim>
125 struct IdProviderTest
126 {
127  void operator()()
128  {
129  using namespace ::pmacc;
130 
131  constexpr uint32_t numBlocks = 4;
132  constexpr uint32_t numIdsPerBlock = 64;
133  constexpr uint32_t numThreads = numBlocks * numIdsPerBlock;
134  constexpr uint32_t numIdsPerThread = 2;
135  constexpr uint32_t numIds = numThreads * numIdsPerThread;
136 
137  using IdProvider = IdProvider< T_dim >;
138  IdProvider::init();
139  // Check initial state
140  typename IdProvider::State state = IdProvider::getState();
141  BOOST_REQUIRE_EQUAL(state.startId, state.nextId);
142  BOOST_REQUIRE_EQUAL(state.maxNumProc, 1u);
143  BOOST_REQUIRE(!IdProvider::isOverflown());
144  std::set<uint64_t> ids;
145  BOOST_REQUIRE_EQUAL(IdProvider::getNewIdHost(), state.nextId);
146  // Generate some IDs using the function
147  for(int i=0; i<numIds; i++)
148  {
149  const uint64_t newId = IdProvider::getNewIdHost();
150  BOOST_REQUIRE(checkDuplicate(ids, newId, false));
151  ids.insert(newId);
152  }
153  // Reset the state
154  IdProvider::setState(state);
155  BOOST_REQUIRE_EQUAL(IdProvider::getNewIdHost(), state.nextId);
156  // Generate the same IDs on the device
157  HostDeviceBuffer< uint64_t, 1 > idBuf(numIds);
158  constexpr uint32_t numWorkers = traits::GetNumWorkers<
159  numIdsPerBlock
160  >::value;
161  PMACC_KERNEL( GenerateIds<
162  numWorkers,
163  numIdsPerBlock,
164  IdProvider
165  >{ })(
166  numBlocks,
167  numWorkers
168  )(
169  idBuf.getDeviceBuffer().getDataBox(),
170  numThreads,
171  numIdsPerThread
172  );
173  idBuf.deviceToHost();
174  BOOST_REQUIRE_EQUAL(numIds, ids.size());
175  auto hostBox = idBuf.getHostBuffer().getDataBox();
176  // Make sure they are the same
177  for(uint32_t i=0; i<numIds; i++)
178  {
179  BOOST_REQUIRE(checkDuplicate(ids, hostBox(i), true));
180  }
181  }
182 };
