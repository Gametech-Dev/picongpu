85  template<typename ParticlesFrame, typename T_Acc>
86  DINLINE void operator()(const T_Acc& acc, ParticlesFrame& particlesFrame, const uint32_t linearThreadIdx)
87  {
88  const float3_X mom = particlesFrame[linearThreadIdx][momentum_];
89  const float_X mom2 = math::dot(mom, mom);
90  float3_X dirVec = mom * math::rsqrt(mom2);
91 
92  /* rotate dirVec into the calorimeter frame. This coordinate transformation
93  * is performed by a matrix vector multiplication. */
94  using namespace pmacc::algorithms::math;
95  dirVec = float3_X(dot(this->calorimeterFrameVecX, dirVec),
96  dot(this->calorimeterFrameVecY, dirVec),
97  dot(this->calorimeterFrameVecZ, dirVec));
98 
99  /* convert dirVec to yaw and pitch */
100  const float_X yaw = atan2(dirVec.x(), dirVec.y());
101  const float_X pitch = asin(dirVec.z());
102 
103  if(abs(yaw) < this->maxYaw && abs(pitch) < this->maxPitch)
104  {
105  const float2_X calorimeterPos = particleCalorimeter::mapYawPitchToNormedRange(
106  yaw, pitch, this->maxYaw, this->maxPitch);
107 
108  // yaw
109  int32_t yawBin = calorimeterPos.x() * static_cast<float_X>(numBinsYaw);
110  // catch out-of-range values
111  yawBin = yawBin >= numBinsYaw ? numBinsYaw - 1 : yawBin;
112  yawBin = yawBin < 0 ? 0 : yawBin;
113 
114  // pitch
115  int32_t pitchBin = calorimeterPos.y() * static_cast<float_X>(numBinsPitch);
116  // catch out-of-range values
117  pitchBin = pitchBin >= numBinsPitch ? numBinsPitch - 1 : pitchBin;
118  pitchBin = pitchBin < 0 ? 0 : pitchBin;
119 
120  // energy
121  const float_X weighting = particlesFrame[linearThreadIdx][weighting_];
122  const float_X normedWeighting = weighting /
123  static_cast<float_X>(particles::TYPICAL_NUM_PARTICLES_PER_MACROPARTICLE);
124  const auto particle = particlesFrame[linearThreadIdx];
125  const float_X mass = attribute::getMass(weighting, particle);
126  const float_X energy = KinEnergy<>()(mom, mass) / weighting;
127 
128  int32_t energyBin = 0;
129  if(this->numBinsEnergy > 1)
130  {
131  const int32_t numBinsOutOfRange = 2;
132  energyBin = math::float2int_rd(((logScale ? log10(energy) : energy) - minEnergy) /
133  (maxEnergy - minEnergy) * static_cast<float_X>(this->numBinsEnergy - numBinsOutOfRange)) + 1;
134 
135  // all entries larger than maxEnergy go into last bin
136  energyBin = energyBin < this->numBinsEnergy ? energyBin : this->numBinsEnergy - 1;
137 
138  // all entries smaller than minEnergy go into bin zero
139  energyBin = energyBin > 0 ? energyBin : 0;
140  }
141 
142  atomicAdd( &(*this->calorimeterCur(yawBin, pitchBin, energyBin)),
143  energy * normedWeighting, ::alpaka::hierarchy::Threads{});
144  }
145  }
146 };
