66 template< typename T_SpeciesFilter >
67 struct ADIOSCountParticles
68 {
69 public:
70 
71  typedef typename T_SpeciesFilter::Species ThisSpecies;
72  typedef typename ThisSpecies::FrameType FrameType;
73  typedef typename FrameType::ParticleDescription ParticleDescription;
74  typedef typename FrameType::ValueTypeSeq ParticleAttributeList;
75 
76  /* delete multiMask and localCellIdx in adios particle*/
77  typedef bmpl::vector<multiMask,localCellIdx> TypesToDelete;
78  typedef typename RemoveFromSeq<ParticleAttributeList, TypesToDelete>::type ParticleCleanedAttributeList;
79 
80  /* add totalCellIdx for adios particle*/
81  typedef typename MakeSeq<
82  ParticleCleanedAttributeList,
83  totalCellIdx
84  >::type ParticleNewAttributeList;
85 
86  typedef
87  typename ReplaceValueTypeSeq<ParticleDescription, ParticleNewAttributeList>::type
88  NewParticleDescription;
89 
90  typedef Frame<OperatorCreateVectorBox, NewParticleDescription> AdiosFrameType;
91 
92  HINLINE void operator()(ThreadParams* params)
93  {
94  DataConnector &dc = Environment<>::get().DataConnector();
95  GridController<simDim>& gc = Environment<simDim>::get().GridController();
96  uint64_t mpiSize = gc.getGlobalSize();
97  uint64_t mpiRank = gc.getGlobalRank();
98 
99  const std::string speciesGroup( T_SpeciesFilter::getName() + "/" );
100  const std::string speciesPath( params->adiosBasePath +
101  std::string(ADIOS_PATH_PARTICLES) + speciesGroup );
102 
103  /* load particle without copy particle data to host */
104  auto speciesTmp = dc.get< ThisSpecies >( ThisSpecies::FrameType::getName(), true );
105  // enforce that the filter interface is fulfilled
106  particles::filter::IUnary< typename T_SpeciesFilter::Filter > particleFilter{ params->currentStep };
107  /* count total number of particles on the device */
108  uint64_cu totalNumParticles = 0;
109  totalNumParticles = pmacc::CountParticles::countOnDevice < CORE + BORDER > (
110  *speciesTmp,
111  *(params->cellDescription),
112  params->localWindowToDomainOffset,
113  params->window.localDimensions.size,
114  particleFilter);
115 
116  /* MPI_Allgather to compute global size and my offset */
117  uint64_t myNumParticles = totalNumParticles;
118  uint64_t allNumParticles[mpiSize];
119  uint64_t globalNumParticles = 0;
120  uint64_t myParticleOffset = 0;
121 
122  // avoid deadlock between not finished pmacc tasks and mpi blocking collectives
123  __getTransactionEvent().waitForFinished();
124  MPI_CHECK(MPI_Allgather(
125  &myNumParticles, 1, MPI_UNSIGNED_LONG_LONG,
126  allNumParticles, 1, MPI_UNSIGNED_LONG_LONG,
127  gc.getCommunicator().getMPIComm()));
128 
129  for (uint64_t i = 0; i < mpiSize; ++i)
130  {
131  globalNumParticles += allNumParticles[i];
132  if (i < mpiRank)
133  myParticleOffset += allNumParticles[i];
134  }
135 
136  /* iterate over all attributes of this species */
137  ForEach<typename AdiosFrameType::ValueTypeSeq, adios::ParticleAttributeSize<bmpl::_1> > attributeSize;
138  attributeSize(params, speciesGroup, myNumParticles, globalNumParticles, myParticleOffset);
139 
140  /* TODO: constant particle records */
141 
142  /* openPMD ED-PIC: additional attributes */
143  traits::PICToAdios<float_64> adiosDoubleType;
144  const float_64 particleShape( GetShape<ThisSpecies>::type::support - 1 );
145  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
146  "particleShape", speciesPath.c_str(),
147  adiosDoubleType.type, 1, (void*)&particleShape ));
148 
149  traits::GetSpeciesFlagName<ThisSpecies, current<> > currentDepositionName;
150  const std::string currentDeposition( currentDepositionName() );
151  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
152  "currentDeposition", speciesPath.c_str(),
153  adios_string, 1, (void*)currentDeposition.c_str() ));
154 
155  traits::GetSpeciesFlagName<ThisSpecies, particlePusher<> > particlePushName;
156  const std::string particlePush( particlePushName() );
157  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
158  "particlePush", speciesPath.c_str(),
159  adios_string, 1, (void*)particlePush.c_str() ));
160 
161  traits::GetSpeciesFlagName<ThisSpecies, interpolation<> > particleInterpolationName;
162  const std::string particleInterpolation( particleInterpolationName() );
163  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
164  "particleInterpolation", speciesPath.c_str(),
165  adios_string, 1, (void*)particleInterpolation.c_str() ));
166 
167  const std::string particleSmoothing( "none" );
168  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
169  "particleSmoothing", speciesPath.c_str(),
170  adios_string, 1, (void*)particleSmoothing.c_str() ));
171 
172  /* define adios var for species index/info table */
173  {
174  const uint64_t localTableSize = 5;
175  traits::PICToAdios<uint64_t> adiosIndexType;
176 
177  const char* path = nullptr;
178  int64_t adiosSpeciesIndexVar = defineAdiosVar<DIM1>(
179  params->adiosGroupHandle,
180  (speciesPath + "particles_info").c_str(),
181  path,
182  adiosIndexType.type,
183  pmacc::math::UInt64<DIM1>(localTableSize),
184  pmacc::math::UInt64<DIM1>(localTableSize * uint64_t(gc.getGlobalSize()) ),
185  pmacc::math::UInt64<DIM1>(localTableSize * uint64_t(gc.getGlobalRank()) ),
186  true,
187  params->adiosCompression);
188 
189  params->adiosSpeciesIndexVarIds.push_back(adiosSpeciesIndexVar);
190 
191  params->adiosGroupSize += sizeof(uint64_t) * localTableSize * gc.getGlobalSize();
192  }
193  }
194 };
