37 template<typename T_Scalar, typename T_Attribute = uint64_t>
38 struct WriteNDScalars
39 {
40  void operator()(ThreadParams& params,
41  const std::string& name, T_Scalar value,
42  const std::string& attrName = "", T_Attribute attribute = T_Attribute())
43  {
44  log<picLog::INPUT_OUTPUT>("HDF5: write %1%D scalars: %2%") % simDim % name;
45 
46  // Size over all processes
47  Dimensions globalSize(1, 1, 1);
48  // Offset for this process
49  Dimensions localOffset(0, 0, 0);
50  // Offset for all processes
51  Dimensions globalOffset(0, 0, 0);
52 
53  for (uint32_t d = 0; d < simDim; ++d)
54  {
55  globalSize[d] = Environment<simDim>::get().GridController().getGpuNodes()[d];
56  localOffset[d] = Environment<simDim>::get().GridController().getPosition()[d];
57  }
58 
59  Dimensions localSize(1, 1, 1);
60 
61  // avoid deadlock between not finished pmacc tasks and mpi calls in adios
62  __getTransactionEvent().waitForFinished();
63 
64  typename traits::PICToSplash<T_Scalar>::type splashType;
65  params.dataCollector->writeDomain(params.currentStep, /* id == time step */
66  globalSize, /* total size of dataset over all processes */
67  localOffset, /* write offset for this process */
68  splashType, /* data type */
69  simDim, /* NDims spatial dimensionality of the field */
70  splash::Selection(localSize), /* data size of this process */
71  name.c_str(), /* data set name */
72  splash::Domain(
73  globalOffset, /* offset of the global domain */
74  globalSize /* size of the global domain */
75  ),
76  DomainCollector::GridType,
77  &value);
78 
79  if(!attrName.empty())
80  {
81  /*simulation attribute for data*/
82  typename traits::PICToSplash<T_Attribute>::type attType;
83 
84  log<picLog::INPUT_OUTPUT>("HDF5: write attribute %1% for scalars: %2%") % attrName % name;
85  params.dataCollector->writeAttribute(params.currentStep,
86  attType, name.c_str(),
87  attrName.c_str(), &attribute);
88  }
89  }
90 };

99 template<typename T_Scalar, typename T_Attribute = uint64_t>
100 struct ReadNDScalars
101 {
102  void operator()(ThreadParams& params,
103  const std::string& name, T_Scalar* value,
104  const std::string& attrName = "", T_Attribute* attribute = nullptr)
105  {
106  log<picLog::INPUT_OUTPUT>("HDF5: read %1%D scalars: %2%") % simDim % name;
107 
108  Dimensions domain_offset(0, 0, 0);
109  for (uint32_t d = 0; d < simDim; ++d)
110  domain_offset[d] = Environment<simDim>::get().GridController().getPosition()[d];
111 
112  // avoid deadlock between not finished pmacc tasks and mpi calls in adios
113  __getTransactionEvent().waitForFinished();
114 
115  DomainCollector::DomDataClass data_class;
116  DataContainer *dataContainer =
117  params.dataCollector->readDomain(params.currentStep,
118  name.c_str(),
119  Domain(domain_offset, Dimensions(1, 1, 1)),
120  &data_class);
121 
122  typename traits::PICToSplash<T_Scalar>::type splashType;
123  *value = *static_cast<T_Scalar*>(dataContainer->getIndex(0)->getData());
124  __delete(dataContainer);
125 
126  if(!attrName.empty())
127  {
128  log<picLog::INPUT_OUTPUT>("HDF5: read attribute %1% for scalars: %2%") % attrName % name;
129  params.dataCollector->readAttributeInfo(params.currentStep, name.c_str(), attrName.c_str()).read(attribute, sizeof(T_Attribute));
130  log<picLog::INPUT_OUTPUT>("HDF5: attribute %1% = %2%") % attrName % *attribute;
131  }
132  }
133 };
