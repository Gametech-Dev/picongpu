55  HDINLINE float_X levichivita( const unsigned int i, const unsigned int j, const unsigned int k )
56  {
57  if( i == j || j == k || i == k ) return float_X(0.0);
58 
59  if( i == x && j == y ) return float_X(1.0);
60  if( i == z && j == x ) return float_X(1.0);
61  if( i == y && j == z ) return float_X(1.0);
62 
63  return float_X(-1.0);
64  }

66  template<
67  typename T_FunctorFieldE,
68  typename T_FunctorFieldB,
69  typename T_Particle,
70  typename T_Pos
71  >
72  HDINLINE void operator()(
73  const T_FunctorFieldB functorBField, /* at t=0 */
74  const T_FunctorFieldE functorEField, /* at t=0 */
75  T_Particle & particle,
76  T_Pos & pos, /* at t=0 */
77  const uint32_t
78  )
79  {
80  float_X const weighting = particle[ weighting_ ];
81  float_X const mass = attribute::getMass( weighting, particle );
82  float_X const charge = attribute::getCharge( weighting, particle );
83 
84  using MomType = momentum::type;
85  MomType mom = particle[ momentum_ ];
86 
87  auto bField = functorBField(pos);
88  auto eField = functorEField(pos);
89 
90  Gamma gammaCalc;
91  Velocity velocityCalc;
92  const float_X epsilon = 1.0e-6;
93  const float_X deltaT = DELTA_T;
94 
95  //const float3_X velocity_atMinusHalf = velocity(mom, mass);
96  const float_X gamma = gammaCalc( mom, mass );
97 
98  const MomType mom_old = mom;
99 
100  const float_X B2 = math::abs2( bField );
101  const float_X B = math::abs( bField );
102 
103  if( B2 > epsilon )
104  {
105  trigo_X sinres;
106  trigo_X cosres;
107  trigo_X arg = B * charge * deltaT / gamma;
108  math::sincos( arg, sinres, cosres );
109 
110  mom.x() = bField.x() * bField.x() * ( eField.x() * charge * deltaT + mom_old.x() );
111  mom.y() = bField.y() * bField.y() * ( eField.y() * charge * deltaT + mom_old.y() );
112  mom.z() = bField.z() * bField.z() * ( eField.z() * charge * deltaT + mom_old.z() );
113 
114 #define SUM_PLINE1(I,J,K) bField.J() * ( -levichivita(I,J,K) * gamma * eField.K() + bField.I() * ( eField.J() * charge * deltaT + mom_old.J() ) )
115 #define SUM_PLINE2(I,J,K) -bField.J() * ( -levichivita(I,J,K) * gamma * eField.K() + bField.I() * mom_old.J() - bField.J() * mom_old.I() )
116 #define SUM_PLINE3(I,J,K) bField.J() * bField.J() * gamma * eField.I() - bField.I() * bField.J() * gamma * eField.J() + levichivita(I,J,K) * mom_old.J() * bField.K() * B2
117 
118  mom.x() += FOR_JK_NOT_I( x, y, z, SUM_PLINE1 );
119  mom.x() += float_X(cosres ) * ( FOR_JK_NOT_I( x, y, z, SUM_PLINE2 ) );
120  mom.x() += float_X(sinres ) / B * ( FOR_JK_NOT_I( x, y, z, SUM_PLINE3 ) );
121 
122  mom.y() += FOR_JK_NOT_I( y, z, x, SUM_PLINE1 );
123  mom.y() += float_X(cosres ) * ( FOR_JK_NOT_I( y, z, x, SUM_PLINE2 ) );
124  mom.y() += float_X(sinres ) / B * ( FOR_JK_NOT_I( y, z, x, SUM_PLINE3 ) );
125 
126  mom.z() += FOR_JK_NOT_I( z, x, y, SUM_PLINE1 );
127  mom.z() += float_X(cosres ) * ( FOR_JK_NOT_I( z, x, y, SUM_PLINE2 ) );
128  mom.z() += float_X(sinres ) / B * ( FOR_JK_NOT_I( z, x, y, SUM_PLINE3 ) );
129 
130  mom *= float_X(1.0) / B2;
131  }
132  else
133  {
134  mom += eField * charge * deltaT;
135  }
136 
137  particle[ momentum_ ] = mom;
138 
139  float3_X dr(float3_X::create(0.0));
140 
141  // old spacial change calculation: linear step
142  if( TrajectoryInterpolation == LINEAR )
143  {
144  const float3_X vel = velocityCalc( mom, mass );
145  dr = float3_X( vel.x() * deltaT / CELL_WIDTH,
146  vel.y() * deltaT / CELL_HEIGHT,
147  vel.z() * deltaT / CELL_DEPTH );
148  }
149 
150  // new spacial change calculation
151  if( TrajectoryInterpolation == NONLINEAR )
152  {
153  const float3_X vel_old = velocityCalc( mom_old, mass );
154  const float_X QoM = charge / mass;
155  const float_X B4 = B2 * B2;
156  float3_X r = pos;
157 
158  if( B4 > epsilon )
159  {
160  trigo_X sinres;
161  trigo_X cosres;
162  trigo_X arg = B * QoM * deltaT / SPEED_OF_LIGHT;
163  math::sincos( arg, sinres, cosres );
164 
165  r.x() = bField.x() * bField.x() * bField.x() * bField.x() * QoM
166  * ( eField.x() * QoM * deltaT * deltaT + 2.0f * ( deltaT * vel_old.x() + pos.x() ) );
167 
168 #define SUM_RLINE1(I,J,K) 2.0 * bField.J() * bField.J() * bField.J() * bField.J() * QoM * pos.x() \
169  + 2.0 * bField.J() * bField.J() * bField.K() * bField.K() * QoM * pos.x() \
170  + bField.J() * bField.J() * bField.J() * ( -levichivita(I,J,K) * 2.0 * SPEED_OF_LIGHT * ( eField.K() * QoM * deltaT + vel_old.K() ) + bField.I() * QoM * deltaT * ( eField.J() * QoM * deltaT + 2.0 * vel_old.J() ) ) \
171  + bField.J() * bField.J() * ( 2.0 * SPEED_OF_LIGHT * SPEED_OF_LIGHT * eField.I() + bField.I() * bField.I() * QoM * ( eField.I() * QoM * deltaT * deltaT + 2.0 * deltaT * vel_old.I() + 4.0 * pos.I() ) + levichivita(I,J,K) * 2.0 * SPEED_OF_LIGHT * bField.K() * vel_old.J() + bField.K() * QoM * ( levichivita(I,J,K) * 2.0 * eField.J() * SPEED_OF_LIGHT * deltaT + bField.I() * bField.K() * QoM * deltaT * deltaT ) ) \
172  + bField.I() * bField.J() * ( bField.I() * bField.I() * QoM * deltaT * ( eField.J() * QoM * deltaT + 2.0 * vel_old.J() ) - levichivita(I,J,K) * 2.0 * bField.I() * SPEED_OF_LIGHT * ( eField.K() * QoM * deltaT + vel_old.K() ) - 2.0 * SPEED_OF_LIGHT * SPEED_OF_LIGHT * eField.J() )
173 
174 #define SUM_RLINE2(I,J,K) - bField.J() * ( SPEED_OF_LIGHT * eField.I() * bField.J() - levichivita(I,J,K) * bField.J() * bField.J() * vel_old.K() - bField.I() * SPEED_OF_LIGHT * eField.J() - levichivita(I,J,K) * bField.J() * vel_old.K() * ( bField.I() * bField.I() + bField.K() *bField.K() ) )
175 
176 #define SUM_RLINE3(I,J,K) levichivita(I,J,K) * bField.J() * ( SPEED_OF_LIGHT * eField.K() + levichivita(I,J,K) * ( bField.J() * vel_old.I() - bField.I() * vel_old.J() ) )
177 
178  r.x() += FOR_JK_NOT_I( x, y, z, SUM_RLINE1 );
179  r.x() += float_X(cosres ) * 2.0 * SPEED_OF_LIGHT * ( FOR_JK_NOT_I( x, y, z, SUM_RLINE2 ) );
180  r.x() += float_X(sinres ) * 2.0 * SPEED_OF_LIGHT * B * ( FOR_JK_NOT_I( x, y, z, SUM_RLINE3 ) );
181 
182  r.y() += FOR_JK_NOT_I( y, z, x, SUM_RLINE1 );
183  r.y() += float_X(cosres ) * 2.0 * SPEED_OF_LIGHT * ( FOR_JK_NOT_I( y, z, x, SUM_RLINE2 ) );
184  r.y() += float_X(sinres ) * 2.0 * SPEED_OF_LIGHT * B * ( FOR_JK_NOT_I( y, z, x, SUM_RLINE3 ) );
185 
186  r.z() += FOR_JK_NOT_I( z, x, y, SUM_RLINE1 );
187  r.z() += float_X(cosres ) * 2.0 * SPEED_OF_LIGHT * ( FOR_JK_NOT_I( z, x, y, SUM_RLINE2 ) );
188  r.z() += float_X(sinres ) * 2.0 * SPEED_OF_LIGHT * B * ( FOR_JK_NOT_I( z, x, y, SUM_RLINE3 ) );
189 
190  r *= float_X(0.5) / B4 / QoM;
191  }
192  else
193  {
194  r += eField * QoM * deltaT * deltaT + vel_old * deltaT;
195  }
196  dr = r - pos;
197 
198  dr *= float3_X::create(1.0) / cellSize;
199 
200  }
201 
202  pos += dr;
203  }
