78  bool isSharedMemoryPool()
79  {
80 #if( PMACC_CUDA_ENABLED != 1 )
81  return true;
82 #else
83  size_t freeInternal = 0;
84  size_t freeAtStart = 0;
85 
86  getMemoryInfo(&freeAtStart);
87 
88  /* alloc 90%, since allocating 100% is a bit risky on a SoC-like device */
89  size_t allocSth = size_t( 0.9 * double(freeAtStart) );
90  uint8_t* c = new uint8_t[allocSth];
91  memset(c, 0, allocSth);
92 
93  getMemoryInfo(&freeInternal);
94  delete [] c;
95 
96  /* if we allocated 90% of available mem, we should have "lost" more
97  * than 50% of memory, even with fluctuations from the OS */
98  if( double(freeInternal)/double(freeAtStart) < 0.5 )
99  return true;
100 
101  return false;
102 #endif
