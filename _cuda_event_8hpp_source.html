71  bool CudaEvent::isFinished()
72  {
73  // avoid cuda driver calls if event is already finished
74  if( finished )
75  return true;
76  assert( isRecorded );
77 
78  cudaError_t rc = cudaEventQuery(event);
79 
80  if(rc == cudaSuccess)
81  {
82  finished = true;
83  return true;
84  }
85  else if(rc == cudaErrorNotReady)
86  return false;
87  else
88  PMACC_PRINT_CUDA_ERROR_AND_THROW(rc, "Event query failed");
89  }
