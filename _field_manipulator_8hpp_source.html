43  template<class BoxedMemory>
44  static void absorbBorder(uint32_t currentStep, MappingDesc &cellDescription, BoxedMemory deviceBox)
45  {
46  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(currentStep);
47  for (uint32_t i = 1; i < NumberOfExchanges<simDim>::value; ++i)
48  {
49  /* only call for planes: left right top bottom back front*/
50  if (FRONT % i == 0 && !(Environment<simDim>::get().GridController().getCommunicationMask().isSet(i)))
51  {
52  uint32_t direction = 0; /*set direction to X (default)*/
53  if (i >= BOTTOM && i <= TOP)
54  direction = 1; /*set direction to Y*/
55  if (i >= BACK)
56  direction = 2; /*set direction to Z*/
57 
58  /* exchange mod 2 to find positive or negative direction
59  * positive direction = 1
60  * negative direction = 0
61  */
62  uint32_t pos_or_neg = i % 2;
63 
64  uint32_t thickness = ABSORBER_CELLS[direction][pos_or_neg];
65  float_X absorber_strength = ABSORBER_STRENGTH[direction][pos_or_neg];
66 
67  if (thickness == 0) continue; /*if the absorber has no thickness we check the next side*/
68 
69  /* allow to enable the absorber on the top side if the laser
70  * initialization plane in y direction is *not* in cell zero
71  */
72  if (fields::laserProfiles::Selected::initPlaneY == 0)
73  {
74  /* disable the absorber on top side if
75  * no slide was performed and
76  * laser init time is not over
77  */
78  if (numSlides == 0 && ((currentStep * DELTA_T) <= fields::laserProfiles::Selected::INIT_TIME))
79  {
80  /* disable absorber on top side */
81  if (i == TOP) continue;
82  }
83  }
84 
85  /* if sliding window is active we disable absorber on bottom side*/
86  if (MovingWindow::getInstance().isSlidingWindowActive() && i == BOTTOM) continue;
87 
88  ExchangeMapping<GUARD, MappingDesc> mapper(cellDescription, i);
89  constexpr uint32_t numWorkers = pmacc::traits::GetNumWorkers<
90  pmacc::math::CT::volume< SuperCellSize >::type::value
91  >::value;
92 
93  PMACC_KERNEL( KernelAbsorbBorder< numWorkers> {} )(
94  mapper.getGridDim(),
95  numWorkers
96  )(
97  deviceBox,
98  thickness,
99  absorber_strength,
100  mapper
101  );
102  }
103  }
104  }

106  static pmacc::traits::StringProperty getStringProperties()
107  {
108  pmacc::traits::StringProperty propList;
109  const DataSpace<DIM3> periodic =
110  Environment<simDim>::get().EnvironmentController().getCommunicator().getPeriodic();
111 
112  for( uint32_t i = 1; i < NumberOfExchanges<simDim>::value; ++i )
113  {
114  // for each planar direction: left right top bottom back front
115  if( FRONT % i == 0 )
116  {
117  const std::string directionName = ExchangeTypeNames()[i];
118  const DataSpace<DIM3> relDir = Mask::getRelativeDirections<DIM3>(i);
119 
120  bool isPeriodic = false;
121  uint32_t axis = 0; // x(0) y(1) z(2)
122  uint32_t axisDir = 0; // negative (0), positive (1)
123  for( uint32_t d = 0; d < simDim; d++ )
124  {
125  if( relDir[d] * periodic[d] != 0 )
126  isPeriodic = true;
127  if( relDir[d] != 0 )
128  axis = d;
129  }
130  if( relDir[axis] > 0 )
131  axisDir = 1;
132 
133  std::string boundaryName = "open"; // absorbing boundary
134  if( isPeriodic )
135  boundaryName = "periodic";
136 
137  if( boundaryName == "open" )
138  {
139  std::ostringstream boundaryParam;
140  boundaryParam << "exponential damping over "
141  << ABSORBER_CELLS[axis][axisDir] << " cells";
142  propList[directionName]["param"] = boundaryParam.str();
143  }
144  else
145  {
146  propList[directionName]["param"] = "none";
147  }
148 
149  propList[directionName]["name"] = boundaryName;
150  }
151  }
152  return propList;
153  }
