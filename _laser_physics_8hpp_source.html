53  typename T_LaserFunctor
54  >

56  T_Acc const & acc,
57  T_LaserFunctor laserFunctor

61  using LaserFunctor = T_LaserFunctor;

63  PMACC_CASSERT_MSG(
64  __LaserPlaneSizeInSuperCell_y_must_be_less_or_equal_than_SuperCellSize_y,
65  LaserPlaneSizeInSuperCell::y::value <= SuperCellSize::y::value
66  );

73  // offset of the superCell (in cells, without any guards) to the origin of the local domain
74 
75  DataSpace< simDim > localSuperCellOffset = DataSpace< simDim >( blockIdx );
76 
77  // add not handled supercells from LaserFunctor::Unitless::initPlaneY
78  localSuperCellOffset.y() += LaserFunctor::Unitless::initPlaneY / SuperCellSize::y::value;

81 
82  mappings::threads::ForEachIdx<
83  mappings::threads::IdxConfig<
84  planeSize,
85  numWorkers
86  >
87  > { workerIdx }(
88  [&](
89  uint32_t const linearIdx,
90  uint32_t const
91  )

94  acc,
95  localSuperCellOffset,
96  mappings::threads::WorkerCfg< numWorkers >{ workerIdx }
97  );

100  DataSpace< simDim > cellIdxInSuperCell = DataSpaceOperations< simDim >::template map< LaserPlaneSizeInSuperCell >( linearIdx );

110  struct LaserPhysics
111  {
112  void operator()(uint32_t currentStep) const
113  {
114  /* The laser can be initialized in the plane of the first cell or
115  * any later x-z plane inside the simulation. Initializing the
116  * laser in planes inside the simulation corresponds to an
117  * evaluation of the field at negatively shifted time.
118  */
119  constexpr float_X laserTimeShift = laserProfiles::Selected::Unitless::initPlaneY * CELL_HEIGHT / SPEED_OF_LIGHT;
120 
121  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(currentStep);
122 
123  /* Disable laser if
124  * - init time of laser is over or
125  * - we have periodic boundaries in Y direction or
126  * - we already performed a slide
127  */
128  bool const laserNone = ( laserProfiles::Selected::Unitless::INIT_TIME == float_X(0.0) );
129  bool const laserInitTimeOver =
130  ( ( currentStep * DELTA_T - laserTimeShift ) >= laserProfiles::Selected::Unitless::INIT_TIME );
131  bool const topBoundariesArePeriodic =
132  ( Environment<simDim>::get().GridController().getCommunicationMask( ).isSet( TOP ) );
133  bool const boxHasSlided = ( numSlides != 0 );
134 
135  bool const disableLaser =
136  laserNone ||
137  laserInitTimeOver ||
138  topBoundariesArePeriodic ||
139  boxHasSlided;
140  if( !disableLaser )
141  {
142  PMACC_VERIFY_MSG(
143  laserProfiles::Selected::Unitless::initPlaneY < static_cast<uint32_t>( Environment<simDim>::get().SubGrid().getLocalDomain().size.y() ),
144  "initPlaneY must be located in the top GPU"
145  );
146 
147  PMACC_CASSERT_MSG(
148  __initPlaneY_needs_to_be_greate_than_the_top_absorber_cells_or_zero,
149  laserProfiles::Selected::Unitless::initPlaneY > ABSORBER_CELLS[1][0] ||
150  laserProfiles::Selected::Unitless::initPlaneY == 0 ||
151  laserProfiles::Selected::Unitless::INIT_TIME == float_X(0.0) /* laser is disabled e.g. laserNone */
152  );
153 
154  /* Calculate how many neighbors to the left we have
155  * to initialize the laser in the E-Field
156  *
157  * Example: Yee needs one neighbor to perform dB = curlE
158  * -> initialize in y=0 plane
159  * A second order solver could need 2 neighbors left:
160  * -> initialize in y=0 and y=1 plane
161  *
162  * Question: Why do other codes initialize the B-Field instead?
163  * Answer: Because our fields are defined on the lower cell side
164  * (C-Style ftw). Therefore, our curls (for example Yee)
165  * are shifted nabla+ <-> nabla- compared to Fortran codes
166  * (in other words: curlLeft <-> curlRight)
167  * for E and B.
168  * For this reason, we have to initialize E instead of B.
169  *
170  * Problem: that's still not our case. For example our Yee does a
171  * dE = curlLeft(B) - therefor, we should init B, too.
172  *
173  *
174  * @todo: might also lack temporal offset since our formulas are E(x,z,t) instead of E(x,y,z,t)
175  * `const int max_y_neighbors = Get<fields::Solver::OffsetOrigin_E, 1 >::value;`
176  *
177  * @todo Right now, the phase could be wrong ( == is cloned)
178  * @see LaserPhysics.hpp
179  *
180  * @todo What about the B-Field in the second plane?
181  *
182  */
183  constexpr int laserInitCellsInY = 1;
184 
185  using LaserPlaneSizeInSuperCells = typename pmacc::math::CT::AssignIfInRange<
186  typename SuperCellSize::vector_type,
187  bmpl::integral_c< uint32_t, 1 >, /* y direction */
188  bmpl::integral_c< int, laserInitCellsInY >
189  >::type;
190 
191  DataSpace< simDim > gridBlocks = Environment< simDim >::get().SubGrid().getLocalDomain().size / SuperCellSize::toRT();
192  // use the one supercell in y to initialize the laser plane
193  gridBlocks.y() = 1;
194 
195  constexpr uint32_t numWorkers = pmacc::traits::GetNumWorkers<
196  pmacc::math::CT::volume< LaserPlaneSizeInSuperCells >::type::value
197  >::value;
198 
199  PMACC_KERNEL(
200  KernelLaser<
201  numWorkers,
202  LaserPlaneSizeInSuperCells
203  >{}
204  )(
205  gridBlocks,
206  numWorkers
207  )(
208  laserProfiles::Selected( currentStep )
209  );
210  }
211  }
212  };
