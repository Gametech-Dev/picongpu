54 #define FOREACH_OPERATOR(Z, N, _) \
55  /* typename C0, typename C1, ... */ \
56  template<typename Zone, BOOST_PP_ENUM_PARAMS(N, typename C), typename Functor> \
57  /* C0 c0, C1 c1, ... */ \
58  void operator()(const Zone& p_zone, BOOST_PP_ENUM_BINARY_PARAMS(N, C, c), const Functor& functor) \
59  { \
60  /* C0 c0_shifted = c0(p_zone.offset); */ \
61  /* C1 c1_shifted = c1(p_zone.offset); */ \
62  /* ... */ \
63  BOOST_PP_REPEAT(N, SHIFT_CURSOR_ZONE, _) \
64  \
65  auto blockSize = BlockDim::toRT(); \
66  detail::SphericMapper<Zone::dim, BlockDim> mapper; \
67  using namespace pmacc; \
68  PMACC_KERNEL(detail::KernelForeach{})(mapper.cudaGridDim(p_zone.size), blockSize) \
69  /* c0_shifted, c1_shifted, ... */ \
70  (mapper, BOOST_PP_ENUM(N, SHIFTED_CURSOR, _), functor); \
71  }
