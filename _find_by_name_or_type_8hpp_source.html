43  /* find a type within a sequence by name or the type itself
44  *
45  * pmacc::traits::GetCTName is used to translate each element of
46  * T_MPLSeq into a name.
47  *
48  * @tparam T_MPLSeq source sequence where we search T_Identifier
49  * @tparam T_Identifier name or type to search
50  */
51  template<
52  typename T_MPLSeq,
53  typename T_Identifier,
54  typename T_KeyNotFoundPolicy = pmacc::errorHandlerPolicies::ThrowValueNotFound
55  >
56  struct FindByNameOrType
57  {
58  using KeyNotFoundPolicy = T_KeyNotFoundPolicy;
59 
60  template< typename T_Value >
61  struct HasTypeOrName
62  {
63  using type = bmpl::or_<
64  boost::is_same<
65  T_Identifier,
66  T_Value
67  >,
68  boost::is_same<
69  pmacc::traits::GetCTName_t< T_Value >,
70  T_Identifier
71  >
72  >;
73  };
74 
75  using FilteredSeq = typename bmpl::copy_if<
76  T_MPLSeq,
77  HasTypeOrName< bmpl::_1 >
78  >::type;
79 
80  using type = typename bmpl::if_<
81  bmpl::empty< FilteredSeq >,
82  bmpl::apply<
83  KeyNotFoundPolicy,
84  T_MPLSeq,
85  T_Identifier
86  >,
87  bmpl::front< FilteredSeq >
88  >::type::type;
89  };
90 
91  template<
92  typename T_MPLSeq,
93  typename T_Identifier,
94  typename T_KeyNotFoundPolicy = pmacc::errorHandlerPolicies::ThrowValueNotFound
95  >
96  using FindByNameOrType_t = typename FindByNameOrType<
97  T_MPLSeq,
98  T_Identifier,
99  T_KeyNotFoundPolicy
100  >::type;
