57  template<typename T_CurrentInterpolation, typename T_Dummy>
58  struct ConditionCheck<DirSplitting< T_CurrentInterpolation >, T_Dummy>
59  {
60  /* Directional Splitting conditions:
61  *
62  * using SI units to avoid round off errors
63  *
64  * The compiler is allowed to evaluate an expression those not depends on a template parameter
65  * even if the class is never instantiated. In that case static assert is always
66  * evaluated (e.g. with clang), this results in an error if the condition is false.
67  * http://www.boost.org/doc/libs/1_60_0/doc/html/boost_staticassert.html
68  *
69  * A workaround is to add a template dependency to the expression.
70  * `sizeof(ANY_TYPE) != 0` is always true and defers the evaluation.
71  */
72  PMACC_CASSERT_MSG(DirectionSplitting_Set_dX_equal_dt_times_c____check_your_gridConfig_param_file,
73  (SI::SPEED_OF_LIGHT_SI * SI::DELTA_T_SI) == SI::CELL_WIDTH_SI &&
74  (sizeof(T_Dummy) != 0));
75  PMACC_CASSERT_MSG(DirectionSplitting_use_cubic_cells____check_your_gridConfig_param_file,
76  SI::CELL_HEIGHT_SI == SI::CELL_WIDTH_SI &&
77  (sizeof(T_Dummy) != 0));
78 #if (SIMDIM == DIM3)
79  PMACC_CASSERT_MSG(DirectionSplitting_use_cubic_cells____check_your_gridConfig_param_file,
80  SI::CELL_DEPTH_SI == SI::CELL_WIDTH_SI &&
81  (sizeof(T_Dummy) != 0));
82 #endif
83  };

93  using namespace cursor::tools;

95 
96  auto gridSizeTwisted = twistComponents<OrientationTwist>(gridSize);
97 
98  /* twist components of the supercell */
99  using BlockDim = typename CT::TwistComponents<SuperCellSize, OrientationTwist>::type;
100 
101  algorithm::kernel::ForeachBlock<BlockDim> foreach;
102  foreach(zone::SphericZone<3>(pmacc::math::Size_t<3>(BlockDim::x::value, gridSizeTwisted.y(), gridSizeTwisted.z())),
103  cursor::make_NestedCursor(twistVectorFieldAxes<OrientationTwist>(cursorE)),
104  cursor::make_NestedCursor(twistVectorFieldAxes<OrientationTwist>(cursorB)),
