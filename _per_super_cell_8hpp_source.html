45 struct CountMakroParticle
46 {
47  template<
48  typename ParBox,
49  typename CounterBox,
50  typename Mapping,
51  typename T_Acc
52  >
53  DINLINE void operator()(
54  T_Acc const & acc,
55  ParBox parBox,
56  CounterBox counterBox,
57  Mapping mapper
58  ) const
59  {
60 
61  typedef MappingDesc::SuperCellSize SuperCellSize;
62  typedef typename ParBox::FrameType FrameType;
63  typedef typename ParBox::FramePtr FramePtr;
64 
65  const DataSpace<simDim> block(mapper.getSuperCellIndex(DataSpace<simDim > (blockIdx)));
66  /* counterBox has no guarding supercells*/
67  const DataSpace<simDim> counterCell = block - mapper.getGuardingSuperCells();
68 
69  const DataSpace<simDim > threadIndex(threadIdx);
70  const int linearThreadIdx = DataSpaceOperations<simDim>::template map<SuperCellSize > (threadIndex);
71 
72  PMACC_SMEM( acc, counterValue, uint64_cu );
73  PMACC_SMEM( acc, frame, FramePtr );
74 
75  if (linearThreadIdx == 0)
76  {
77  counterValue = 0;
78  frame = parBox.getLastFrame(block);
79  if (!frame.isValid())
80  {
81  counterBox(counterCell) = counterValue;
82  }
83  }
84  __syncthreads();
85  if (!frame.isValid())
86  return; //end kernel if we have no frames
87 
88  bool isParticle = frame[linearThreadIdx][multiMask_];
89 
90  while (frame.isValid())
91  {
92  if (isParticle)
93  {
94  atomicAdd(&counterValue, static_cast<uint64_cu> (1LU), ::alpaka::hierarchy::Blocks{});
95  }
96  __syncthreads();
97  if (linearThreadIdx == 0)
98  {
99  frame = parBox.getPreviousFrame(frame);
100  }
101  isParticle = true;
102  __syncthreads();
103  }
104 
105  if (linearThreadIdx == 0)
106  counterBox(counterCell) = counterValue;
107  }
108 };

288  void openH5File()
289  {

302  mpi_size = gc.getGpuNodes();
303 
304  splashMpiPos.set(0, 0, 0);
305  splashMpiSize.set(1, 1, 1);
306 
307  for (uint32_t i = 0; i < simDim; ++i)
308  {
309  splashMpiPos[i] = mpi_pos[i];
310  splashMpiSize[i] = mpi_size[i];
311  }
312 
313 
314  const uint32_t maxOpenFilesPerNode = 1;
315  dataCollector = new ParallelDomainCollector(
316  gc.getCommunicator().getMPIComm(),
317  gc.getCommunicator().getMPIInfo(),
318  splashMpiSize,
319  maxOpenFilesPerNode);
320  // set attributes for datacollector files
321  DataCollector::FileCreationAttr h5_attr;
322  h5_attr.enableCompression = false;
323  h5_attr.fileAccType = DataCollector::FAT_CREATE;
324  h5_attr.mpiPosition.set(splashMpiPos);
325  h5_attr.mpiSize.set(splashMpiSize);
326  }
327 
328 
329  // open datacollector
330  try
331  {
332  std::string filename = (foldername + std::string("/makroParticlePerSupercell"));
333  log<picLog::INPUT_OUTPUT > ("HDF5 open DataCollector with file: %1%") %
334  filename;
335  dataCollector->open(filename.c_str(), h5_attr);
336  }
337  catch (const DCException& e)
338  {
339  std::cerr << e.what() << std::endl;
340  throw std::runtime_error("Failed to open datacollector");
341  }
342  }
