148 template<

154 struct ManipulateDerive
155 {
156  using DestSpeciesType = pmacc::particles::compileTime::FindByNameOrType_t<
157  VectorAllSpecies,
158  T_DestSpeciesType
159  >;
160  using DestFrameType = typename DestSpeciesType::FrameType;
161  using SrcSpeciesType = pmacc::particles::compileTime::FindByNameOrType_t<
162  VectorAllSpecies,
163  T_SrcSpeciesType
164  >;
165  using SrcFrameType = typename SrcSpeciesType::FrameType;
166 
167  using DestFunctor = typename bmpl::apply1<
168  T_Manipulator,
169  DestSpeciesType
170  >::type;
171 
172  using SrcFilter = typename bmpl::apply1<
173  T_SrcFilter,
174  SrcSpeciesType
175  >::type;
176 
177  /* note: this is a FilteredManipulator with filter::All for
178  * destination species, users can filter the destination directly via if's
179  * in the T_Manipulator.
180  */
181  using FilteredManipulator = manipulators::IBinary< DestFunctor >;
182  using SrcFilterInterfaced = filter::IUnary< SrcFilter >;
183 
184  HINLINE void operator()( const uint32_t currentStep )
185  {
186  DataConnector &dc = Environment<>::get().DataConnector();
187  auto speciesPtr = dc.get< DestSpeciesType >( DestFrameType::getName(), true );
188  auto srcSpeciesPtr = dc.get< SrcSpeciesType >( SrcFrameType::getName(), true );
189 
190  FilteredManipulator filteredManipulator( currentStep );
191  SrcFilterInterfaced srcFilter( currentStep );
192 
193  speciesPtr->deviceDeriveFrom( *srcSpeciesPtr, filteredManipulator, srcFilter );
194 
195  dc.releaseData( DestFrameType::getName() );
196  dc.releaseData( SrcFrameType::getName() );
197  }
198 };
